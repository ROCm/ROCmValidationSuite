119 bool gst_action::do_gpu_stress_test(map<int, uint16_t> gst_gpus_device_index) {
120  size_t k = 0;
121  for (;;) {
122  unsigned int i = 0;
123  if (property_wait != 0) // delay gst execution
124  sleep(property_wait);
125 
126  vector<GSTWorker> workers(gst_gpus_device_index.size());
127 
128  map<int, uint16_t>::iterator it;
129 
130  // all worker instances have the same json settings
131  GSTWorker::set_use_json(bjson);
132 
133  for (it = gst_gpus_device_index.begin();
134  it != gst_gpus_device_index.end(); ++it) {
135  // set worker thread stress test params
136  workers[i].set_name(action_name);
137  workers[i].set_gpu_id(it->second);
138  workers[i].set_gpu_device_index(it->first);
139  workers[i].set_run_wait_ms(property_wait);
140  workers[i].set_run_duration_ms(property_duration);
141  workers[i].set_ramp_interval(gst_ramp_interval);
142  workers[i].set_log_interval(property_log_interval);
143  workers[i].set_max_violations(gst_max_violations);
144  workers[i].set_copy_matrix(gst_copy_matrix);
145  workers[i].set_target_stress(gst_target_stress);
146  workers[i].set_tolerance(gst_tolerance);
147  workers[i].set_gst_hot_calls(gst_hot_calls);
148  workers[i].set_matrix_size_a(gst_matrix_size_a);
149  workers[i].set_matrix_size_b(gst_matrix_size_b);
150  workers[i].set_matrix_size_c(gst_matrix_size_c);
151  workers[i].set_gst_ops_type(gst_ops_type);
152  workers[i].set_matrix_transpose_a(gst_trans_a);
153  workers[i].set_matrix_transpose_b(gst_trans_b);
154  workers[i].set_alpha_val(gst_alpha_val);
155  workers[i].set_beta_val(gst_beta_val);
156  workers[i].set_lda_offset(gst_lda_offset);
157  workers[i].set_ldb_offset(gst_ldb_offset);
158  workers[i].set_ldc_offset(gst_ldc_offset);
159 
160  i++;
161  }
162 
163  if (property_parallel) {
164  for (i = 0; i < gst_gpus_device_index.size(); i++)
165  workers[i].start();
166 
167  // join threads
168  for (i = 0; i < gst_gpus_device_index.size(); i++)
169  workers[i].join();
170  } else {
171  for (i = 0; i < gst_gpus_device_index.size(); i++) {
172  workers[i].start();
173  workers[i].join();
174 
175  // check if stop signal was received
176  if (rvs::lp::Stopping())
177  return false;
178  }
179  }
180 
181  // check if stop signal was received
182  if (rvs::lp::Stopping())
183  return false;
184 
185  if (property_count != 0) {
186  k++;
187  if (k == property_count)
188  break;
189  }
190  }
191 
192  return rvs::lp::Stopping() ? false : true;
193 }

435 int gst_action::get_num_amd_gpu_devices(void) {
436  int hip_num_gpu_devices;
437  string msg;
438 
439  hipGetDeviceCount(&hip_num_gpu_devices);
440  if (hip_num_gpu_devices == 0) { // no AMD compatible GPU

442  rvs::lp::Log(msg, rvs::logerror);
443 
444  if (bjson) {

446  unsigned int usec;
447  rvs::lp::get_ticks(&sec, &usec);

451  // log the error

459  }
460  return 0;
461  }

526 int gst_action::run(void) {
527  string msg;
528 
529  // get the action name
530  if (property_get(RVS_CONF_NAME_KEY, &action_name)) {
531  rvs::lp::Err("Action name missing", MODULE_NAME_CAPS);
532  return -1;
533  }
534 
535  // check for -j flag (json logging)
536  if (property.find("cli.-j") != property.end())
537  bjson = true;
538 
539  if (!get_all_common_config_keys())
540  return -1;
541  if (!get_all_gst_config_keys())
542  return -1;
543 
544  if (property_duration > 0 && (property_duration < gst_ramp_interval)) {
545  msg = "'" +
546  std::string(RVS_CONF_DURATION_KEY) + "' cannot be less than '" +
547  std::string(RVS_CONF_RAMP_INTERVAL_KEY) + "'";
548  rvs::lp::Err(msg, MODULE_NAME_CAPS, action_name);
549  return -1;
550  }
551  if(bjson){
552  // add prelims for each action, dtype and target stress
553  json_add_primary_fields();
554  }
555  auto res = get_all_selected_gpus();
556  if(bjson){
557  rvs::lp::JsonActionEndNodeCreate();
558  }
559  return res;
560 }
