189  gst_target_stress = std::stof(it->second);

191  } catch (const std::regex_error& e) {
192  *error = 1; // something went wrong with the regex
193  }
194  } else {
195  *error = 1;
196  }

246 bool action::do_gpu_stress_test(map<int, uint16_t> gst_gpus_device_index) {
247  size_t k = 0;
248  for (;;) {
249  unsigned int i = 0;
250  if (gst_run_wait_ms != 0) // delay gst execution
251  sleep(gst_run_wait_ms);
252 
253  vector<GSTWorker> workers(gst_gpus_device_index.size());
254 
255  map<int, uint16_t>::iterator it;
256 
257  // all worker instances have the same json settings
258  GSTWorker::set_use_json(bjson);
259 
260  for (it = gst_gpus_device_index.begin();
261  it != gst_gpus_device_index.end(); ++it) {
262  // set worker thread stress test params
263  workers[i].set_name(action_name);
264  workers[i].set_gpu_id(it->second);
265  workers[i].set_gpu_device_index(it->first);
266  workers[i].set_run_wait_ms(gst_run_wait_ms);
267  workers[i].set_run_duration_ms(gst_run_duration_ms);
268  workers[i].set_ramp_interval(gst_ramp_interval);
269  workers[i].set_log_interval(gst_log_interval);
270  workers[i].set_max_violations(gst_max_violations);
271  workers[i].set_copy_matrix(gst_copy_matrix);
272  workers[i].set_target_stress(gst_target_stress);
273  workers[i].set_tolerance(gst_tolerance);
274  workers[i].set_matrix_size(gst_matrix_size);
275  i++;
276  }
277 
278  if (gst_runs_parallel) {
279  for (i = 0; i < gst_gpus_device_index.size(); i++)
280  workers[i].start();
281 
282  // join threads
283  for (i = 0; i < gst_gpus_device_index.size(); i++)
284  workers[i].join();
285  } else {
286  for (i = 0; i < gst_gpus_device_index.size(); i++) {
287  workers[i].start();
288  workers[i].join();
289 
290  // check if stop signal was received
291  if (rvs::lp::Stopping())
292  return false;
293  }
294  }
295 
296  // check if stop signal was received
297  if (rvs::lp::Stopping())
298  return false;
299 
300  if (gst_run_count != 0) {
301  k++;
302  if (k == gst_run_count)
303  break;
304  }
305  }
306 
307  return rvs::lp::Stopping() ? false : true;
308 }

452 int action::get_num_amd_gpu_devices(void) {

455

459  log(msg.c_str(), rvs::logerror);

467  if (!json_root_node) {
468  // log the error
469  string msg = action_name + " " + MODULE_NAME + " "
470  + JSON_CREATE_NODE_ERROR;
471  cerr << "RVS-GST: " << msg;
472  return -1;
473  }
474 
475  rvs::lp::AddString(json_root_node, "ERROR", GST_NO_COMPATIBLE_GPUS);
476  rvs::lp::LogRecordFlush(json_root_node);
477  }
478  return 0;
479  }
480  return hip_num_gpu_devices;
481 }

487 int action::get_all_selected_gpus(void) {
488  int hip_num_gpu_devices;
489  bool amd_gpus_found = false;
490  map<int, uint16_t> gst_gpus_device_index;
491 
492  hip_num_gpu_devices = get_num_amd_gpu_devices();
493  if (hip_num_gpu_devices < 1)
494  return hip_num_gpu_devices;
495 
496  // iterate over all available & compatible AMD GPUs
497  for (int i = 0; i < hip_num_gpu_devices; i++) {
498  // get GPU device properties
499  hipDeviceProp_t props;
500  hipGetDeviceProperties(&props, i);
501 
502  // compute device location_id (needed in order to identify this device
503  // in the gpus_id/gpus_device_id list
504  unsigned int dev_location_id =
505  ((((unsigned int) (props.pciBusID)) << 8) | (props.pciDeviceID));
506 
507  int32_t devId =
508  rvs::gpulist::GetDeviceIdFromLocationId(dev_location_id);
509  if (-1 == devId) {
510  continue;
511  }
512 
513  // check for deviceid filtering
514  if (!device_id_filtering ||
515  (device_id_filtering && static_cast<uint16_t>(devId)
516  == deviceid)) {
517  // check if this GPU is part of the GPU stress test
518  // (device = "all" or the gpu_id is in the device: <gpu id> list)
519  bool cur_gpu_selected = false;
520  uint16_t gpu_id = rvs::gpulist::GetGpuId(dev_location_id);
521  if (device_all_selected) {
522  cur_gpu_selected = true;
523  } else {
524  // search for this gpu in the list
525  // provided under the <device> property
526  vector<string>::iterator it_gpu_id = find(
527  device_prop_gpu_id_list.begin(),
528  device_prop_gpu_id_list.end(),
529  std::to_string(gpu_id));
530 
531  if (it_gpu_id != device_prop_gpu_id_list.end())
532  cur_gpu_selected = true;
533  }
534 
535  if (cur_gpu_selected) {
536  gst_gpus_device_index.insert
537  (std::pair<int, uint16_t>(i, gpu_id));
538  amd_gpus_found = true;
539  }
540  }
541  }
542 
543  if (amd_gpus_found) {
544  if (do_gpu_stress_test(gst_gpus_device_index))
545  return 0;
546 
547  return -1;
548  }
549 
550  return 0;
551 }

557 int action::run(void) {
558  string msg;
559  int error;
560 
561  // get the action name
562  rvs::actionbase::property_get_action_name(&error);
563  if (error == 2) {
564  msg = "action name field is missing in gst module";
565  cerr << "RVS-GST: " << msg;
566  return -1;
567  }
568 
569  device_all_selected = false;
570  device_id_filtering = false;
571 
572  // check for -j flag (json logging)
573  if (property.find("cli.-j") != property.end())
574  bjson = true;
575 
576  if (!get_all_common_config_keys())
577  return -1;
578  if (!get_all_gst_config_keys())
579  return -1;
580 
581  if (gst_run_duration_ms > 0 && (gst_run_duration_ms < gst_ramp_interval)) {
582  cerr << "RVS-GST: action: " << action_name << " '" <<
583  RVS_CONF_DURATION_KEY << "' cannot be less than '" <<
584  RVS_CONF_RAMP_INTERVAL_KEY << "'" << std::endl;
585  return -1;
586  }
587 
588  return get_all_selected_gpus();
