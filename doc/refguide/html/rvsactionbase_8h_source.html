70  int property_get_uint_list(const std::string& key,
71  const std::string& delimiter,
72  std::vector<T>* pval,
73  bool* pball) {
74  std::string strval;
75 
76  // fetch key value if any
77  if (!has_property(key, &strval)) {
78  return 2;
79  }
80 
81  // found and is "all" - set flag and return
82  if (strval == "all") {
83  *pball = true;
84  pval->clear();

87  *pball = false;
88  }

92 
93  // convert str arary into uint16_t array
94  int sts = rvs_util_strarr_to_uintarr<T>(strarray, pval);
95

98  return 1;
99  }
100 
101  return 0;
102  }
