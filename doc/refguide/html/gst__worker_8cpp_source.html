108 void GSTWorker::hit_max_gflops(int *error, string *err_description) {
109  std::chrono::time_point<std::chrono::system_clock> gst_start_time,
110  gst_end_time,
111  gst_log_interval_time;
112  double seconds_elapsed = 0, curr_gflops;
113  uint16_t num_sgemm_ops_log_interval = 0;
114  uint64_t millis_sgemm_ops;
115  string msg;
116 
117  *error = 0;
118  gst_start_time = std::chrono::system_clock::now();
119  gst_log_interval_time = std::chrono::system_clock::now();
120 
121  for (;;) {
122  // check if stop signal was received
123  if (rvs::lp::Stopping())
124  break;
125 
126  gst_end_time = std::chrono::system_clock::now();
127  if (time_diff(gst_end_time, gst_start_time) >=
128  NMAX_MS_GPU_RUN_PEAK_PERFORMANCE)
129  break;
130 
131  if (copy_matrix) {
132  // copy matrix before each GEMM
133  if (!gpu_blas->copy_data_to_gpu()) {
134  *error = 1;
135  *err_description = GST_BLAS_MEMCPY_ERROR;
136  return;
137  }
138  }
139 
140  // run GEMM & wait for completion
141  if (!gpu_blas->run_blass_gemm())
142  continue; // failed to run the current SGEMM
143 
144  while (!gpu_blas->is_gemm_op_complete()) {}
145 
146  num_sgemm_ops_log_interval++;
147 
148  gst_end_time = std::chrono::system_clock::now();
149  millis_sgemm_ops = time_diff(gst_end_time, gst_log_interval_time);
150 
151  if (millis_sgemm_ops >= log_interval) {
152  // compute the GFLOPS
153  seconds_elapsed = static_cast<double> (millis_sgemm_ops) / 1000;
154  if (seconds_elapsed != 0) {
155  curr_gflops = static_cast<double>(gpu_blas->gemm_gflop_count() *
156  num_sgemm_ops_log_interval) / seconds_elapsed;
157  log_interval_gflops(curr_gflops);
158  }
159 
160  num_sgemm_ops_log_interval = 0;
161  gst_log_interval_time = std::chrono::system_clock::now();
162  }
163  }
164 }

174 bool GSTWorker::do_gst_ramp(int *error, string *err_description) {
175  std::chrono::time_point<std::chrono::system_clock> gst_start_time,
176  gst_end_time,
177  gst_log_interval_time,
178  gst_start_gflops_time,
179  gst_last_sgemm_start_time,
180  gst_last_sgemm_end_time;
181  double seconds_elapsed, curr_gflops, dyn_delay_target_stress;
182  uint16_t num_sgemm_ops = 0, num_sgemm_ops_log_interval = 0;
183  uint64_t millis_sgemm_ops, millis_last_sgemm;
184  uint16_t proc_delay = 0;
185  string msg;
186 
187  // make sure that the ramp_interval & duration are not less than
188  // NMAX_MS_GPU_RUN_PEAK_PERFORMANCE (e.g.: 1000)
189  if (run_duration_ms < NMAX_MS_GPU_RUN_PEAK_PERFORMANCE)
190  run_duration_ms += NMAX_MS_GPU_RUN_PEAK_PERFORMANCE;
191  if (ramp_interval < NMAX_MS_GPU_RUN_PEAK_PERFORMANCE)
192  ramp_interval += NMAX_MS_GPU_RUN_PEAK_PERFORMANCE;
193 
194  // stage 1. setup rvs blas
195  setup_blas(error, err_description);
196  if (*error)
197  return false;
198 
199  // stage 2. run SGEMM with the given matrix_size for about
200  // NMAX_MS_GPU_RUN_PEAK_PERFORMANCE ms (e.g.: 1000) in order to hit the
201  // maximum Gflops value
202  hit_max_gflops(error, err_description);
203  if (*error)
204  return false;
205 
206  // check if stop signal was received
207  if (rvs::lp::Stopping())
208  return false;
209 
210  // stage 3. reduce the SGEMM frequency and try to achieve the desired Gflops
211  // the delay which gives the SGEMM frequency will be dynamically computed
212  delay_target_stress = 0;
213 
214  gst_start_time = std::chrono::system_clock::now();
215  gst_log_interval_time = std::chrono::system_clock::now();
216  gst_start_gflops_time = std::chrono::system_clock::now();
217 
218  for (;;) {
219  // check if stop signal was received
220  if (rvs::lp::Stopping())
221  return false;
222 
223  gst_end_time = std::chrono::system_clock::now();
224  if (time_diff(gst_end_time, gst_start_time) >
225  ramp_interval - NMAX_MS_GPU_RUN_PEAK_PERFORMANCE)
226  return false;
227 
228  gst_last_sgemm_start_time = std::chrono::system_clock::now();
229 
230  if (copy_matrix) {
231  // copy matrix before each GEMM
232  if (!gpu_blas->copy_data_to_gpu()) {
233  *error = 1;
234  *err_description = GST_BLAS_MEMCPY_ERROR;
235  return false;
236  }
237  }
238 
239  // run GEMM & wait for completion
240  if (!gpu_blas->run_blass_gemm())
241  continue; // failed to run the current SGEMM
242  while (!gpu_blas->is_gemm_op_complete()) {}
243  gst_last_sgemm_end_time = std::chrono::system_clock::now();
244  millis_last_sgemm =
245  time_diff(gst_last_sgemm_end_time, gst_last_sgemm_start_time);
246  if (static_cast<double>(
247  (1000 * gpu_blas->gemm_gflop_count()) /
248  target_stress) <
249  millis_last_sgemm) {
250  // last SGEMM timed-out (it took more than it should)
251  dyn_delay_target_stress = 1;
252  } else {
253  dyn_delay_target_stress =
254  static_cast<double>((
255  1000 * gpu_blas->gemm_gflop_count()) /
256  (target_stress + target_stress * tolerance / 2)) -
257  millis_last_sgemm;
258  if (dyn_delay_target_stress > proc_delay + 1)
259  dyn_delay_target_stress -= proc_delay;
260  }
261 
262  if (dyn_delay_target_stress > 1) {
263  usleep_ex(dyn_delay_target_stress * 1000);
264  delay_target_stress += dyn_delay_target_stress;
265  }
266 
267  num_sgemm_ops++;
268  num_sgemm_ops_log_interval++;
269 
270  gst_end_time = std::chrono::system_clock::now();
271  millis_sgemm_ops =
272  time_diff(gst_end_time, gst_start_gflops_time);
273  if (millis_sgemm_ops >= NMAX_MS_SGEMM_OPS_RAMP_SUB_INTERVAL) {
274  // compute the GFLOPS
275  seconds_elapsed = static_cast<double>
276  (millis_sgemm_ops) / 1000;
277  if (seconds_elapsed > 0) {
278  curr_gflops = static_cast<double>(
279  gpu_blas->gemm_gflop_count() *
280  num_sgemm_ops) / seconds_elapsed;
281  if (curr_gflops >= target_stress && curr_gflops <
282  target_stress + target_stress * tolerance/2) {
283  ramp_actual_time =
284  time_diff(gst_end_time, gst_start_time) +
285  NMAX_MS_GPU_RUN_PEAK_PERFORMANCE;
286  delay_target_stress /= num_sgemm_ops;
287  return true;
288  }
289  }
290  proc_delay +=
291  (delay_target_stress * PROC_DEC_INC_SGEMM_FREQ_DELAY) / 100;
292  num_sgemm_ops = 0;
293  delay_target_stress = 0;
294  gst_start_gflops_time = std::chrono::system_clock::now();
295  }
296 
297  millis_sgemm_ops =
298  time_diff(gst_end_time, gst_log_interval_time);
299  if (millis_sgemm_ops >= log_interval) {
300  // compute the GFLOPS
301  seconds_elapsed = static_cast<double>
302  (millis_sgemm_ops) / 1000;
303  if (seconds_elapsed > 0) {
304  curr_gflops = static_cast<double>(
305  gpu_blas->gemm_gflop_count() *
306  num_sgemm_ops_log_interval) / seconds_elapsed;
307  log_interval_gflops(curr_gflops);
308  }
309 
310  num_sgemm_ops_log_interval = 0;
311  gst_log_interval_time = std::chrono::system_clock::now();
312  }
313  }
314 
315  return false;
316 }

362 bool GSTWorker::do_gst_stress_test(int *error, std::string *err_description) {
363  uint16_t num_sgemm_ops = 0, num_gflops_violations = 0;
364  uint64_t total_milliseconds, log_interval_milliseconds;
365  double seconds_elapsed, gflops_interval;
366  string msg;
367  std::chrono::time_point<std::chrono::system_clock> gst_start_time,
368  gst_end_time, gst_log_interval_time;
369 
370  *error = 0;
371  max_gflops = 0;
372  num_sgemm_ops = 0;
373 
374  gst_start_time = std::chrono::system_clock::now();
375  gst_log_interval_time = std::chrono::system_clock::now();
376 
377  for (;;) {
378  // check if stop signal was received
379  if (rvs::lp::Stopping())
380  return false;
381 
382  if (copy_matrix) {
383  // copy matrix before each GEMM
384  if (!gpu_blas->copy_data_to_gpu()) {
385  *error = 1;
386  *err_description = GST_BLAS_MEMCPY_ERROR;
387  return false;
388  }
389  }
390 
391  bool sgemm_success = true;
392  // run GEMM & wait for completion
393  if (gpu_blas->run_blass_gemm()) {
394  while (!gpu_blas->is_gemm_op_complete()) {}
395  } else {
396  sgemm_success = false;
397  }
398 
399  if (sgemm_success)
400  num_sgemm_ops++;
401 
402  usleep_ex(delay_target_stress * 1000);
403  gst_end_time = std::chrono::system_clock::now();
404  total_milliseconds = time_diff(gst_end_time, gst_start_time);
405  log_interval_milliseconds = time_diff(gst_end_time,
406  gst_log_interval_time);
407 
408  if (log_interval_milliseconds >= log_interval && num_sgemm_ops > 0) {
409  seconds_elapsed = static_cast<double> (log_interval_milliseconds) /
410  1000;
411  if (seconds_elapsed != 0) {
412  gflops_interval =
413  static_cast<double>(gpu_blas->gemm_gflop_count()
414  * num_sgemm_ops) / seconds_elapsed;
415  if (gflops_interval > max_gflops)
416  max_gflops = gflops_interval;
417 
418  log_interval_gflops(gflops_interval);
419  if (check_gflops_violation(gflops_interval))
420  num_gflops_violations++;
421 
422  // reset time & gflops related data
423  num_sgemm_ops = 0;
424  gst_log_interval_time = std::chrono::system_clock::now();
425  }
426  }
427 
428  if (total_milliseconds >= run_duration_ms - ramp_actual_time)
429  break;
430  }
431 
432  if (num_gflops_violations > max_violations)
433  return false;
434 
435  return true;
436 }

441 void GSTWorker::run() {
442  string msg, err_description;
443  int error = 0;
444  bool gst_test_passed;
445 
446  max_gflops = 0;
447 
448  // log GST stress test - start message
449  msg = "[" + action_name + "] " + MODULE_NAME + " " +
450  std::to_string(gpu_id) + " " + GST_START_MSG + " " +
451  std::to_string(target_stress) + " " + GST_COPY_MATRIX_MSG + ":" +
452  (copy_matrix ? "true":"false");
453  log(msg.c_str(), rvs::loginfo);
454 
455  log_to_json(GST_START_MSG, std::to_string(target_stress), rvs::loginfo);
456  log_to_json(GST_COPY_MATRIX_MSG, (copy_matrix ? "true":"false"),
457  rvs::loginfo);
458 
459  // let the GPU ramp-up and check the result
460  bool ramp_up_success = do_gst_ramp(&error, &err_description);
461 
462  // GPU was not able to do the processing (HIP/rocBlas error(s) occurred)
463  if (error) {
464  string msg = "[" + action_name + "] " + MODULE_NAME + " "
465  + std::to_string(gpu_id) + " " + err_description;
466  log(msg.c_str(), rvs::logerror);
467  log_to_json("err", err_description, rvs::logerror);
468 
469  return;
470  }
471 
472  if (!ramp_up_success) {
473  // check if stop signal was received
474  if (rvs::lp::Stopping())
475  return;
476 
477  // the selected GPU was not able to achieve the target_stress GFLOPS
478  msg = "[" + action_name + "] " + MODULE_NAME + " " +
479  std::to_string(gpu_id) + " " + GST_RAMP_EXCEEDED_MSG + " " +
480  std::to_string(ramp_interval);
481  log(msg.c_str(), rvs::loginfo);
482  log_to_json(GST_RAMP_EXCEEDED_MSG, std::to_string(ramp_interval),
483  rvs::loginfo);
484  gst_test_passed = false;
485  } else {
486  // the GPU succeeded to achieve the target_stress GFLOPS
487  // continue with the same workload for the rest of the test duration
488  msg = "[" + action_name + "] " + MODULE_NAME + " " +
489  std::to_string(gpu_id) + " " + GST_TARGET_ACHIEVED_MSG + " " +
490  std::to_string(target_stress);
491  log(msg.c_str(), rvs::loginfo);
492  log_to_json(GST_TARGET_ACHIEVED_MSG, std::to_string(target_stress),
493  rvs::loginfo);
494  if (run_duration_ms > 0) {
495  gst_test_passed = do_gst_stress_test(&error, &err_description);
496  // check if stop signal was received
497  if (rvs::lp::Stopping())
498  return;
499 
500  if (error) {
501  // GPU didn't complete the test (HIP/rocBlas error(s) occurred)
502  string msg = "[" + action_name + "] " + MODULE_NAME + " " +
503  std::to_string(gpu_id) + " " + err_description;
504  log(msg.c_str(), rvs::logerror);
505  log_to_json("err", err_description, rvs::logerror);
506  return;
507  }
508  }
509  }
510 
511  log_gst_test_result(gst_test_passed);
512 }

518 void GSTWorker::log_gst_test_result(bool gst_test_passed) {
519  string msg;
520 
521  double flops_per_op = (2 * (static_cast<double>(gpu_blas->get_m())/1000) *
522  (static_cast<double>(gpu_blas->get_n())/1000) *
523  (static_cast<double>(gpu_blas->get_k())/1000));
524  msg = "[" + action_name + "] " + MODULE_NAME + " " +
525  std::to_string(gpu_id) + " " + GST_MAX_GFLOPS_OUTPUT_KEY + ": " +
526  std::to_string(max_gflops) + " " + GST_FLOPS_PER_OP_OUTPUT_KEY + ": " +
527  std::to_string(flops_per_op) + "x1e9" + " " +
528  GST_BYTES_COPIED_PER_OP_OUTPUT_KEY + ": " +
529  std::to_string(gpu_blas->get_bytes_copied_per_op()) +
530  " " + GST_TRY_OPS_PER_SEC_OUTPUT_KEY + ": "+
531  std::to_string(target_stress / gpu_blas->gemm_gflop_count()) +
532  " " + GST_PASS_KEY + ": " +
533  (gst_test_passed ? GST_RESULT_PASS_MESSAGE : GST_RESULT_FAIL_MESSAGE);
534  log(msg.c_str(), rvs::logresults);
535 
536  log_to_json(GST_MAX_GFLOPS_OUTPUT_KEY, std::to_string(max_gflops),
537  rvs::loginfo);
538  log_to_json(GST_FLOPS_PER_OP_OUTPUT_KEY, std::to_string(flops_per_op) +
539  "x1e9", rvs::loginfo);
540  log_to_json(GST_BYTES_COPIED_PER_OP_OUTPUT_KEY,
541  std::to_string(gpu_blas->get_bytes_copied_per_op()),
542  rvs::loginfo);
543  log_to_json(GST_TRY_OPS_PER_SEC_OUTPUT_KEY,
544  std::to_string(target_stress / gpu_blas->gemm_gflop_count()),
545  rvs::loginfo);
546  log_to_json(GST_PASS_KEY, (gst_test_passed ?
547  GST_RESULT_PASS_MESSAGE : GST_RESULT_FAIL_MESSAGE),
548  rvs::logresults);
549 }

593 void GSTWorker::usleep_ex(uint64_t microseconds) {
594  uint64_t total_microseconds = microseconds;
595  for (;;) {
596  if (total_microseconds > USLEEP_MAX_VAL) {
597  usleep(USLEEP_MAX_VAL);
598  total_microseconds -= USLEEP_MAX_VAL;
599  } else {
600  usleep(total_microseconds);
601  return;
602  }
603  }
