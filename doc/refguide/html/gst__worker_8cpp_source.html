112 void GSTWorker::hit_max_gflops(int *error, string *err_description) {
113  std::chrono::time_point<std::chrono::system_clock> gst_start_time,
114  gst_end_time,
115  gst_log_interval_time;
116  double seconds_elapsed = 0, curr_gflops;
117  uint16_t num_sgemm_ops_log_interval = 0;
118  uint64_t millis_sgemm_ops;
119  string msg;
120 
121  *error = 0;
122  gst_start_time = std::chrono::system_clock::now();
123  gst_log_interval_time = std::chrono::system_clock::now();
124 
125  for (;;) {
126  // check if stop signal was received
127  if (rvs::lp::Stopping())
128  break;
129 
130  gst_end_time = std::chrono::system_clock::now();
131  if (time_diff(gst_end_time, gst_start_time) >=
132  NMAX_MS_GPU_RUN_PEAK_PERFORMANCE)
133  break;
134 
135  if (copy_matrix) {
136  // copy matrix before each GEMM
137  if (!gpu_blas->copy_data_to_gpu(gst_ops_type)) {
138  *error = 1;
139  *err_description = GST_BLAS_MEMCPY_ERROR;
140  return;
141  }
142  }
143 
144  // run GEMM & wait for completion
145  if (!gpu_blas->run_blass_gemm(gst_ops_type) )
146  continue; // failed to run the current SGEMM
147 
148  while (!gpu_blas->is_gemm_op_complete()) {}
149 
150  num_sgemm_ops_log_interval++;
151 
152  gst_end_time = std::chrono::system_clock::now();
153  millis_sgemm_ops = time_diff(gst_end_time, gst_log_interval_time);
154  if (millis_sgemm_ops >= log_interval) {
155  // compute the GFLOPS
156  seconds_elapsed = static_cast<double> (millis_sgemm_ops) / 1000;
157  if (seconds_elapsed != 0) {
158  curr_gflops = static_cast<double>(gpu_blas->gemm_gflop_count() *
159  num_sgemm_ops_log_interval) / seconds_elapsed;
160  log_interval_gflops(curr_gflops);
161  }
162 
163  num_sgemm_ops_log_interval = 0;
164  gst_log_interval_time = std::chrono::system_clock::now();
165  }
166  }

177 bool GSTWorker::do_gst_ramp(int *error, string *err_description) {
178  std::chrono::time_point<std::chrono::system_clock> gst_start_time,
179  gst_end_time,
180  gst_log_interval_time,
181  gst_start_gflops_time,
182  gst_last_sgemm_start_time,
183  gst_last_sgemm_end_time;
184  double seconds_elapsed, curr_gflops, dyn_delay_target_stress;
185  uint16_t num_sgemm_ops = 0, num_sgemm_ops_log_interval = 0;
186  uint64_t millis_sgemm_ops, millis_last_sgemm;
187  uint16_t proc_delay = 0;
188  uint64_t start_time, end_time;
189  double timetakenforoneiteration, gflops_interval;
190  string msg;
191 
192  // make sure that the ramp_interval & duration are not less than
193  // NMAX_MS_GPU_RUN_PEAK_PERFORMANCE (e.g.: 1000)
194  if (run_duration_ms < NMAX_MS_GPU_RUN_PEAK_PERFORMANCE)
195  run_duration_ms += NMAX_MS_GPU_RUN_PEAK_PERFORMANCE;
196 
197  if (ramp_interval < NMAX_MS_GPU_RUN_PEAK_PERFORMANCE)
198  ramp_interval += NMAX_MS_GPU_RUN_PEAK_PERFORMANCE;
199 
200  // stage 1. setup rvs blas
201  setup_blas(error, err_description);
202  if (*error)
203  return false;
204 
205  // check if stop signal was received
206  if (rvs::lp::Stopping())
207  return false;
208 
209  // stage 3. reduce the SGEMM frequency and try to achieve the desired Gflops
210  // the delay which gives the SGEMM frequency will be dynamically computed
211  delay_target_stress = 0;
212 
213  gst_start_time = std::chrono::system_clock::now();
214  gst_log_interval_time = std::chrono::system_clock::now();
215  gst_start_gflops_time = std::chrono::system_clock::now();
216 
217  for (;;) {
218  // check if stop signal was received
219  if (rvs::lp::Stopping())
220  return false;
221 
222  gst_end_time = std::chrono::system_clock::now();
223  if (time_diff(gst_end_time, gst_start_time) >
224  ramp_interval - NMAX_MS_GPU_RUN_PEAK_PERFORMANCE)
225  return false;
226 
227  gst_last_sgemm_start_time = std::chrono::system_clock::now();
228 
229  if (copy_matrix) {
230  // Genrate random matrix data
231  gpu_blas->generate_random_matrix_data();
232  // copy matrix before each GEMM
233  if (!gpu_blas->copy_data_to_gpu(gst_ops_type)) {
234  *error = 1;
235  *err_description = GST_BLAS_MEMCPY_ERROR;
236  return false;
237  }
238  }
239 
240  //Start the timer
241  start_time = gpu_blas->get_time_us();
242 
243  // run GEMM & wait for completion
244  gpu_blas->run_blass_gemm(gst_ops_type);
245  while (!gpu_blas->is_gemm_op_complete()) {}
246  //End the timer
247  end_time = gpu_blas->get_time_us();
248 
249  //Converting microseconds to seconds
250  timetakenforoneiteration = (end_time - start_time)/1e6;
251 
252  gflops_interval = gpu_blas->gemm_gflop_count()/timetakenforoneiteration;
253 
254 
255  gst_last_sgemm_end_time = std::chrono::system_clock::now();
256  millis_last_sgemm =
257  time_diff(gst_last_sgemm_end_time, gst_last_sgemm_start_time);
258  if (static_cast<double>(
259  (1000 * gpu_blas->gemm_gflop_count()) /
260  target_stress) <
261  millis_last_sgemm) {
262  // last SGEMM timed-out (it took more than it should)
263  dyn_delay_target_stress = 1;
264  }
265 
266 
267  num_sgemm_ops++;
268  num_sgemm_ops_log_interval++;
269 
270  gst_end_time = std::chrono::system_clock::now();
271  millis_sgemm_ops =
272  time_diff(gst_end_time, gst_start_gflops_time);
273  if (millis_sgemm_ops >= NMAX_MS_SGEMM_OPS_RAMP_SUB_INTERVAL) {
274  // compute the GFLOPS
275  seconds_elapsed = static_cast<double>
276  (millis_sgemm_ops) / 1000;
277  if (seconds_elapsed > 0) {
278  curr_gflops = static_cast<double>(
279  gpu_blas->gemm_gflop_count() *
280  num_sgemm_ops) / seconds_elapsed;
281  if (curr_gflops >= target_stress && curr_gflops <
282  target_stress + target_stress * tolerance/2) {
283  ramp_actual_time =
284  time_diff(gst_end_time, gst_start_time) +
285  NMAX_MS_GPU_RUN_PEAK_PERFORMANCE;
286  delay_target_stress /= num_sgemm_ops;
287  return true;
288  }
289  }
290  proc_delay +=
291  (delay_target_stress * PROC_DEC_INC_SGEMM_FREQ_DELAY) / 100;
292  num_sgemm_ops = 0;
293  delay_target_stress = 0;
294  gst_start_gflops_time = std::chrono::system_clock::now();
295  }
296 
297  millis_sgemm_ops =
298  time_diff(gst_end_time, gst_log_interval_time);
299  if (millis_sgemm_ops >= log_interval) {
300  // compute the GFLOPS
301  seconds_elapsed = static_cast<double>
302  (millis_sgemm_ops) / 1000;
303 
304  if (seconds_elapsed > 0) {
305  curr_gflops = static_cast<double>(
306  gpu_blas->gemm_gflop_count() *
307  num_sgemm_ops_log_interval) / seconds_elapsed;
308  log_interval_gflops(gflops_interval);
309  }
310 
311  num_sgemm_ops_log_interval = 0;
312  gst_log_interval_time = std::chrono::system_clock::now();
313  }
314  }
315 
316  return false;
317 }

389 bool GSTWorker::do_gst_stress_test(int *error, std::string *err_description) {
390  uint16_t num_sgemm_ops = 0;
391  uint64_t total_milliseconds, log_interval_milliseconds;
392  uint64_t start_time, end_time;
393  double seconds_elapsed, gflops_interval;
394  double timetakenforoneiteration;
395  string msg;
396  std::chrono::time_point<std::chrono::system_clock> gst_start_time,
397  gst_end_time, gst_log_interval_time;
398 
399  *error = 0;
400  max_gflops = 0;
401  num_sgemm_ops = 0;
402  start_time = 0;
403  end_time = 0;
404 
405  gst_start_time = std::chrono::system_clock::now();
406  gst_log_interval_time = std::chrono::system_clock::now();
407 
408  for (;;) {
409  // check if stop signal was received
410  if (rvs::lp::Stopping())
411  return false;
412 
413  if (copy_matrix) {
414  // copy matrix before each GEMM
415  if (!gpu_blas->copy_data_to_gpu(gst_ops_type)) {
416  *error = 1;
417  *err_description = GST_BLAS_MEMCPY_ERROR;
418  return false;
419  }
420  }
421 
422  //Start the timer
423  start_time = gpu_blas->get_time_us();
424 
425  // run GEMM & wait for completion
426  gpu_blas->run_blass_gemm(gst_ops_type);
427  while (!gpu_blas->is_gemm_op_complete()) {} // wait till gemm tasks complete
428  //End the timer
429  end_time = gpu_blas->get_time_us();
430 
431  num_sgemm_ops++;
432 
433  gst_end_time = std::chrono::system_clock::now();
434  total_milliseconds = time_diff(gst_end_time, gst_start_time);
435 
436  log_interval_milliseconds = time_diff(gst_end_time,
437  gst_log_interval_time);
438  if (log_interval_milliseconds >= log_interval && num_sgemm_ops > 0) {
439  seconds_elapsed = static_cast<double> (log_interval_milliseconds) /
440  1000;
441  if (seconds_elapsed != 0) {
442 
443  //Converting microseconds to seconds
444  timetakenforoneiteration = (end_time - start_time)/1e6;
445 
446  gflops_interval = gpu_blas->gemm_gflop_count()/timetakenforoneiteration;
447 
448  if (gflops_interval > max_gflops)
449  max_gflops = gflops_interval;
450 
451 
452  log_interval_gflops(max_gflops);
453 
454  // reset time & gflops related data
455  num_sgemm_ops = 0;
456  gst_log_interval_time = std::chrono::system_clock::now();
457  }
458  }
459 
460 
461  if(!gst_hot_calls) {
462  msg = "[" + action_name + "] " + MODULE_NAME + " " +
463  std::to_string(gpu_id) + " " + GST_START_MSG + " " +
464  " Execution time in milliseconds :" + std::to_string(total_milliseconds) +
465  " run_duration_ms :" + std::to_string(run_duration_ms);
466  rvs::lp::Log(msg, rvs::logtrace);
467  if (total_milliseconds >= run_duration_ms)
468  break;
469  }else{
470  msg = "[" + action_name + "] " + MODULE_NAME + " " +
471  std::to_string(gpu_id) + " " + GST_START_MSG + " " +
472  " Executing hot calls loop :" + std::to_string(gst_hot_calls);
473  rvs::lp::Log(msg, rvs::logtrace);
474 
475  gst_hot_calls--;
476  }
477  }
478 
479  return true;
480 }

485 void GSTWorker::run() {
486  string msg, err_description;
487  int error = 0;
488  bool gst_test_passed = true;
489 
490  max_gflops = 0;
491 
492  // log GST stress test - start message
493  msg = "[" + action_name + "] " + MODULE_NAME + " " +
494  std::to_string(gpu_id) + " " + GST_START_MSG + " " +
495  " Starting the GST stress test ";
496  rvs::lp::Log(msg, rvs::logtrace);
497 
498  // let the GPU ramp-up and check the result
499  bool ramp_up_success = do_gst_ramp(&error, &err_description);
500 
501  // GPU was not able to do the processing (HIP/rocBlas error(s) occurred)
502  if (error) {
503  string msg = "[" + action_name + "] " + MODULE_NAME + " "
504  + std::to_string(gpu_id) + " " + err_description;
505  rvs::lp::Log(msg, rvs::logerror);
506  log_to_json("err", err_description, rvs::logerror);
507 
508  return;
509  }
510 
511  // the GPU succeeded to achieve the target_stress GFLOPS
512  // continue with the same workload for the rest of the test duration
513  msg = "[" + action_name + "] " + MODULE_NAME + " " +
514  std::to_string(gpu_id) + " " + " GST ramp completed for interval :" + " " +
515  std::to_string(ramp_interval);
516  rvs::lp::Log(msg, rvs::loginfo);
517  //log_to_json(GST_TARGET_ACHIEVED_MSG, std::to_string(target_stress),
518  // rvs::loginfo);
519  if (run_duration_ms > 0) {
520  gst_test_passed = do_gst_stress_test(&error, &err_description);
521  // check if stop signal was received
522  if (rvs::lp::Stopping())
523  return;
524 
525  if (error) {
526  // GPU didn't complete the test (HIP/rocBlas error(s) occurred)
527  string msg = "[" + action_name + "] " + MODULE_NAME + " " +
528  std::to_string(gpu_id) + " " + err_description;
529  rvs::lp::Log(msg, rvs::logerror);
530  log_to_json("err", err_description, rvs::logerror);
531  return;
532  }
533  }
534 
535  log_interval_gflops(max_gflops);
536  check_target_stress(max_gflops);
537 }
