117  virtual void run() {
118  do {
119  std::chrono::time_point<std::chrono::system_clock> curr_time;
120  // wait for time to ellsapse (or for timer to be stopped)
121  while (brun) {
122  std::this_thread::sleep_for(std::chrono::milliseconds(1));
123  curr_time = std::chrono::system_clock::now();
124  if (std::chrono::duration_cast<std::chrono::milliseconds>
125  (curr_time - end_time).count() >= 0) {
126  break;
127  }
128  }
129 
130  // if timer is not stopped, call the callback function
131  if (brun) {
132  (cbarg->*cbfunc)();
133  }
134 
135  if (brunonce) {
136  brun = false;
137  } else {
138  end_time = std::chrono::system_clock::now() +
139  std::chrono::milliseconds(timeset);
140  }
141  } while (brun);
142  }
