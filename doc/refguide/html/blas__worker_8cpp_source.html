63 void blas_worker::setup_blas(void) {
64  blas_error = 0;
65  // setup rvsBlas
66  gpu_blas = std::unique_ptr<rvs_blas>(
67  new rvs_blas(gpu_device_index, matrix_size, matrix_size, matrix_size));
68 
69  // no lock guard for blas_error atm because there are no sync issues
70  if (gpu_blas == nullptr) {
71  blas_error = IET_MEM_ALLOC_ERROR;
72  set_setup_complete();
73  return;
74  }
75 
76  if (gpu_blas->error()) {
77  blas_error = IET_BLAS_ERROR;
78  set_setup_complete();
79  return;
80  }
81 
82  // generate random matrix & copy it to the GPU
83  gpu_blas->generate_random_matrix_data();
84  if (!gpu_blas->copy_data_to_gpu()) {
85  blas_error = IET_BLAS_MEMCPY_ERROR;
86  set_setup_complete();
87  return;
88  }
89 
90  set_setup_complete();
91 }

191 void blas_worker::run() {
192  setup_blas();
193  if (blas_error)
194  return;
195 
196  {
197  std::lock_guard<std::mutex> lck(mtx_brun);
198  brun = true;
199  }
200 
201  {
202  std::lock_guard<std::mutex> lck(mtx_num_sgemm);
203  num_sgemm_ops = 0;
204  }
205 
206  for (;;) {
207  {
208  std::lock_guard<std::mutex> lck(mtx_brun);
209  if (!brun)
210  break;
211  }
212 
213  {
214  std::lock_guard<std::mutex> lck(mtx_bpaused);
215  if (bpaused)
216  continue;
217  }
218 
219  {
220  std::lock_guard<std::mutex> lck(mtx_bsgemm_done);
221  sgemm_done = false;
222  }
223 
224  bool sgemm_success = true;
225  // run SGEMM & wait for completion
226  if (gpu_blas->run_blass_gemm()) {
227  while (!gpu_blas->is_gemm_op_complete()) {}
228  } else {
229  sgemm_success = false;
230  }
231 
232  {
233  std::lock_guard<std::mutex> lck(mtx_bsgemm_done);
234  sgemm_done = true;
235  }
236 
237  // increase number of SGEMM ops
238  if (sgemm_success) {
239  {
240  std::lock_guard<std::mutex> lck(mtx_bcount_sgemm);
241  if (bcount_sgemm) {
242  // lock_guard [num_sgemm_ops]
243  std::lock_guard<std::mutex> lck(mtx_num_sgemm);
244  num_sgemm_ops++;
245  }
246  }
247 
248  // lock_guard [sgemm_delay]
249  {
250  std::lock_guard<std::mutex> lck(mtx_sgemm_delay);
251  usleep_ex(sgemm_delay);
252  }
253  }
254 
255  // check if stop signal was received
256  if (rvs::lp::Stopping())
257  break;
258  }
259 }

265 void blas_worker::usleep_ex(uint64_t microseconds) {
266  uint64_t total_microseconds = microseconds;
267  for (;;) {
268  if (total_microseconds > USLEEP_MAX_VAL) {
269  usleep(USLEEP_MAX_VAL);
270  total_microseconds -= USLEEP_MAX_VAL;
271  } else {
272  usleep(total_microseconds);
273  return;
274  }
275  }
