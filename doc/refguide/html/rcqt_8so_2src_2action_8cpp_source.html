107 int rcqt_action::run() {
108  string msg;
109  bool propchk = false;
110 
111  // get the action name
112  if (property_get(RVS_CONF_NAME_KEY, &action_name)) {
113  rvs::lp::Err("Action name missing", MODULE_NAME_CAPS);
114  return 1;
115  }
116 
117  // check for -j flag (json logging)
118  if (property.find("cli.-j") != property.end()) {
119  unsigned int sec;
120  unsigned int usec;
121  rvs::lp::get_ticks(&sec, &usec);
122  bjson = true;
123  json_rcqt_node = rvs::lp::LogRecordCreate(MODULE_NAME,
124  action_name.c_str(), rvs::logresults, sec, usec);
125  if (json_rcqt_node == NULL) {
126  // log the error
127  msg =
128  action_name + " " + MODULE_NAME + " "
129  + JSON_CREATE_NODE_ERROR;
130  rvs::lp::Err(msg, MODULE_NAME_CAPS, action_name);
131  return 1;
132  }
133  }
134 
135  // check if package check action is going to trigger
136  propchk = rvs::actionbase::has_property(PACKAGE);
137  if (propchk == true)
138  return pkgchk_run();
139 
140  propchk = rvs::actionbase::has_property(PACKAGELIST);
141  if (propchk == true)
142  return pkglist_run();
143 
144  return -1;
145 }

152 int rcqt_action::pkgchk_run() {
153 
154  string package_name;
155  string package;
156  string msg;
157 
158  has_property(PACKAGE, &package_name);
159  std::stringstream ss{package_name};
160  vector<string> package_list;
161  while(ss >> package){
162  if(!package.empty())
163  package_list.push_back(remSpaces(package));
164  }
165 
166  handlerCreator creator;
167  for( auto pkg : package_list){
168 
169  auto handler = creator.getPackageHandler(pkg);
170  if(!handler){
171  std::cout << "Failed to create handler " << std::endl;
172  return -1;
173  }
174 
175  handler->parseManifest();
176  handler->validatePackages();
177  }
178 
179  return 0;
180 }
