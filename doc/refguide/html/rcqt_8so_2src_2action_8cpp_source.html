95 int action::run() {
96  int error = 0;
97  string msg;
98  bool pkgchk_bool = false;
99  bool usrchk_bool = false;
100  bool kernelchk_os_bool = false;
101  bool ldcfgchk_so_bool = false;
102  bool filechk_bool = false;
103 
104  // get the action name
105  rvs::actionbase::property_get_action_name(&error);
106  if (error == 2) {
107  msg = "action field is missing in gst module";
108  cerr << "RVS-RCQT: " << msg;
109  return -1;
110  }
111 
112  // check for -j flag (json logging)
113  if (property.find("cli.-j") != property.end()) {
114  unsigned int sec;
115  unsigned int usec;
116  rvs::lp::get_ticks(&sec, &usec);
117  bjson = true;
118  json_rcqt_node = rvs::lp::LogRecordCreate(MODULE_NAME,
119  action_name.c_str(), rvs::loginfo, sec, usec);
120  if (json_rcqt_node == NULL) {
121  // log the error
122  msg =
123  action_name + " " + MODULE_NAME + " "
124  + JSON_CREATE_NODE_ERROR;
125  cerr << "RVS-RCQT: " << msg;
126  }
127  }
128 
129  // check if package check action is going to trigger
130  pkgchk_bool = rvs::actionbase::has_property(PACKAGE);
131  if (pkgchk_bool == true)
132  return pkgchk_run();
133 
134  // check if usrer check action is going to trigger
135  usrchk_bool = rvs::actionbase::has_property(USER);
136 
137  if (usrchk_bool == true)
138  return usrchk_run();
139 
140  // chck if kernel version action is going to trigger
141  kernelchk_os_bool = rvs::actionbase::has_property(OS_VERSION);
142 
143  if (kernelchk_os_bool)
144  return kernelchk_run();
145 
146  // check if ldcfg check action is going to trigger
147  ldcfgchk_so_bool = rvs::actionbase::has_property(SONAME);
148

166 int action::pkgchk_run() {
167  string package_name;
168  string msg;
169  if (has_property(PACKAGE, &package_name)) {
170  bool version_exists = false;
171 
172  // Checking if version field exists
173  string version_name;
174  version_exists = has_property(VERSION, &version_name);
175  pid_t pid;
176  int fd[2];
177  pipe(fd);
178 
179  pid = fork();
180  if (pid == 0) {
181  // Child process
182 
183  // Pipe the standard output to the fd[1]
184  dup2(fd[1], STDOUT_FILENO);
185  dup2(fd[1], STDERR_FILENO);
186  char buffer[BUFFER_SIZE];
187 
188  snprintf(buffer, BUFFER_SIZE, \
189  "dpkg-query -W -f='${Status} ${Version}\n' %s", package_name.c_str());
190 
191  // We execute the dpkg-querry
192  system(buffer);
193  exit(0);
194  } else if (pid > 0) {
195  // Parent
196  char result[BUFFER_SIZE];
197  int count;
198  close(fd[1]);
199 
200  // We read the result from the dpk-querry from the fd[0]
201  count = read(fd[0], result, BUFFER_SIZE);
202 
203  result[count] = 0;
204  string result1 = result;
205 
206  // We parse the given result
207  string version_value = result1.substr(21, count - 21);
208  int ending = version_value.find_first_of('\n');
209  if (ending > 0) {
210  version_value = version_value.substr(0, ending);
211  }
212  string passed = "[" + action_name + "] " + "rcqt packagecheck "
213  + package_name + " true";
214  string failed = "[" + action_name + "] " + "rcqt packagecheck "
215  + package_name + " false";
216 
217  /*
218  * If result start with dpkg-querry: then we haven't found the package
219  * If we get something different, then we confirme that the package is found
220  * if version is equal to the required then the test pass
221  */
222 
223  if (strstr(result, "dpkg-query:") == result) {
224  log(failed.c_str(), rvs::logresults);
225  if (bjson && json_rcqt_node != nullptr) {
226  rvs::lp::AddString(json_rcqt_node, package_name, "not exists");
227  rvs::lp::AddString(json_rcqt_node, "pkgchk", "false");
228  rvs::lp::LogRecordFlush(json_rcqt_node);
229  }
230  } else if (version_exists == false) {
231  log(passed.c_str(), rvs::logresults);
232  if (bjson && json_rcqt_node != nullptr) {
233  rvs::lp::AddString(json_rcqt_node, package_name, "exists");
234  rvs::lp::AddString(json_rcqt_node, "pkgchk", "true");
235  rvs::lp::LogRecordFlush(json_rcqt_node);
236  }
237  } else if (version_name.compare(version_value) == 0) {
238  log(passed.c_str(), rvs::logresults);
239  if (bjson && json_rcqt_node != nullptr) {
240  rvs::lp::AddString(json_rcqt_node, package_name, "exists");
241  rvs::lp::AddString(json_rcqt_node, version_name, "exists");
242  rvs::lp::AddString(json_rcqt_node, "pkgchk", "true");
243  rvs::lp::LogRecordFlush(json_rcqt_node);
244  }
245  } else {
246  log(failed.c_str(), rvs::logresults);
247  if (bjson && json_rcqt_node != nullptr) {
248  rvs::lp::AddString(json_rcqt_node, package_name, "exists");
249  rvs::lp::AddString(json_rcqt_node, version_name, "not exists");
250  rvs::lp::AddString(json_rcqt_node, "pkgchk", "false");
251  rvs::lp::LogRecordFlush(json_rcqt_node);
252  }
253  }
254  } else {
255  // fork process error
256  cerr << "RVS-RCQT: INTERNAL_ERROR" << '\n';
257  return -1;
258  }
259  return 0;
260  }
261  return -1;
262 }

269 int action::usrchk_run() {
270  string err_msg, msg;
271  string user_name;
272  if (has_property(USER, &user_name)) {
273  bool group_exists = false;
274  string group_values_string;
275 
276  // Check if gruop exists
277  group_exists = has_property(GROUP, &group_values_string);
278 
279  // Structures for checking group and user
280  struct passwd pwd, *result;
281 
282  char pwdbuffer[2000];
283  int pwdbufflenght = 2000;
284  struct group grp, *grprst;
285  string user_exists = "[" + action_name + "] " + "rcqt usercheck "
286  + user_name + " true";
287  string user_not_exists = "[" + action_name + "] " + "rcqt usercheck "
288  + user_name + " false";
289 
290  // Check for given user
291  if (getpwnam_r(user_name.c_str()
292  , &pwd, pwdbuffer, pwdbufflenght, &result) != 0) {
293  cerr << "RVS-RCQT: Error with getpwnam_r" << endl;
294  return -1;
295  }
296  if (result == nullptr) {
297  log(user_not_exists.c_str(), rvs::logresults);
298  if (bjson && json_rcqt_node != nullptr) {
299  rvs::lp::AddString(json_rcqt_node, user_name, " false");
300  }
301  } else {
302  log(user_exists.c_str(), rvs::logresults);
303  if (bjson && json_rcqt_node != nullptr) {
304  rvs::lp::AddString(json_rcqt_node, user_name, " true");
305  }
306  }
307  if (group_exists && result != nullptr) {
308  // Put the group list into vector
309  string delimiter = ",";
310  vector<string> group_vector;
311  group_vector = str_split(group_values_string, delimiter);
312 
313  // Check if the group exists
314  for (vector<string>::iterator vector_iter = group_vector.begin()
315  ; vector_iter != group_vector.end(); vector_iter++) {
316  string user_group = "rcqt usercheck " + user_name;
317  int error_group;
318 
319  if ((error_group = getgrnam_r(vector_iter->c_str()
320  , &grp, pwdbuffer, pwdbufflenght, &grprst)) != 0) {
321  cerr << "RVS-RCQT: Error with getgrnam_r" << endl;
322  return -1;
323  }
324  if (error_group == EIO) {
325  cerr << "RVS-RCQT: IO error" << endl;
326  return -1;
327  } else if (error_group == EINTR) {
328  cerr << "RVS-RCQT: Error sginal was caught during getgrnam_r" << endl;
329  return -1;
330  } else if (error_group == EMFILE) {
331  cerr << "RVS-RCQT: Error file descriptors are currently open" << endl;
332  return -1;
333  } else if (error_group == ERANGE) {
334  cerr << "RVS-RCQT: Error insufficient buffer in getgrnam_r" << endl;
335  return -1;
336  }
337  string err_msg;
338  if (grprst == nullptr) {
339  err_msg = "group ";
340  err_msg += vector_iter->c_str();
341  err_msg += " does not exist";
342  cerr << "RVS-RCQT: " << err_msg << endl;
343  return -1;
344  }
345 
346  int i;
347  int j = 0;
348 
349  // Compare if the user group id is equal to the group id
350  for (i = 0; grp.gr_mem[i] != NULL; i++) {
351  if (strcmp(grp.gr_mem[i], user_name.c_str()) == 0) {
352  user_group = "[" + action_name + "] " + user_group + \
353  " " + vector_iter->c_str() + " true";
354  log(user_group.c_str(), rvs::logresults);
355  if (bjson && json_rcqt_node != nullptr) {
356  rvs::lp::AddString(json_rcqt_node
357  , user_group + " " + vector_iter->c_str(), " true");
358  }
359  j = 1;
360  break;
361  }
362  }
363 
364  // If the index is 0 then we user id doesn't match the group id
365  if (j == 0) {
366  // printf("user is not in the group\n");
367 
368  user_group = "[" + action_name + "] " + user_group + " " \
369  + vector_iter->c_str() + " false";
370  log(user_group.c_str(), rvs::logresults);
371  if (bjson && json_rcqt_node != nullptr) {
372  rvs::lp::AddString(json_rcqt_node,
373  user_group + " " + vector_iter->c_str(), "false");
374  }
375  j = 1;
376  }
377  }
378  }
379  if (bjson && json_rcqt_node != nullptr) {
380  rvs::lp::LogRecordFlush(json_rcqt_node);
381  }
382  return 0;
383  }
384  return -1;
385 }

392 int action::kernelchk_run() {
393  string msg;
394  string os_version_values;
395  string kernel_version_values;
396 
397  if (has_property(OS_VERSION, &os_version_values)) {
398  // Check kernel version
399  if (has_property(KERNEL_VERSION,
400  &kernel_version_values) == false) {
401  cerr << "RVS-RCQT: Kernel version missing in config" << endl;
402  return -1;
403  }
404 
405  /*
406  * Fill the os version vector and kernel version vector with
407  */
408  vector<string> os_version_vector = str_split(os_version_values, ",");
409 
410  vector<string> kernel_version_vector =
411  str_split(kernel_version_values, ",");
412 
413  /*
414  * Parsing /etc/os-release file for pretty name to extract
415  */
416  std::ifstream os_version_read("/etc/os-release");
417  string os_actual = "";
418  string os_file_line;
419  bool os_version_correct = false;
420  bool os_version_found_in_system = false;
421  while (getline(os_version_read, os_file_line)) {
422  if (strcasestr(os_file_line.c_str(), "pretty_name") != nullptr) {
423  os_version_found_in_system = true;
424  os_actual = os_file_line.substr(13, os_file_line.length() - 14);
425  if (bjson && json_rcqt_node != nullptr) {
426  rvs::lp::AddString(json_rcqt_node, "os version", os_actual);
427  }
428  vector<string>::iterator os_iter;
429  for (os_iter = os_version_vector.begin()
430  ; os_iter != os_version_vector.end(); os_iter++) {
431  if (strcmp(os_iter->c_str(), os_actual.c_str()) == 0) {
432  os_version_correct = true;
433  break;
434  }
435  }
436  if (os_version_correct == true)
437  break;
438  }
439  }
440  if (os_version_found_in_system == false) {
441  cerr << "RVS-RCQT: Unable to locate actual OS installed" << endl;
442  return -1;
443  }
444 
445  // Get data about the kernel version
446  struct utsname kernel_version_struct;
447  if (uname(&kernel_version_struct) != 0) {
448  cerr << "RVS-RCQT: Unable to read kernel version" << endl;
449  return -1;
450  }
451 
452  string kernel_actual = kernel_version_struct.release;
453  if (bjson && json_rcqt_node != nullptr) {
454  rvs::lp::AddString(json_rcqt_node, "kernel version", kernel_actual);
455  }
456  bool kernel_version_correct = false;
457 
458  // Check if the given kernel version matches one from the list
459  vector<string>::iterator kernel_iter;
460  for (kernel_iter = kernel_version_vector.begin() ; \
461  kernel_iter != kernel_version_vector.end(); kernel_iter++)
462  if (kernel_actual.compare(*kernel_iter) == 0) {
463  kernel_version_correct = true;
464  break;
465  }
466  string result = "[" + action_name + "] " + "rcqt kernelcheck " + \
467  os_actual + " " + kernel_actual + " " + \
468  (os_version_correct && kernel_version_correct ? "true" : "false");
469  log(result.c_str(), rvs::logresults);
470  if (bjson && json_rcqt_node != nullptr) {
471  rvs::lp::AddString(json_rcqt_node, "kerelchk"
472  , os_version_correct && kernel_version_correct ? "true" : "false");
473  rvs::lp::LogRecordFlush(json_rcqt_node);
474  }
475  return 0;
476  }
477  return -1;
478 }

485 int action::ldcfgchk_run() {
486  string msg;
487  string soname_requested;
488  string arch_requested;
489  string ldpath_requested;
490  if (has_property(SONAME, &soname_requested)) {
491  if (has_property(ARCH, &arch_requested) == false) {
492  cerr << "RVS-RCQT: acrhitecture field missing in config" << endl;
493  return -1;
494  }
495  if (has_property(LDPATH, &ldpath_requested) == false) {
496  cerr << "RVS-RCQT: library path field missing in config" << endl;
497  return -1;
498  }
499  // Full path of shared object
500  string full_ld_path = ldpath_requested + "/" + soname_requested;
501 
502  int fd[2];
503  pid_t pid;
504  pipe(fd);
505  pid = fork();
506  if (pid == 0) {
507  // child process
508  dup2(fd[1], STDOUT_FILENO);
509  dup2(fd[1], STDERR_FILENO);
510  char buffer[256];
511  snprintf(buffer, sizeof(buffer), "objdump -f %s", full_ld_path.c_str());
512 
513  system(buffer);
514  exit(0);
515  } else if (pid > 0) {
516  // Parent process
517  char result[BUFFER_SIZE];
518  close(fd[1]);
519  string ld_config_result = "[" + action_name + "] " +
520  "rcqt ldconfigcheck ";
521  read(fd[0], result, BUFFER_SIZE);
522  string result_string = result;
523 
524  if (strstr(result, "architecture:") != nullptr) {
525  vector<string> objdump_lines = str_split(result_string, "\n");
526  int begin_of_the_arch_string = 0;
527  int end_of_the_arch_string = 0;
528  for (uint i = 0; i < objdump_lines.size(); i++) {
529  if (objdump_lines[i].find("architecture") != string::npos) {
530  begin_of_the_arch_string = objdump_lines[i].find(":");
531  end_of_the_arch_string = objdump_lines[i].find(",");
532  string arch_found = objdump_lines[i]
533  .substr(begin_of_the_arch_string + 2
534  , end_of_the_arch_string - begin_of_the_arch_string - 2);
535  if (bjson && json_rcqt_node != nullptr) {
536  rvs::lp::AddString(json_rcqt_node, "soname", soname_requested);
537  rvs::lp::AddString(json_rcqt_node, "architecture", arch_found);
538  }
539  if (arch_found.compare(arch_requested) == 0) {
540  string arch_pass = ld_config_result + soname_requested
541  + " " + full_ld_path + " " + arch_found + " true";
542  log(arch_pass.c_str(), rvs::logresults);
543  if (bjson && json_rcqt_node != nullptr) {
544  rvs::lp::AddString(json_rcqt_node, "ldchk", "true");
545  }
546  } else {
547  string arch_pass = ld_config_result + soname_requested + " "
548  + full_ld_path + " " + arch_found + " false";
549  log(arch_pass.c_str(), rvs::logresults);
550  if (bjson && json_rcqt_node != nullptr) {
551  rvs::lp::AddString(json_rcqt_node, "ldchk", "false");
552  }
553  }
554  }
555  }
556  } else {
557  string lib_fail = ld_config_result + soname_requested
558  + " " + full_ld_path + " " + arch_requested + " false";
559  log(lib_fail.c_str(), rvs::logresults);
560  if (bjson && json_rcqt_node != nullptr) {
561  rvs::lp::AddString(json_rcqt_node, "soname", soname_requested);
562  rvs::lp::AddString(json_rcqt_node, "ldchk", "false");
563  }
564  }
565  } else {
566  cerr << "RVS-RCQT: Internal Error" << endl;
567  return -1;
568  }
569  if (bjson && json_rcqt_node != nullptr) {
570  rvs::lp::LogRecordFlush(json_rcqt_node);
571  }
572 
573  return 0;
574  }
575  return -1;
576 }

594 int action::filechk_run() {
595  string exists_string, file, owner, group, msg, check;
596  int permission, type;
597  bool exists;
598  struct stat info;
599 
600  map<string, string>::iterator iter;
601  iter = property.find("name");
602  std::string action_name = iter->second;
603 
604  // get from property which file we are checking
605  iter = property.find("file");
606  file = iter->second;
607 
608  // get bool value of property exists
609  iter = property.find("exists");
610  if (iter == property.end())
611  exists = true;
612  else
613  exists_string = iter->second;
614  if (exists_string == "false")
615  exists = false;
616  else if (exists_string == "true")
617  exists = true;
618 
619  // check if exists property corresponds to real existence of the file
620  if (exists == false) {
621  if (stat(file.c_str(), &info) < 0)
622  check = "true";
623  else
624  check = "false";
625  msg = "[" + action_name + "] " + "rcqt filecheck "+ file +" DNE " + check;
626  log(msg.c_str(), rvs::logresults);
627  if (bjson && json_rcqt_node != nullptr) {
628  rvs::lp::AddString(json_rcqt_node
629  , "exists", file);
630  }
631  } else {
632  // when exists propetry is true,but file cannot be found
633  if (stat(file.c_str(), &info) < 0) {
634  msg = "[" + action_name + "] " + "rcqt filecheck "+ file +
635  " file is not found";
636  rvs::lp::Log(msg, rvs::logerror);
637  // if exists property is set to true and file is found,check each parameter
638  } else {
639  // check if owner is tested
640  iter = property.find("owner");
641  if (iter != property.end()) {
642  // check if value from property is equal to real one
643  owner = iter->second;
644  struct passwd p, *result;
645  char pbuff[256];
646  if ((getpwuid_r(info.st_uid, &p, pbuff, sizeof(pbuff), &result) != 0)) {
647  cerr << "RVS-RCQT: ["<< action_name << "] Error with getpwuid_r"
648  << endl;
649  return -1;
650  }
651  if (p.pw_name == owner)
652  check = "true";
653  else
654  check = "false";
655  msg = "[" + action_name + "] " + "rcqt filecheck " \
656  + owner +" " + check;
657  log(msg.c_str(), rvs::logresults);
658  if (bjson && json_rcqt_node != nullptr) {
659  rvs::lp::AddString(json_rcqt_node
660  , "owner", owner);
661  }
662  }
663  // check if group is tested
