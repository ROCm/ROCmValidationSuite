68 int rvs::exec::do_yaml(const std::string& config_file) {
69  int sts = 0;
70 
71  YAML::Node config = YAML::LoadFile(config_file);
72 
73  // find "actions" map
74  const YAML::Node& actions = config["actions"];
75 
76 
77  // for all actions...
78  for (YAML::const_iterator it = actions.begin(); it != actions.end(); ++it) {
79  const YAML::Node& action = *it;
80 
81  rvs::logger::log("Action name :" + action["name"].as<std::string>(), rvs::logresults);
82 
83  // if stop was requested
84  if (rvs::logger::Stopping()) {
85  return -1;
86  }
87 
88  // find module name
89  std::string rvsmodule;
90  try {
91  rvsmodule = action["module"].as<std::string>();
92  } catch(...) {
93  }
94 
95  // not found or empty
96  if (rvsmodule == "") {
97  // report error and go to next action
98  char buff[1024];
99  snprintf(buff, sizeof(buff), "action '%s' does not specify module.",
100  action["name"].as<std::string>().c_str());
101  rvs::logger::Err(buff, MODULE_NAME_CAPS);
102  return -1;
103  }
104 
105  // create action excutor in .so
106  rvs::action* pa = module::action_create(rvsmodule.c_str());
107  if (!pa) {
108  char buff[1024];
109  snprintf(buff, sizeof(buff),
110  "action '%s' could not crate action object in module '%s'",
111  action["name"].as<std::string>().c_str(),
112  rvsmodule.c_str());
113  rvs::logger::Err(buff, MODULE_NAME_CAPS);
114  return -1;
115  }
116 
117  if1* pif1 = dynamic_cast<if1*>(pa->get_interface(1));
118  if (!pif1) {
119  char buff[1024];
120  snprintf(buff, sizeof(buff),
121  "action '%s' could not obtain interface if1",
122  action["name"].as<std::string>().c_str());
123  module::action_destroy(pa);
124  return -1;
125  }
126 
127  // load action properties from yaml file
128  sts += do_yaml_properties(action, rvsmodule, pif1);
129  if (sts) {
130  module::action_destroy(pa);
131  return sts;
132  }
133 
134  // set also command line options:
135  for (auto clit = rvs::options::get().begin();
136  clit != rvs::options::get().end(); ++clit) {
137  std::string p(clit->first);
138  p = "cli." + p;
139  pif1->property_set(p, clit->second);
140  }
141 
142  // execute action
143  sts = pif1->run();
144 
145  // processing finished, release action object
146  module::action_destroy(pa);
147 
148  // errors?
149  if (sts) {
150  // cancel actions and return
151  return sts;
152  }
153  }
154 
155  return 0;
156 }

164 int rvs::exec::do_yaml_properties(const YAML::Node& node,
165  const std::string& module_name,
166  rvs::if1* pif1) {
167  int sts = 0;
168 
169  string indexes;
170  bool indexes_provided = false;
171  if (rvs::options::has_option("-i", &indexes) && (!indexes.empty()))
172  indexes_provided = true;
173 
174  rvs::logger::log("Module name :" + module_name, rvs::logresults);
175 
176  // for all child nodes
177  for (YAML::const_iterator it = node.begin(); it != node.end(); it++) {
178  // if property is collection of module specific properties,
179  if (is_yaml_properties_collection(module_name,
180  it->first.as<std::string>())) {
181  // pass properties collection to .so action object
182  sts += do_yaml_properties_collection(it->second,
183  it->first.as<std::string>(),
184  pif1);
185  } else {
186  // just set this one propertiy
187  if (indexes_provided && it->first.as<std::string>() == "device") {
188  std::replace(indexes.begin(), indexes.end(), ',', ' ');
189  sts += pif1->property_set("device", indexes);
190  } else {
191  sts += pif1->property_set(it->first.as<std::string>(),
192  it->second.as<std::string>());
193  }
194  }
195  }
196 
197  return sts;
198 }

206 int rvs::exec::do_yaml_properties_collection(const YAML::Node& node,
207  const std::string& parent_name,
208  if1* pif1) {
209  int sts = 0;
210 
211  // for all child nodes
212  for (YAML::const_iterator it = node.begin(); it != node.end(); it++) {
213  // prepend dot separated parent name and pass property to module
214  sts += pif1->property_set(parent_name + "." + it->first.as<std::string>(),
215  it->second.IsNull() ? std::string("") : it->second.as<std::string>());
216  }
217 
218  return sts;
219 }

230  const std::string& module_name,
231  const std::string& property_name) {
232 
233  if (module_name == "gpup") {
234  if (property_name == "properties")
235  return true;
236 
237  if (property_name == "io_links-properties")
238  return true;
239  } else {
240  if (module_name == "peqt") {
241  if (property_name == "capability") {
242  return true;
243  }
244  } else {
245  if (module_name == "gm") {
246  if (property_name == "metrics")
247  return true;
248  }
249  }
250  }
251 
252  return false;
253 }
