66 int rvs::exec::do_yaml(const std::string& config_file) {
67  int sts = 0;
68 
69  YAML::Node config = YAML::LoadFile(config_file);
70 
71  // find "actions" map
72  const YAML::Node& actions = config["actions"];
73 
74  // for all actions...
75  for (YAML::const_iterator it = actions.begin(); it != actions.end(); ++it) {
76  const YAML::Node& action = *it;
77 
78  // if stop was requested
79  if (rvs::logger::Stopping()) {
80  return -1;
81  }
82 
83  // find module name
84  std::string rvsmodule = action["module"].as<std::string>();
85 
86  // not found or empty
87  if (rvsmodule == "") {
88  // report error and go to next action
89  std::cerr << "RVS-CLI: action '"<< action["name"].as<std::string>()
90  << "' does not specify module.\n";
91  continue;
92  }
93 
94  // create action excutor in .so
95  rvs::action* pa = module::action_create(rvsmodule.c_str());
96  if (!pa) {
97  std::cerr << "RVS-CLI: action '"<< action["name"].as<std::string>()
98  << "' could not crate action object in module '" << rvsmodule.c_str()
99  << "'\n";
100  continue;
101  }
102 
103  if1* pif1 = dynamic_cast<if1*>(pa->get_interface(1));
104  if (!pif1) {
105  std::cerr << "RVS-CLI: action '"<< action["name"].as<std::string>()
106  << "' could not obtain interface IF1\n";
107 
108  module::action_destroy(pa);
109  continue;
110  }
111 
112  // load action properties from yaml file
113  sts += do_yaml_properties(action, rvsmodule, pif1);
114  if (sts) {
115  module::action_destroy(pa);
116  return sts;
117  }
118 
119  // set also command line options:
120  for (auto clit = rvs::options::get().begin();
121  clit != rvs::options::get().end(); ++clit) {
122  std::string p(clit->first);
123  p = "cli." + p;
124  pif1->property_set(p, clit->second);
125  }
126 
127  // execute action
128  sts = pif1->run();
129 
130  // procssin finished, release action object
131  module::action_destroy(pa);
132 
133  // errors?
134  if (sts) {
135  // cancel actions and return
136  return sts;
137  }
138  }
139 
140  return 0;
141 }

149 int rvs::exec::do_yaml_properties(const YAML::Node& node,
150  const std::string& module_name,
151  rvs::if1* pif1) {
152  int sts = 0;
153 
154  string indexes;
155  bool indexes_provided = false;
156  if (rvs::options::has_option("-i", &indexes) && (!indexes.empty()))
157  indexes_provided = true;
158 
159 
160  // for all child nodes
161  for (YAML::const_iterator it = node.begin(); it != node.end(); it++) {
162  const YAML::Node& child = *it;
163 
164  // if property is collection of module specific properties,
165  if (is_yaml_properties_collection(module_name,
166  it->first.as<std::string>())) {
167  // pass properties collection to .so action object
168  sts += do_yaml_properties_collection(it->second,
169  it->first.as<std::string>(),
170  pif1);
171  } else {
172  // just set this one propertiy
173  if (indexes_provided && it->first.as<std::string>() == "device") {
174  std::replace(indexes.begin(), indexes.end(), ',', ' ');
175  sts += pif1->property_set("device", indexes);
176  } else {
177  sts += pif1->property_set(it->first.as<std::string>(),
178  it->second.as<std::string>());
179  }
180  }
181  }
182 
183  return sts;
184 }

192 int rvs::exec::do_yaml_properties_collection(const YAML::Node& node,
193  const std::string& parent_name,
194  if1* pif1) {
195  int sts = 0;
196 
197  // for all child nodes
198  for (YAML::const_iterator it = node.begin(); it != node.end(); it++) {
199  // prepend dot separated parent name and pass property to module
200  sts += pif1->property_set(parent_name + "." + it->first.as<std::string>(),
201  it->second.IsNull() ? std::string("") : it->second.as<std::string>());
202  }
203 
204  return sts;
205 }
