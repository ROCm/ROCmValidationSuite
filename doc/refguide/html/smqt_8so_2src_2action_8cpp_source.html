52 // Prints to the provided buffer a nice number of bytes (KB, MB, GB, etc)
53 string action::pretty_print(ulong bytes, string action_name, string bar_name) {
54  std::string suffix[5] = { " B", " KB", " MB", " GB", " TB"};
55  std::stringstream ss;
56 
57  uint s = 0; // which suffix to use
58  double count = bytes;
59  while (count >= 1024 && s < 5) {
60  s++;
61  count /= 1024;
62  }
63  ss << "[" << action_name << "] smqt " << bar_name << " "\
64  << bytes << " (" << std::fixed << std::setprecision(2) <<
65  count << suffix[s] << ")";
66 
67  return ss.str();
68 }

123 int action::run(void) {
124  // config
125  int temp =rvs::actionbase::property_get_deviceid(int &error);
126  ulong bar1_req_size, bar1_base_addr_min, bar1_base_addr_max;
127  ulong bar2_req_size, bar2_base_addr_min, bar2_base_addr_max;
128  ulong bar4_req_size, bar4_base_addr_min, bar4_base_addr_max, bar5_req_size;
129  // output
130  ulong bar1_size, bar1_base_addr, bar2_size, bar2_base_addr;
131  ulong bar4_size, bar4_base_addr, bar5_size;
132  bool pass;
133  int error = 0;
134  string msg;
135  struct pci_access *pacc;
136  // get the action name
137  rvs::actionbase::property_get_action_name(&error);
138  if (error == 2) {
139  msg = "action field is missing in smqt module";
140  cerr << "RVS-SMQT: " << msg;
141  return -1;
142  }
143 
144  if (!get_all_common_config_keys()) {
145  std::cerr << "RVS-SMQT: Couldn't fetch common config keys "
146  << "from the configuration file!\n";
147  return -1;
148  }
149 
150  // get the pci_access structure
151  pacc = pci_alloc();
152  // initialize the PCI library
153  pci_init(pacc);
154  // get the list of devices
155  pci_scan_bus(pacc);
156 
157  struct pci_dev *dev;
158  dev = pacc->devices;
159 
160  if (!has_property("name", &action_name)) {
161  std::cerr << "RVS-SMQT: Error fetching action_name\n";
162  return false;
163  }
164  bar1_req_size = get_property("bar1_req_size");
165  bar2_req_size = get_property("bar2_req_size");
166  bar4_req_size = get_property("bar4_req_size");
167  bar5_req_size = get_property("bar5_req_size");
168  bar1_base_addr_min = get_property("bar1_base_addr_min");
169  bar2_base_addr_min = get_property("bar2_base_addr_min");
170  bar4_base_addr_min = get_property("bar4_base_addr_min");
171  bar1_base_addr_max = get_property("bar1_base_addr_max");
172  bar2_base_addr_max = get_property("bar2_base_addr_max");
173  bar4_base_addr_max = get_property("bar4_base_addr_max");
174 
175  // iterate over devices
176  for (dev = pacc->devices; dev; dev = dev->next) {
177  // fil in the info
178  pci_fill_info(dev, PCI_FILL_IDENT | PCI_FILL_BASES \
179  | PCI_FILL_CLASS | PCI_FILL_EXT_CAPS | PCI_FILL_CAPS | PCI_FILL_PHYS_SLOT);
180 
181  // computes the actual dev's location_id (sysfs entry)
182  uint16_t dev_location_id = ((((uint16_t)(dev->bus)) << 8) | (dev->func));
183 
184  int32_t gpu_id = rvs::gpulist::GetGpuId(dev_location_id);
185 
186  // if the device is not in the list, move on
187  if (-1 == gpu_id)
188  continue;
189 
190  // get actual values
191  bar1_base_addr = dev->base_addr[0];
192  bar1_size = dev->size[0];
193  bar2_base_addr = dev->base_addr[2];
194  bar2_size = dev->size[2];
195  bar4_base_addr = dev->base_addr[5];
196  bar4_size = dev->size[5];
197  bar5_size = dev->rom_size;
198 
199  // check if values are as expected
200  pass = true;
201 
202  if (bar1_base_addr < bar1_base_addr_min ||
203  bar1_base_addr > bar1_base_addr_max)
204  pass = false;
205  if (bar2_base_addr < bar2_base_addr_min ||
206  bar2_base_addr > bar2_base_addr_max)
207  pass = false;
208  if (bar4_base_addr < bar4_base_addr_min ||
209  bar4_base_addr > bar4_base_addr_max)
210  pass = false;
211 
212  if (bar1_req_size > bar1_size ||
213  bar2_req_size < bar2_size ||
214  bar4_req_size < bar4_size ||
215  bar5_req_size < bar5_size)
216  pass = false;
217 
218  // loginfo
219  unsigned int sec;
220  unsigned int usec;
221  rvs::lp::get_ticks(&sec, &usec);
222  string msgs1, msgs2, msgs4, msgs5, msga1, msga2, msga4, pmsg, str, pass_str;
223  char hex_value[30];
224 
225  if (pass)
226  pass_str = "pass";
227  else
228  pass_str = "fail";
229 
230  // formating bar1 size for print
231  msgs1 = pretty_print(bar1_size, action_name, "bar1_size");
232 
233  // formating bar2 size for print
234  msgs2 = pretty_print(bar2_size, action_name, "bar2_size");
235 
236  // formating bar4 size for print
237  msgs4 = pretty_print(bar4_size, action_name, "bar4_size");
238 
239  // formating bar5 size for print
240  msgs5 = pretty_print(bar5_size, action_name, "bar5_size");
241 
242  snprintf(hex_value, 0xFF, "%lX", bar1_base_addr);
243  msga1 = "[" + action_name + "] " + " smqt bar1_base_addr " + hex_value;
244  snprintf(hex_value, 0xFF, "%lX", bar2_base_addr);
245  msga2 = "[" + action_name + "] " + " smqt bar2_base_addr " + hex_value;
246  snprintf(hex_value, 0xFF, "%lX", bar4_base_addr);
247  msga4 = "[" + action_name + "] " + " smqt bar4_base_addr " + hex_value;
248  pmsg = "[" + action_name + "] " + " smqt " + pass_str;
249 
250  void* r = rvs::lp::LogRecordCreate("SMQT", action_name.c_str(),
251  rvs::loginfo, sec, usec);
252 
253  rvs::lp::Log(msgs1, rvs::loginfo, sec, usec);
254  rvs::lp::Log(msga1, rvs::loginfo, sec, usec);
255  rvs::lp::Log(msgs2, rvs::loginfo, sec, usec);
256  rvs::lp::Log(msga2, rvs::loginfo, sec, usec);
257  rvs::lp::Log(msgs4, rvs::loginfo, sec, usec);
258  rvs::lp::Log(msga4, rvs::loginfo, sec, usec);
259  rvs::lp::Log(msgs5, rvs::loginfo, sec, usec);
260  rvs::lp::Log(pmsg, rvs::logresults);
261  rvs::lp::AddString(r, "bar1_size", std::to_string(bar1_size));
262  rvs::lp::AddString(r, "bar1_base_addr", std::to_string(bar1_base_addr));
263  rvs::lp::AddString(r, "bar2_size", std::to_string(bar2_size));
264  rvs::lp::AddString(r, "bar2_base_addr", std::to_string(bar2_base_addr));
265  rvs::lp::AddString(r, "bar4_size", std::to_string(bar4_size));
266  rvs::lp::AddString(r, "bar4_base_addr", std::to_string(bar4_base_addr));
267  rvs::lp::AddString(r, "bar5_size", std::to_string(bar4_size));
268  rvs::lp::AddString(r, "pass", std::to_string(pass));
269  rvs::lp::LogRecordFlush(r);
270  }
271  return pass;
272 }
