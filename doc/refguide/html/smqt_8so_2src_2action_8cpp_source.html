63 string smqt_action::pretty_print(ulong bytes, uint16_t gpu_id,
64  string action_name, string bar_name) {
65  std::string suffix[5] = { " B", " KB", " MB", " GB", " TB"};
66  std::stringstream ss;
67 
68  uint s = 0; // which suffix to use
69  double count = bytes;
70  while (count >= 1024 && s < 5) {
71  s++;
72  count /= 1024;
73  }
74  ss << "[" << action_name << "] smqt " << gpu_id << " " <<
75  bar_name << " "
76  << bytes << " (" << std::fixed << std::setprecision(2) <<
77  count << suffix[s] << ")";
78 
79  return ss.str();
80 }

164 int smqt_action::run(void) {
165  bool global_pass = true;
166  string msg;
167  struct pci_access *pacc;
168  bool devid_found = false;
169 
170  if (!get_all_common_config_keys()) {
171  msg = "Couldn't fetch common config keys from the configuration file!";
172  rvs::lp::Err(msg, MODULE_NAME, action_name);
173  return -1;
174  }
175 
176  if (!get_all_smqt_config_keys()) {
177  msg = "Couldn't fetch bar config keys from the configuration file!";
178  rvs::lp::Err(msg, MODULE_NAME, action_name);
179  return -1;
180  }
181 
182  // get the pci_access structure
183  pacc = pci_alloc();
184  // initialize the PCI library
185  pci_init(pacc);
186  // get the list of devices
187  pci_scan_bus(pacc);
188 
189  struct pci_dev *dev;
190  dev = pacc->devices;
191 
192  // iterate over devices
193  for (dev = pacc->devices; dev; dev = dev->next) {
194  bool pass = true;
195  // fil in the info
196  pci_fill_info(dev, PCI_FILL_IDENT | PCI_FILL_BASES \
197  | PCI_FILL_CLASS | PCI_FILL_EXT_CAPS | PCI_FILL_CAPS | PCI_FILL_PHYS_SLOT);
198 
199  // computes the actual dev's location_id (sysfs entry)
200  uint16_t dev_location_id = ((((uint16_t)(dev->bus)) << 8) | (dev->func));
201 
202  uint16_t gpu_id;
203  // if not and AMD GPU just continue
204  if (rvs::gpulist::location2gpu(dev_location_id, &gpu_id))
205  continue;
206 
207 #ifdef RVS_UNIT_TEST
208  on_set_device_gpu_id();
209 #endif
210 
211  // filter by device id if needed
212  if (property_device_id > 0) {
213  rvs::gpulist::gpu2device(gpu_id, &dev_id);
214  if (property_device_id != dev_id) {
215  continue;
216  keysts = false;
217  }
218  }
219 
220  devid_found = true;
221 
222  // filter by list of devices if needed
223  if (!property_device_all) {
224  if (property_device.end() ==
225  std::find(property_device.begin(), property_device.end(), gpu_id))
226  continue;
227  }
228 
229  // get actual values
230  bar1_base_addr = dev->base_addr[0];
231  bar1_size = dev->size[0];
232  bar2_base_addr = dev->base_addr[2];
233  bar2_size = dev->size[2];
234  bar4_base_addr = dev->base_addr[5];
235  bar4_size = dev->size[5];
236  bar5_size = dev->rom_size;
237 
238 #ifdef RVS_UNIT_TEST
239  on_bar_data_read();
240 #endif
241 
242  // check if values are as expected
243  if (bar1_base_addr < bar1_base_addr_min ||
244  bar1_base_addr > bar1_base_addr_max)
245  pass = false;
246  if (bar2_base_addr < bar2_base_addr_min ||
247  bar2_base_addr > bar2_base_addr_max)
248  pass = false;
249  if (bar4_base_addr < bar4_base_addr_min ||
250  bar4_base_addr > bar4_base_addr_max)
251  pass = false;
252 
253  if (bar1_req_size > bar1_size ||
254  bar2_req_size < bar2_size ||
255  bar4_req_size < bar4_size ||
256  bar5_req_size < bar5_size)
257  pass = false;
258 
259  // loginfo
260  unsigned int sec;
261  unsigned int usec;
262  rvs::lp::get_ticks(&sec, &usec);
263  string msgs1, msgs2, msgs4, msgs5, msga1, msga2, msga4, pmsg, str, pass_str;
264  char hex_value[30];
265 
266  if (pass)
267  pass_str = "true";
268  else
269  pass_str = "false";
270 
271  // formating bar1 size for print
272  msgs1 = pretty_print(bar1_size, gpu_id, action_name, "bar1_size");
273 
274  // formating bar2 size for print
275  msgs2 = pretty_print(bar2_size, gpu_id, action_name, "bar2_size");
276 
277  // formating bar4 size for print
278  msgs4 = pretty_print(bar4_size, gpu_id, action_name, "bar4_size");
279 
280  // formating bar5 size for print
281  msgs5 = pretty_print(bar5_size, gpu_id, action_name, "bar5_size");
282 
283  snprintf(hex_value, sizeof(hex_value), "%lX", bar1_base_addr);
284  msga1 = "[" + action_name + "] " + " smqt " + std::to_string(gpu_id) +
285  " bar1_base_addr " + hex_value;
286  snprintf(hex_value, sizeof(hex_value), "%lX", bar2_base_addr);
287  msga2 = "[" + action_name + "] " + " smqt " + std::to_string(gpu_id) +
288  " bar2_base_addr " + hex_value;
289  snprintf(hex_value, sizeof(hex_value), "%lX", bar4_base_addr);
290  msga4 = "[" + action_name + "] " + " smqt " + std::to_string(gpu_id) +
291  " bar4_base_addr " + hex_value;
292  pmsg = "[" + action_name + "] " + " smqt " + std::to_string(gpu_id) +
293  " " +pass_str;
294 
295  void* r = rvs::lp::LogRecordCreate("SMQT", action_name.c_str(),
296  rvs::loginfo, sec, usec);
297 
298  void* res = rvs::lp::LogRecordCreate("SMQT", action_name.c_str(),
299  rvs::logresults, sec, usec);
300 
301  rvs::lp::Log(msgs1, rvs::loginfo, sec, usec);
302  rvs::lp::Log(msga1, rvs::loginfo, sec, usec);
303  rvs::lp::Log(msgs2, rvs::loginfo, sec, usec);
304  rvs::lp::Log(msga2, rvs::loginfo, sec, usec);
305  rvs::lp::Log(msgs4, rvs::loginfo, sec, usec);
306  rvs::lp::Log(msga4, rvs::loginfo, sec, usec);
307  rvs::lp::Log(msgs5, rvs::loginfo, sec, usec);
308  rvs::lp::Log(pmsg, rvs::logresults);
309  rvs::lp::AddInt(r, "gpu", gpu_id);
310  rvs::lp::AddString(r, "bar1_size", std::to_string(bar1_size));
311  rvs::lp::AddString(r, "bar1_base_addr", std::to_string(bar1_base_addr));
312  rvs::lp::AddString(r, "bar2_size", std::to_string(bar2_size));
313  rvs::lp::AddString(r, "bar2_base_addr", std::to_string(bar2_base_addr));
314  rvs::lp::AddString(r, "bar4_size", std::to_string(bar4_size));
315  rvs::lp::AddString(r, "bar4_base_addr", std::to_string(bar4_base_addr));
316  rvs::lp::AddString(r, "bar5_size", std::to_string(bar4_size));
317  rvs::lp::AddString(res, "pass", std::to_string(pass));
318  rvs::lp::LogRecordFlush(r);
319  rvs::lp::LogRecordFlush(res);
320  if (!pass)
321  global_pass = false;
322  }
323  if (!devid_found) {
324  global_pass = false;
325  msg = "No devices match criteria from the test configuration.";
326  rvs::lp::Err(msg, MODULE_NAME, action_name);
327  return -1;
328  }
329  return global_pass ? 0 : -1;
330 }
