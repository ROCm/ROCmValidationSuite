63 int rvs::exec::run() {
64  int sts = 0;
65  string val;
66  string path;
67 
68  options::has_option("pwd", &path);
69 
70  // check -h options
71  if (rvs::options::has_option("-h", &val)) {
72  do_help();
73  return 0;
74  }
75 
76  // check -v options
77  if (rvs::options::has_option("-ver", &val)) {
78  do_version();
79  return 0;
80  }
81 
82  // check -d options
83  if (rvs::options::has_option("-d", &val)) {
84  int level;
85  try {
86  level = std::stoi(val);
87  }
88  catch(...) {
89  cerr << "RVS-CLI: syntax error: logging level not integer: " << val <<endl;
90  return -1;
91  }
92  if (level < 0 || level > 5) {
93  cerr << "RVS-CLI: syntax error: logging level not in range [0..5]: "
94  << val <<endl;
95  return -1;
96  }
97  logger::log_level(level);
98  }
99 
100  // if verbose is set, set logging level to the max value (i.e. 5)
101  if (rvs::options::has_option("-v")) {
102  logger::log_level(5);
103  }
104 
105  // check -a options
106  if (rvs::options::has_option("-a", &val)) {
107  logger::append(true);
108  }
109 
110  // check -j options
111  if (rvs::options::has_option("-j", &val)) {
112  logger::to_json(true);
113  }
114 
115  string config_file;
116  if (rvs::options::has_option("-c", &val)) {
117  config_file = val;
118  } else {
119  config_file = "conf/rvs.conf";
120  config_file = path + config_file;
121  }
122 
123  // Check if pConfig file exists
124  std::ifstream file(config_file);
125 
126  if (!file.good()) {
127  cerr << "RVS-CLI: " << config_file << " file is missing.\n";
128  return -1;
129  } else {
130  file.close();
131  }
132 
133  // construct modules configuration file relative path
134  val = path + ".rvsmodules.config";
135  rvs::module::initialize(val.c_str());
136 
137  if (rvs::options::has_option("-t", &val)) {
138  cout << endl << "ROCm Validation Suite (version " <<
139  LIB_VERSION_STRING << ")" << endl << endl;
140  cout << "Modules available:" << endl;
141  rvs::module::do_list_modules();
142  return 0;
143  }
144 
145  logger::initialize();
146 
147  if (rvs::options::has_option("-g")) {
148  int sts = do_gpu_list();
149  logger::terminate();
150  rvs::module::terminate();
151  return sts;
152  }
153 
154  try {
155  sts = do_yaml(config_file);
156  } catch(std::exception& e) {
157  sts = -999;
158  cerr << "RVS-CLI: error processing configuration file " << config_file << endl;
159  cerr << "RVS-CLI: exception: " << e.what() << endl;
160  }
161 
162  logger::terminate();
163 
164  rvs::module::terminate();
165 
166  return sts;
167 }

175 void rvs::exec::do_help() {
176  cout << "\nUsage: rvs [options]\n";
177  cout << "\nOptions:\n\n";
178  cout << "-a --appendLog When generating a debug logfile, do not overwrite the contents\n";
179  cout << " of a current log. Used in conjuction with the -d and -l options.\n";
180  cout << "-c --config Specify the configuration file to be used.\n";
181  cout << " The default is <install base>/conf/RVS.conf\n";
182  cout << " --configless Run RVS in a configless mode. Executes a \"long\" test on all\n";
183  cout << " supported GPUs.\n";
184  cout << "-d --debugLevel Specify the debug level for the output log. The range is\n";
185  cout << " 0 to 5 with 5 being the most verbose.\n";
186  cout << " Used in conjunction with the -l flag.\n";
187  cout << "-g --listGpus List the GPUs available and exit. This will only list GPUs\n";
188  cout << " that are supported by RVS.\n";
189  cout << "-i --indexes Comma separated list of indexes devices to run RVS on. This will\n";
190  cout << " override the device values specified in the configuration file for\n";
191  cout << " every action in the configuration file, including the ‘all’ value.\n";
192  cout << "-j --json Output should use the JSON format.\n";
193  cout << "-l --debugLogFile Specify the logfile for debug information. This will produce a log\n";
194  cout << " file intended for post-run analysis after an error.\n";
195  cout << " --quiet No console output given. See logs and return code for errors.\n";
196  cout << "-m --modulepath Specify a custom path for the RVS modules.\n";
197  cout << " --specifiedtest Run a specific test in a configless mode. Multiple word tests\n";
198  cout << " should be in quotes. This action will default to all devices,\n";
199  cout << " unless the indexes option is specifie.\n";
200  cout << "-t --listTests List the modules available to be executed through RVS and exit.\n";
201  cout << " This will list only the readily loadable modules\n";
202  cout << " given the current path and library conditions.\n";
203  cout << "-v --verbose Enable verbose reporting. This is equivalent to\n";
204  cout << " specifying the -d 5 option.\n";
205  cout << " --version Displays the version information and exits.\n";
206  cout << "-h --help Display usage information and exit.\n";
207 }

210 int rvs::exec::do_gpu_list() {
211  cout << "\nROCm Validation Suite (version " << LIB_VERSION_STRING << ")\n\n";
212 
213  // create action excutor in .so
214  rvs::action* pa = module::action_create("pesm");
215  if (!pa) {
216  cerr << "RVS-CLI: could not list GPUs\n";
217  return 1;
218  }
219 
220  // obtain interface to set parameters and execute action
221  if1* pif1 = static_cast<if1*>(pa->get_interface(1));
222  if (!pif1) {
223  cerr << "RVS-CLI: could not obtain interface IF1\n";
224  module::action_destroy(pa);
225  return 1;
226  }
227 
228  pif1->property_set("name", "(launcher)");
229 
230  // specify "list GPUs" action
231  pif1->property_set("do_gpu_list", "");
232 
233  // set command line options:
234  for (auto clit = rvs::options::get().begin();
235  clit != rvs::options::get().end(); ++clit) {
236  string p(clit->first);
237  p = "cli." + p;
238  pif1->property_set(p, clit->second);
239  }
240 
241  // execute action
242  int sts = pif1->run();
243 
244  // procssing finished, release action object
245  module::action_destroy(pa);
246 
247  return sts;
248 }
