63 int rvs::exec::run() {
64  int sts = 0;
65  string val;
66  string path;
67 
68  options::has_option("pwd", &path);
69 
70  // check -h options
71  if (rvs::options::has_option("-h", &val)) {
72  do_help();
73  return 0;
74  }
75 
76  // check -v options
77  logger::log_level(rvs::logerror);
78  if (rvs::options::has_option("-ver", &val)) {
79  do_version();
80  return 0;
81  }
82 
83  // check -d options
84  if (rvs::options::has_option("-d", &val)) {
85  int level;
86  try {
87  level = std::stoi(val);
88  }
89  catch(...) {
90  char buff[1024];
91  snprintf(buff, sizeof(buff),
92  "logging level not integer: %s", val.c_str());
93  rvs::logger::Err(buff, MODULE_NAME_CAPS);
94  return -1;
95  }
96  if (level < 0 || level > 5) {
97  char buff[1024];
98  snprintf(buff, sizeof(buff),
99  "logging level not in range [0..5]: %s", val.c_str());
100  rvs::logger::Err(buff, MODULE_NAME_CAPS);
101  return -1;
102  }
103  logger::log_level(level);
104  }
105 
106  // if verbose is set, set logging level to the max value (i.e. 5)
107  if (rvs::options::has_option("-v")) {
108  logger::log_level(5);
109  }
110 
111  // check -a option
112  if (rvs::options::has_option("-a", &val)) {
113  logger::append(true);
114  }
115 
116  // check -l option
117  std::string s_log_file;
118  if (rvs::options::has_option("-l", &s_log_file)) {
119  logger::set_log_file(s_log_file);
120  }
121 
122  // check -j option
123  if (rvs::options::has_option("-j", &val)) {
124  logger::to_json(true);
125  }
126 
127  string config_file;
128  if (rvs::options::has_option("-c", &val)) {
129  config_file = val;
130  } else {
131  config_file = "../share/rocm-validation-suite/conf/rvs.conf";
132  // Check if pConfig file exist if not use old path for backward compatibility
133  std::ifstream file(path + config_file);
134  if (!file.good()) {
135  config_file = "conf/rvs.conf";
136  }
137  file.close();
138  config_file = path + config_file;
139  }
140 
141  // Check if pConfig file exists
142  std::ifstream file(config_file);
143  if (!file.good()) {
144  char buff[1024];
145  snprintf(buff, sizeof(buff),
146  "%s file is missing.", config_file.c_str());
147  rvs::logger::Err(buff, MODULE_NAME_CAPS);
148  return -1;
149  } else {
150  file.close();
151  }
152 
153  // construct modules configuration file relative path
154  val = path + "../share/rocm-validation-suite/conf/.rvsmodules.config";
155  // Check if config file exists if not check the old file location for backward compatibility
156  std::ifstream conf_file(val);
157  if (!conf_file.good()) {
158  val = path + ".rvsmodules.config";
159  }
160  conf_file.close();
161 
162  if (rvs::module::initialize(val.c_str())) {
163  return 1;
164  }
165 
166  if (rvs::options::has_option("-t", &val)) {
167  cout << endl << "ROCm Validation Suite (version " <<
168  LIB_VERSION_STRING << ")" << endl << endl;
169  cout << "Modules available:" << endl;
170  rvs::module::do_list_modules();
171  return 0;
172  }
173 
174  if (rvs::options::has_option("-q")) {
175  rvs::logger::quiet();
176  }
177 
178  if (logger::init_log_file()) {
179  char buff[1024];
180  snprintf(buff, sizeof(buff),
181  "could not access log file: %s", s_log_file.c_str());
182  rvs::logger::Err(buff, MODULE_NAME_CAPS);
183  return -1;
184  }
185 
186  if (rvs::options::has_option("-g")) {
187  int sts = do_gpu_list();
188  rvs::module::terminate();
189  logger::terminate();
190  return sts;
191  }
192 
193  DTRACE_
194  try {
195  sts = do_yaml(config_file);
196  } catch(std::exception& e) {
197  sts = -999;
198  char buff[1024];
199  snprintf(buff, sizeof(buff),
200  "error processing configuration file: %s", config_file.c_str());
201  rvs::logger::Err(buff, MODULE_NAME_CAPS);
202  snprintf(buff, sizeof(buff),
203  "exception: %s", e.what());
204  rvs::logger::Err(buff, MODULE_NAME_CAPS);
205  }
206 
207  rvs::module::terminate();
208  logger::terminate();
209 
210  DTRACE_
211  if (sts) {
212  DTRACE_
213  return sts;
214  }
215 
216  // if stop was requested
217  if (rvs::logger::Stopping()) {
218  DTRACE_
219  return -1;
220  }
221 
222  DTRACE_
223  return 0;
224 }

283 int rvs::exec::do_gpu_list() {
284  cout << "\nROCm Validation Suite (version " << LIB_VERSION_STRING << ")\n\n";
285 
286  // create action excutor in .so
287  rvs::action* pa = module::action_create("pesm");
288  if (!pa) {
289  rvs::logger::Err("could not list GPUs.", MODULE_NAME_CAPS);
290  return 1;
291  }
292 
293  // obtain interface to set parameters and execute action
294  if1* pif1 = static_cast<if1*>(pa->get_interface(1));
295  if (!pif1) {
296  rvs::logger::Err("could not obtain interface if1.", MODULE_NAME_CAPS);
297  module::action_destroy(pa);
298  return 1;
299  }
300 
301  pif1->property_set("name", "(launcher)");
302 
303  // specify "list GPUs" action
304  pif1->property_set("do_gpu_list", "");
305 
306  // set command line options:
307  for (auto clit = rvs::options::get().begin();
308  clit != rvs::options::get().end(); ++clit) {
309  string p(clit->first);
310  p = "cli." + p;
311  pif1->property_set(p, clit->second);
312  }
313 
314  // execute action
315  int sts = pif1->run();
316 
317  // procssing finished, release action object
318  module::action_destroy(pa);
319 
320  return sts;
321 }
