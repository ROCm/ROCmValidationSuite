237 void Worker::run() {
238  brun = true;
239  vector<uint16_t> gpus_location_id;
240  std::string val_str;
241  std::vector<std::string> val_vec;
242 
243  char path[IRQ_PATH_MAX_LENGTH];
244  uint32_t value;
245  uint32_t value2;
246 
247  string msg;
248  int ret;
249 
250  amd::smi::RocmSMI hw;
251  std::vector<std::shared_ptr<amd::smi::Device>> monitor_devices;
252 
253  struct pci_access *pacc;
254  struct pci_dev *dev_pci;
255 
256  unsigned int sec;
257  unsigned int usec;
258  void* r;
259 
260  rvs::timer<Worker> timer_running(&Worker::do_metric_values, this);
261 
262  // get timestamp
263  rvs::lp::get_ticks(&sec, &usec);
264 
265  // DiscoverDevices() will seach for devices and monitors and update internal
266  // data structures.
267  hw.DiscoverDevices();
268 
269  // IterateSMIDevices will iterate through all the known devices and apply
270  // the provided call-back to each device found.
271  hw.IterateSMIDevices(GetMonitorDevices,
272  reinterpret_cast<void *>(&monitor_devices));
273 
274  // add JSON output
275  r = rvs::lp::LogRecordCreate("gm", action_name.c_str(), rvs::loginfo,
276  sec, usec);
277 
278  // get the pci_access structure
279  pacc = pci_alloc();
280  // initialize the PCI library
281  pci_init(pacc);
282  // get the list of devices
283  pci_scan_bus(pacc);
284 
285  for (auto dev : monitor_devices) {
286  // get irq of device
287  snprintf(path, IRQ_PATH_MAX_LENGTH, "%s/device/irq", (dev->path()).c_str());
288  val_str = get_irq(path);
289 
290  // iterate over devices
291  for (dev_pci = pacc->devices; dev_pci; dev_pci = dev_pci->next) {
292  // fill in the info
293  pci_fill_info(dev_pci,
294  PCI_FILL_IDENT | PCI_FILL_BASES | PCI_FILL_CLASS
295  | PCI_FILL_EXT_CAPS | PCI_FILL_CAPS | PCI_FILL_PHYS_SLOT);
296 
297  // computes the actual dev's location_id (sysfs entry)
298  uint16_t dev_location_id = ((((uint16_t) (dev_pci->bus)) << 8)
299  | (dev_pci->func));
300 
301  if (dev_pci->irq == std::stoi(val_str)) {
302  uint32_t gpu_id = rvs::gpulist::GetGpuId(dev_location_id);
303  if (std::find(gpuids.begin(), gpuids.end(),
304  gpu_id) !=gpuids.end()) {
305  msg = "[" + action_name + "] gm " + std::to_string(gpu_id) +
306  " started";
307  rvs::lp::Log(msg, rvs::logresults, sec, usec);
308  rvs::lp::AddString(r, "device", std::to_string(gpu_id));
309 
310  auto metric_length = std::end(metric_names) -
311  std::begin(metric_names);
312  for (int i = 0; i < metric_length; i++) {
313  if (bounds[metric_names[i]].mon_metric) {
314  msg = "[" + action_name + "] " + MODULE_NAME + " " +
315  std::to_string(gpu_id) + " " + " monitoring " +
316  metric_names[i];
317  if (bounds[metric_names[i]].check_bounds) {
318  msg+= " bounds min:" +
319  std::to_string(bounds[metric_names[i]].min_val) +
320  " max:" + std::to_string
321  (bounds[metric_names[i]].max_val);
322  }
323  log(msg.c_str(), rvs::loginfo);
324  rvs::lp::AddString(r, metric_names[i], msg);
325  }
326  }
327 
328  irq_gpu_ids.insert(std::pair<string, Dev_metrics>
329  (val_str, {gpu_id, 0, 0, 0, 0, 0}));
330  met_violation.insert(std::pair<string, Metric_violation>
331  (val_str, {0, 0, 0, 0, 0}));
332  met_value.insert(std::pair<string, Metric_value>
333  (val_str, {0, 0, 0, 0, 0}));
334  count = 0;
335  break;
336  }
337  }
338  }
339  }
340  rvs::lp::LogRecordFlush(r);
341  // if log_interval timer starts
342  if (log_interval) {
343  timer_running.start(log_interval);
344  }
345  // worker thread has started
346  while (brun) {
347  rvs::lp::Log("[" + action_name + "] gm worker thread is running...",
348  rvs::logtrace);
349  for (auto dev : monitor_devices) {
350  // if irq is not in map skip monitoring this device
351  snprintf(path, IRQ_PATH_MAX_LENGTH, "%s/device/irq",
352  (dev->path()).c_str());
353  val_str = get_irq(path);
354 
355  auto it = irq_gpu_ids.find(val_str);
356  if (it == irq_gpu_ids.end()) {
357  break;
358  }
359 
360  if (bounds[GM_MEM_CLOCK].mon_metric) {
361  dev->readDevInfo(amd::smi::kDevGPUMClk, &val_vec);
362  for (auto vs : val_vec) {
363  size_t cur_pos = vs.find('*');
364 
365  if (cur_pos != string::npos) {
366  size_t cur_pos = vs.find("Mhz");
367  string token = vs.substr(2, cur_pos-2);
368  int mhz = stoi(token);
369  met_value[val_str].mem_clock = mhz;
370  if (!(mhz >= bounds[GM_MEM_CLOCK].min_val && mhz <=
371  bounds[GM_MEM_CLOCK].max_val) &&
372  bounds[GM_MEM_CLOCK].check_bounds) {
373  // write info and increase number of violations
374  msg = "[" + action_name + "] " + MODULE_NAME + " " +
375  std::to_string(irq_gpu_ids[val_str].gpu_id) + " " +
376  GM_MEM_CLOCK + " " + " bounds violation " +
377  std::to_string(mhz) + "Mhz";
378  log(msg.c_str(), rvs::loginfo);
379  met_violation[val_str].mem_clock_violation++;
380  if (term) {
381  if (force) {
382  // stop logging
383  rvs::lp::Stop(1);
384  // force exit
385  exit(EXIT_FAILURE);
386  } else {
387  // just signal stop processing
388  rvs::lp::Stop(0);
389  }
390  brun = false;
391  break;
392  }
393  }
394  irq_gpu_ids[val_str].av_mem_clock += mhz;
395  }
396  }
397  if (!brun) {
398  break;
399  }
400  val_vec.clear();
401  }
402 
403  if (bounds[GM_CLOCK].mon_metric) {
404  dev->readDevInfo(amd::smi::kDevGPUSClk, &val_vec);
405  for (auto vs : val_vec) {
406  size_t cur_pos = vs.find('*');
407  if (cur_pos != string::npos) {
408  size_t cur_pos = vs.find("Mhz");
409  string token = vs.substr(2, cur_pos-2);
410  int mhz = stoi(token);
411  met_value[val_str].clock = mhz;
412  if (!(mhz >= bounds[GM_CLOCK].min_val && mhz <=
413  bounds[GM_CLOCK].max_val) &&
414  bounds[GM_CLOCK].check_bounds) {
415  // write info
416  msg = "[" + action_name + "] " + MODULE_NAME + " " +
417  std::to_string(irq_gpu_ids[val_str].gpu_id) + " " +
418  GM_CLOCK + " " + " bounds violation " +
419  std::to_string(mhz) + "Mhz";
420  log(msg.c_str(), rvs::loginfo);
421  met_violation[val_str].clock_violation++;
422  if (term) {
423  if (force) {
424  // stop logging
425  rvs::lp::Stop(1);
426  // force exit
427  exit(EXIT_FAILURE);
428  } else {
429  // just signal stop processing
430  rvs::lp::Stop(0);
431  }
432  brun = false;
433  break;
434  }
435  }
436  irq_gpu_ids[val_str].av_clock += mhz;
437  }
438  }
439  val_vec.clear();
440  if (!brun) {
441  break;
442  }
443  }
444 
445  if (bounds[GM_TEMP].mon_metric) {
446  ret = dev->monitor()->readMonitor(amd::smi::kMonTemp, &value);
447  if (ret != -1) {
448  int temper = (static_cast<float>(value)/1000.0);
449  met_value[val_str].temp = temper;
450  irq_gpu_ids[val_str].av_temp += temper;
451  if (!(temper >= bounds[GM_TEMP].min_val && temper <=
452  bounds[GM_TEMP].max_val) &&
453  bounds[GM_TEMP].check_bounds) {
454  // write info
455  msg = "[" + action_name + "] " + MODULE_NAME + " " +
456  std::to_string(irq_gpu_ids[val_str].gpu_id) + " " +
457  + GM_TEMP + " " + " bounds violation " +
458  std::to_string(temper) + "C";
459  log(msg.c_str(), rvs::loginfo);
460  met_violation[val_str].temp_violation++;
461  if (term) {
462  if (force) {
463  // stop logging
464  rvs::lp::Stop(1);
465  // force exit
466  exit(EXIT_FAILURE);
467  } else {
468  // just signal stop processing
469  rvs::lp::Stop(0);
470  }
471  brun = false;
472  break;
473  }
474  }
475  } else {
476  msg = "[" + action_name + "] " + MODULE_NAME + " " +
477  std::to_string(irq_gpu_ids[val_str].gpu_id) + " " +
478  GM_TEMP + " Not available";
479  log(msg.c_str(), rvs::loginfo);
480  }
481  }
482 
483  if (bounds[GM_FAN].mon_metric) {
484  ret = dev->monitor()->readMonitor(amd::smi::kMonMaxFanSpeed,
485  &value);
486  if (ret == 0) {
487  ret = dev->monitor()->readMonitor(amd::smi::kMonFanSpeed,
488  &value2);
489  if (ret != -1) {
490  int fan = value2/static_cast<float>(value) * 100;
491  met_value[val_str].fan = fan;
492  irq_gpu_ids[val_str].av_fan += fan;
493  if (!(fan >= bounds[GM_FAN].min_val && fan <=
494  bounds[GM_FAN].max_val) &&
495  bounds[GM_FAN].check_bounds) {
496  // write info
497  msg = "[" + action_name + "] " + MODULE_NAME + " " +
498  std::to_string(irq_gpu_ids[val_str].gpu_id) + " " +
499  + GM_FAN + " " + " bounds violation " +
500  std::to_string(fan) + "%";
501  log(msg.c_str(), rvs::loginfo);
502  met_violation[val_str].fan_violation++;
503  if (term) {
504  if (force) {
505  // stop logging
506  rvs::lp::Stop(1);
507  // force exit
508  exit(EXIT_FAILURE);
509  } else {
510  // just signal stop processing
511  rvs::lp::Stop(0);
512  }
513  brun = false;
514  break;
515  }
516  }
517  } else {
518  msg = "[" + action_name + "] " + MODULE_NAME + " " +
519  std::to_string(irq_gpu_ids[val_str].gpu_id) + " " +
520  GM_FAN + " Not available";
521  log(msg.c_str(), rvs::loginfo);
522  }
523  }
524  }
525 
526  if (bounds[GM_POWER].mon_metric) {
527  int power = get_power(dev->monitor()-> path())/1000000;
528 
529  met_value[val_str].power = power;
530  irq_gpu_ids[val_str].av_power += power;
531  if (!(power >= bounds[GM_POWER].min_val && power <=
532  bounds[GM_POWER].max_val) &&
533  bounds[GM_POWER].check_bounds) {
534  // write info
535  msg = "[" + action_name + "] " + MODULE_NAME + " " +
536  std::to_string(irq_gpu_ids[val_str].gpu_id) + " " +
537  GM_POWER + " " + " bounds violation " +
538  std::to_string(power) + "Watts";
539  log(msg.c_str(), rvs::loginfo);
540  met_violation[val_str].power_violation++;
541  if (term) {
542  if (force) {
543  // stop logging
544  rvs::lp::Stop(1);
545  // force exit
546  exit(EXIT_FAILURE);
547  } else {
548  // just signal stop processing
549  rvs::lp::Stop(0);
550  }
551  brun = false;
552  break;
553  }
554  }
555  }
556  }
557  count++;
558  sleep(sample_interval);
559  }
560 
561  pci_cleanup(pacc);
562 
563  timer_running.stop();
564  // get timestamp
565  rvs::lp::get_ticks(&sec, &usec);
566 
567  for (map<string, Dev_metrics>::iterator it = irq_gpu_ids.begin();
568  it != irq_gpu_ids.end(); it++) {
569  // add string output
570  msg = "[" + action_name + "] gm " +
571  std::to_string((it->second).gpu_id) + " stopped";
572  rvs::lp::Log(msg, rvs::logresults, sec, usec);
573  }
574 
575  rvs::lp::Log("[" + stop_action_name + "] gm worker thread has finished",
576  rvs::logdebug);
577 }
