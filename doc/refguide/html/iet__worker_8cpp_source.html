134 void IETWorker::blasThread(int gpuIdx, uint64_t matrix_size, std::string iet_ops_type,
135  bool start, uint64_t run_duration_ms, int transa, int transb, float alpha, float beta,
136  int iet_lda_offset, int iet_ldb_offset, int iet_ldc_offset){
137 
138  std::chrono::time_point<std::chrono::system_clock> iet_start_time, iet_end_time;
139  double timetakenforoneiteration;
140  double gflops_interval;
141  double duration;
142  uint64_t gem_ops;
143  std::unique_ptr<rvs_blas> gpu_blas;
144  rvs_blas *free_gpublas;
145 
146  duration = 0;
147  gem_ops = 0;
148  // setup rvsBlas
149  gpu_blas = std::unique_ptr<rvs_blas>(new rvs_blas(gpuIdx, matrix_size, matrix_size, matrix_size, transa, transb, alpha, beta,
150  iet_lda_offset, iet_ldb_offset, iet_ldc_offset, iet_ops_type));
151 
152  //Genreate random matrix data
153  gpu_blas->generate_random_matrix_data();
154 
155  //Copy data to GPU
156  gpu_blas->copy_data_to_gpu(iet_ops_type);
157 
158  iet_start_time = std::chrono::system_clock::now();
159 
160  //Hit the GPU with load to increase temperature
161  while ( (duration < run_duration_ms) && (endtest == false) ){
162  //call the gemm blas
163  gpu_blas->run_blass_gemm(iet_ops_type);
164 
165  /* Wait for the previous gemm operations to complete */
166  while (!gpu_blas->is_gemm_op_complete()) {}
167 
168  //get the end time
169  iet_end_time = std::chrono::system_clock::now();
170  //Duration in the call
171  duration = time_diff(iet_end_time, iet_start_time);
172  gem_ops++;
173 
174  //Converting microseconds to seconds
175  timetakenforoneiteration = duration/1e6;
176  //calculating Gemm count
177  gflops_interval = gpu_blas->gemm_gflop_count()/timetakenforoneiteration;
178  //Print the gflops interval
179  log_interval_gflops(gflops_interval);
180  // check end test to avoid unnecessary sleep
181  if (endtest)
182  break;
183  //if gemm ops greater than 10000, lets yield
184  //if this is not happening we are ending up in
185  //out of memmory state
186  if(gem_ops > 10000) {
187  sleep(1);
188  gem_ops = 0;
189  }
190  }
191 
192 }

200 bool IETWorker::do_iet_power_stress(void) {
201 
202  std::chrono::time_point<std::chrono::system_clock> iet_start_time, end_time,
203  sampling_start_time;
204  uint64_t total_time_ms;
205  uint64_t last_avg_power;
206  string msg;
207  float cur_power_value = 0;
208  float totalpower = 0;
209  float max_power = 0;
210  bool result = true;
211  bool start = true;
212 
213  std::thread t(&IETWorker::blasThread,this, gpu_device_index, matrix_size_a, iet_ops_type, start, run_duration_ms,
214  iet_trans_a, iet_trans_b, iet_alpha_val, iet_beta_val, iet_lda_offset, iet_ldb_offset, iet_ldc_offset);
215 
216  // record EDPp ramp-up start time
217  iet_start_time = std::chrono::system_clock::now();
218 
219  for (;;) {
220  // check if stop signal was received
221  if (rvs::lp::Stopping())
222  break;
223  // get GPU's current average power
224  rsmi_status_t rmsi_stat = rsmi_dev_power_ave_get(smi_device_index , 0,
225  &last_avg_power);
226 
227  if (rmsi_stat == RSMI_STATUS_SUCCESS) {
228  cur_power_value = static_cast<float>(last_avg_power)/1e6;
229  }
230 
231  msg = "[" + action_name + "] " + MODULE_NAME + " " +
232  std::to_string(gpu_id) + " " + " Target power is : " + " " + std::to_string(target_power);
233  rvs::lp::Log(msg, rvs::logtrace);
234 
235  //update power to max if it is valid
236  if(cur_power_value > 0){
237  max_power = std::max(max_power, cur_power_value);// max of averages
238  }
239 
240  end_time = std::chrono::system_clock::now();
241 
242  total_time_ms = time_diff(end_time, iet_start_time);
243 
244  msg = "[" + action_name + "] " + MODULE_NAME + " " +
245  std::to_string(gpu_id) + " " + " Average power " + " " + std::to_string(cur_power_value);
246  rvs::lp::Log(msg, rvs::loginfo);
247 
248  msg = "[" + action_name + "] " + MODULE_NAME + " " +
249  std::to_string(gpu_id) + " " + " Total time in ms " + " " + std::to_string(total_time_ms) +
250  " Run duration in ms " + " " + std::to_string(run_duration_ms);
251  rvs::lp::Log(msg, rvs::logtrace);
252 
253  if (total_time_ms > run_duration_ms) {
254  break;
255  }
256 
257  //It doesnt make sense to read power continously so slowing down
258  sleep(1000);
259 
260  // check if stop signal was received
261  if (rvs::lp::Stopping()) {
262  result = true;
263  goto end;
264  }
265  }
266 
267  // json log the avg power
268  log_to_json(IET_AVERAGE_POWER_KEY, std::to_string(max_power),
269  rvs::loginfo);
270  //check whether we reached the target power
271  if(max_power >= target_power) {
272  msg = "[" + action_name + "] " + MODULE_NAME + " " +
273  std::to_string(gpu_id) + " " + " Average power met the target power :" + " " + std::to_string(max_power);
274  rvs::lp::Log(msg, rvs::loginfo);
275  result = true;
276  }else{
277  msg = "[" + action_name + "] " + MODULE_NAME + " " +
278  std::to_string(gpu_id) + " " + " Average power couldnt meet the target power \
279  in the given interval, increase the duration and try again, \
280  Average power is :" + " " + std::to_string(max_power);
281  rvs::lp::Log(msg, rvs::loginfo);
282  result = false;
283  }
284 
285  msg = "[" + action_name + "] " + MODULE_NAME + " " +
286  std::to_string(gpu_id) + " " + " End of worker thread " ;
287  rvs::lp::Log(msg, rvs::loginfo);
288 
289 end:
290 
291  endtest = true;
292 
293  if (true == t.joinable()) {
294 
295  try {
296  t.join();
297  }
298  catch (std::exception& e) {
299  std::cout << "Standard exception: " << e.what() << std::endl;
300  }
301  }
302  return result;
303 }

312  msg = "[" + action_name + "] " + MODULE_NAME + " " +
