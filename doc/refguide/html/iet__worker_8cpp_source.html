119 bool IETWorker::do_gpu_init_training(string *err_description) {
120  std::chrono::time_point<std::chrono::system_clock> start_time, end_time;
121  float cur_power_value;
122  uint64_t power_sampling_iters = 0;
123 
124  // init with no error
125  *err_description = "";
126 
127  // let the GPU run SGEMMs for MAX_MS_TRAIN_GPU ms (e.g.: 1000) and:
128  // 1. get the number of SGEMMs the GPU managed to run (needed in order
129  // to detect/change the SGEMMs frequency)
130  // 2. get the max power
131  num_sgemms_training = 0;
132  avg_power_training = 0;
133 
134  gpu_worker = std::unique_ptr<blas_worker>(
135  new blas_worker(gpu_device_index, matrix_size));
136  if (gpu_worker == nullptr) {
137  *err_description = IET_MEM_ALLOC_ERROR;
138  return false;
139  }
140  if (gpu_worker->get_blas_error()) {
141  *err_description = IET_BLAS_FAILURE;
142  return false;
143  }
144  gpu_worker->set_sgemm_delay(0);
145  gpu_worker->set_bcount_sgemm(true);
146 
147  // start the SGEMM workload
148  gpu_worker->start();
149 
150  // wait for the BLAS setup to complete
151  while (!gpu_worker->is_setup_complete()) {}
152  if (gpu_worker->get_blas_error()) {
153  *err_description = IET_BLAS_FAILURE;
154  return false;
155  }
156 
157  // record inital time
158  start_time = std::chrono::system_clock::now();
159 
160  for (;;) {
161  // check if stop signal was received
162  if (rvs::lp::Stopping())
163  return false;
164 
165  // get power data
166  cur_power_value = get_power_data(gpu_hwmon_entry);
167  if (cur_power_value != 0) {
168  avg_power_training += cur_power_value;
169  power_sampling_iters++;
170  }
171  usleep(POWER_PROCESS_DELAY);
172 
173  end_time = std::chrono::system_clock::now();
174  uint64_t diff_ms = time_diff(end_time, start_time);
175  if (diff_ms >= MAX_MS_TRAIN_GPU) {
176  // wait for the last sgemm to finish
177  while (!gpu_worker->is_sgemm_complete()) { }
178  // record the actual training time
179  end_time = std::chrono::system_clock::now();
180  training_time_ms = time_diff(end_time, start_time);
181  // stop the training
182  break;
183  }
184  }
185 
186  // gather the GPUS stats
187  num_sgemms_training = gpu_worker->get_num_sgemm_ops();
188  if (num_sgemms_training == 0) {
189  *err_description = IET_SGEMM_FAILURE;
190  return false;
191  }
192 
193  if (power_sampling_iters != 0) {
194  avg_power_training /= power_sampling_iters;
195  if (avg_power_training > 0)
196  return true;
197 
198  *err_description = IET_POWER_PROC_ERROR;
199  return false;
200  }
201 
202  *err_description = IET_POWER_PROC_ERROR;
203  return false;
204 }

209 void IETWorker::compute_gpu_stats(void) {
210  float ms_per_sgemm, sgemm_target_power;
211  float sgemm_target_power_si, total_ms_sgemm_si;
212 
213  // compute SGEMM time (ms)
214  ms_per_sgemm = static_cast<float>(training_time_ms) / num_sgemms_training;
215  // compute required number of SGEMM for the given target_power
216  sgemm_target_power =
217  (target_power * num_sgemms_training) / avg_power_training;
218  sgemm_target_power_si =
219  (sample_interval * sgemm_target_power) / training_time_ms;
220  // compute the actual SGEMM frequency for the given target_power
221  total_ms_sgemm_si = sgemm_target_power_si * ms_per_sgemm;
222  sgemm_si_delay = sample_interval - total_ms_sgemm_si;
223  if (sgemm_si_delay < 0) {
224  sgemm_si_delay = 0;
225  } else {
226  if (sgemm_target_power_si > 0)
227  sgemm_si_delay /= sgemm_target_power_si;
228  else
229  sgemm_target_power_si = sample_interval;
230 
231  sgemm_si_delay = sgemm_si_delay + sgemm_si_delay / SGEMM_DELAY_FREQ_DEV;
232  }
233 }

240 void IETWorker::compute_new_sgemm_freq(float avg_power) {
241  // compute the difference between the actual power data and the target_power
242  float diff_power = avg_power - target_power;
243  // gradually & dynamically increase/decrease the SGEMM frequency
244  float sgemm_delay_dev = (abs(diff_power) * sgemm_si_delay) / target_power;
245  if (diff_power < 0) {
246  if (sgemm_si_delay - sgemm_delay_dev < 0)
247  sgemm_si_delay = 1;
248  else
249  sgemm_si_delay -= sgemm_delay_dev;
250  } else {
251  sgemm_si_delay += sgemm_delay_dev;
252  }
253 }

263 bool IETWorker::do_iet_ramp(int *error, string *err_description) {
264  std::chrono::time_point<std::chrono::system_clock> iet_start_time, end_time,
265  sampling_start_time;
266  float cur_power_value, avg_power = 0;
267  uint64_t power_sampling_iters = 0, cur_milis_sampling;
268  string msg;
269 
270  *error = 0;
271  *err_description = "";
272 
273  if (!do_gpu_init_training(err_description)) {
274  *error = 1;
275  return false;
276  }
277 
278  pwr_log_worker = std::unique_ptr<log_worker>(
279  new log_worker(IETWorker::bjson));
280  if (pwr_log_worker == nullptr) {
281  *error = 1;
282  *err_description = IET_MEM_ALLOC_ERROR;
283  return false;
284  }
285 
286  pwr_log_worker->set_name(action_name);
287  pwr_log_worker->set_gpu_hwmon_entry(gpu_hwmon_entry);
288  pwr_log_worker->set_gpu_id(gpu_id);
289  pwr_log_worker->set_log_interval(log_interval);
290 
291  compute_gpu_stats();
292 
293  gpu_worker->pause();
294  // let the BLAS worker complete the last SGEMM
295  usleep(MAX_MS_WAIT_BLAS_THREAD);
296  gpu_worker->set_sgemm_delay(sgemm_si_delay * 1000);
297 
298  // record EDPp ramp-up start time
299  iet_start_time = std::chrono::system_clock::now();
300  sampling_start_time = std::chrono::system_clock::now();
301 
302  // restart the worker
303  gpu_worker->resume();
304  pwr_log_worker->start();
305 
306  for (;;) {
307  // check if stop signal was received
308  if (rvs::lp::Stopping())
309  return false;
310 
311  // get GPU's current average power
312  cur_power_value = get_power_data(gpu_hwmon_entry);
313  if (cur_power_value != 0) {
314  avg_power += cur_power_value;
315  power_sampling_iters++;
316  }
317 
318  end_time = std::chrono::system_clock::now();
319  cur_milis_sampling = time_diff(end_time, sampling_start_time);
320  if (cur_milis_sampling >= sample_interval &&
321  gpu_worker->is_sgemm_complete()) {
322  gpu_worker->pause();
323  // it's sampling time => check the power value against target_power
324  if (power_sampling_iters != 0) {
325  avg_power /= power_sampling_iters;
326  if (!(avg_power >= target_power - tolerance * target_power &&
327  avg_power <= target_power + tolerance * target_power)) {
328  // compute the new SGEMMs frequency
329  compute_new_sgemm_freq(avg_power);
330  // set the new SGEMM frequency
331  gpu_worker->set_sgemm_delay(sgemm_si_delay * 1000);
332 
333  } else {
334  ramp_actual_time = training_time_ms +
335  time_diff(end_time, iet_start_time);
336  return true;
337  }
338  }
339 
340  avg_power = 0;
341  power_sampling_iters = 0;
342  sampling_start_time = std::chrono::system_clock::now();
343  gpu_worker->resume();
344  }
345 
346  cur_milis_sampling = time_diff(end_time, iet_start_time);
347  if (cur_milis_sampling > ramp_interval - training_time_ms)
348  return false;
349 
350  usleep(POWER_PROCESS_DELAY);
351  }
352 }

360 bool IETWorker::do_iet_power_stress(void) {
361  std::chrono::time_point<std::chrono::system_clock> iet_start_time, end_time,
362  sampling_start_time;
363  float cur_power_value, avg_power = 0;
364  uint64_t power_sampling_iters = 0, cur_milis_sampling, total_time_ms;
365  uint16_t num_power_violations = 0;
366  string msg;
367 
368  // record EDPp ramp-up start time
369  iet_start_time = std::chrono::system_clock::now();
370  sampling_start_time = std::chrono::system_clock::now();
371 
372  // restart the worker
373  gpu_worker->resume();
374 
375  for (;;) {
376  // check if stop signal was received
377  if (rvs::lp::Stopping())
378  break;
379 
380  // get GPU's current average power
381  cur_power_value = get_power_data(gpu_hwmon_entry);
382  if (cur_power_value != 0) {
383  avg_power += cur_power_value;
384  power_sampling_iters++;
385  }
386 
387  end_time = std::chrono::system_clock::now();
388  cur_milis_sampling = time_diff(end_time, sampling_start_time);
389  if (cur_milis_sampling >= sample_interval &&
390  gpu_worker->is_sgemm_complete()) {
391  gpu_worker->pause();
392  // it's sampling time => check the power value against target_power
393  if (power_sampling_iters != 0) {
394  avg_power /= power_sampling_iters;
395  if (!(avg_power >= target_power - tolerance * target_power &&
396  avg_power <= target_power + tolerance * target_power)) {
397  // detected a target_power violation
398  num_power_violations++;
399  msg = "[" + action_name + "] " + MODULE_NAME + " " +
400  std::to_string(gpu_id) + " " + IET_PWR_VIOLATION_MSG +
401  " " + std::to_string(avg_power);
402  log(msg.c_str(), rvs::loginfo);
403  log_to_json(IET_PWR_VIOLATION_MSG,
404  std::to_string(avg_power), rvs::loginfo);
405  }
406  }
407 
408  avg_power = 0;
409  power_sampling_iters = 0;
410  sampling_start_time = std::chrono::system_clock::now();
411  gpu_worker->resume();
412  }
413 
414  total_time_ms = time_diff(end_time, iet_start_time);
415  if (total_time_ms > run_duration_ms - ramp_actual_time)
416  break;
417 
418  usleep(POWER_PROCESS_DELAY);
419  }
420 
421  pwr_log_worker->stop();
422 
423  gpu_worker->stop();
424  usleep(MAX_MS_WAIT_BLAS_THREAD);
425  gpu_worker->join();
426 
427  // check if stop signal was received
428  if (rvs::lp::Stopping())
429  return false;
430 
431  if (num_power_violations > max_violations)
432  return false;
433 
434  return true;
435 }

440 void IETWorker::run() {
441  string msg, err_description;
442  int error;
443 
444  msg = "[" + action_name + "] " + MODULE_NAME + " " +
445  std::to_string(gpu_id) + " start " + std::to_string(target_power);
446  log(msg.c_str(), rvs::loginfo);
447  log_to_json("start", std::to_string(target_power), rvs::loginfo);
448 
449  if (ramp_interval < MAX_MS_TRAIN_GPU)
450  ramp_interval += MAX_MS_TRAIN_GPU;
451  if (run_duration_ms < MAX_MS_TRAIN_GPU)
452  run_duration_ms += MAX_MS_TRAIN_GPU;
453 
454  if (!do_iet_ramp(&error, &err_description)) {
455  if (gpu_worker != nullptr) {
456  // terminate the blas worker thread
457  gpu_worker->stop();
458  usleep(MAX_MS_WAIT_BLAS_THREAD);
459  gpu_worker->join();
460  }
461 
462  if (pwr_log_worker != nullptr)
463  pwr_log_worker->stop();
464 
465  // check if stop signal was received
466  if (rvs::lp::Stopping())
467  return;
468 
469  if (error) {
470  log_to_json("ERROR", err_description, rvs::logerror);
471  msg = "[" + action_name + "] " + MODULE_NAME + " "
472  + std::to_string(gpu_id) + " " + err_description;
473  log(msg.c_str(), rvs::logerror);
474  } else {
475  log_to_json(IET_PWR_RAMP_EXCEEDED_MSG,
476  std::to_string(ramp_interval), rvs::loginfo);
477 
478  msg = "[" + action_name + "] " + MODULE_NAME + " " +
479  std::to_string(gpu_id) + " " + IET_PWR_RAMP_EXCEEDED_MSG + " " +
480  std::to_string(ramp_interval);
481  log(msg.c_str(), rvs::loginfo);
482  }
483 
484  msg = "[" + action_name + "] " + MODULE_NAME + " " +
485  std::to_string(gpu_id) + " " + IET_PASS_KEY + ": " +
486  IET_RESULT_FAIL_MESSAGE;
487  log(msg.c_str(), rvs::logresults);
488 
489  log_to_json(IET_PASS_KEY, IET_RESULT_FAIL_MESSAGE, rvs::logresults);
490 
491  } else {
492  // the GPU succeeded in achieving the given target_power
493  // => log a message and start the sustained stress test
494  msg = "[" + action_name + "] " + MODULE_NAME + " " +
495  std::to_string(gpu_id) + " " + IET_PWR_TARGET_ACHIEVED_MSG +
496  " " + std::to_string(target_power);
497  log(msg.c_str(), rvs::loginfo);
498  log_to_json(IET_PWR_TARGET_ACHIEVED_MSG,
499  std::to_string(target_power), rvs::loginfo);
500 
501 
502  bool pass = do_iet_power_stress();
503 
504  // check if stop signal was received
505  if (rvs::lp::Stopping())
506  return;
507 
508  msg = "[" + action_name + "] " + MODULE_NAME + " " +
509  std::to_string(gpu_id) + " " + IET_PASS_KEY + ": " +
510  (pass ? IET_RESULT_PASS_MESSAGE : IET_RESULT_FAIL_MESSAGE);
511  log(msg.c_str(), rvs::logresults);
512  log_to_json(IET_PASS_KEY,
513  (pass ? IET_RESULT_PASS_MESSAGE : IET_RESULT_FAIL_MESSAGE),
514  rvs::logresults);
515  }
