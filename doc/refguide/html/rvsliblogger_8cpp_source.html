178 int rvs::logger::LogExt(const char* Message, const int LogLevel,
179  const unsigned int Sec, const unsigned int uSec) {
180  DTRACE_
181  // stop logging requested?
182  if (bStop) {
183  DTRACE_
184  if (stop_flags) {
185  DTRACE_
186  // just return
187  return 0;
188  }
189  }
190 
191  if (LogLevel < lognone || LogLevel > logtrace) {
192  DTRACE_
193  char buff[128];
194  snprintf(buff, sizeof(buff), "unknown logging level: %d", LogLevel);
195  Err(buff, "CLI");
196  return -1;
197  }
198 
199  // log level too high?
200  if (LogLevel > loglevel_m) {
201  DTRACE_
202  return 0;
203  }
204 
205  uint32_t secs = 0;
206  uint32_t usecs = 0;
207 
208  if (Sec|uSec) {
209  DTRACE_
210  secs = Sec;
211  usecs = uSec;
212  } else {
213  DTRACE_
214  get_ticks(&secs, &usecs);
215  }
216 
217  DTRACE_
218  char buff[64];
219  snprintf(buff, sizeof(buff), "%6d.%-6d", secs, usecs);
220 
221  std::string row("[");
222  row += loglevelname[LogLevel];
223  row +="] [";
224  row += buff;
225  row +="] ";
226  row += Message;
227 
228  // if no quiet option given, output to cout
229  if (!b_quiet) {
230  DTRACE_
231  // lock cout_mutex for the duration of this block
232  std::lock_guard<std::mutex> lk(cout_mutex);
233  cout << row << '\n';
234  }
235 
236  // this stream does not output JSON
237  if (to_json()) {
238  DTRACE_
239  return 0;
240  }
241 
242  DTRACE_
243  // send to file if requested
244  if (isfirstrecord_m) {
245  DTRACE_
246  isfirstrecord_m = false;
247  } else {
248  DTRACE_
249  row = RVSENDL + row;
250  }
251  DTRACE_
252 
253  if (true) {
254  // lock log_mutex for the duration of this block
255  std::lock_guard<std::mutex> lk(log_mutex);
256  ToFile(row);
257  }
258 
259  DTRACE_
260  return 0;
261 }

290 void* rvs::logger::LogRecordCreate(const char* Module, const char* Action,
291  const int LogLevel, const unsigned int Sec,
292  const unsigned int uSec, bool minimal) {
293  uint32_t sec;
294  uint32_t usec;
295  if( json_log_file.empty()){
296  json_log_file = json_filename(Module);
297  std::lock_guard<std::mutex> lk(cout_mutex);
298  std::cout << "json log file is " << json_log_file<< std::endl;
299  }
300  if( minimal){
301  rvs::MinNode* minrec = new rvs::MinNode(Action, LogLevel);
302  return static_cast<void*>(minrec);
303  }
304  if ((Sec|uSec)) {
305  sec = Sec;
306  usec = uSec;
307  } else {
308  get_ticks(&sec, &usec);
309  }
310 
311  rvs::LogNodeRec* rec = new LogNodeRec(Action, LogLevel, sec, usec);
312  AddString(rec, "action", Action);
313  AddString(rec, "module", Module);
314  AddString(rec, "loglevelname", (LogLevel >= lognone && LogLevel < logtrace) ?
315  loglevelname[LogLevel] : "UNKNOWN");
316 
317  return static_cast<void*>(rec);
318 }

392 int rvs::logger::LogRecordFlush(void* pLogRecord, bool minimal) {
393  // lock log_mutex for the duration of this block
394  std::lock_guard<std::mutex> lk(json_log_mutex);
395  std::string val;
396  LogNode *r = nullptr;
397  DTRACE_
398  if(minimal){
399  r = static_cast<MinNode*>(pLogRecord);
400  }else{
401  r = static_cast<LogNodeRec*>(pLogRecord);
402  }
403  // no JSON loggin requested
404  if (!to_json()) {
405  DTRACE_
406  delete r;
407  return 0;
408  }
409 
410  // assert log levl
411  int level = r->LogLevel();
412  if (level < lognone || level > logtrace) {
413  char buff[128];
414  snprintf(buff, sizeof(buff), "unknown logging level: %d", r->LogLevel());
415  Err(buff, "CLI");
416  delete r;
417  return -1;
418  }
419  // if too high, ignore record
420  if (level > loglevel_m) {
421  DTRACE_
422  delete r;
423  return 0;
424  }
425 
426  // do not pre-pend "," separator for the first row
427  std::string row;
428  if (append_m) {
429  DTRACE_
430  row = ",";
431  } else {
432  DTRACE_
433  if (!isfirstrecord_m) {
434  DTRACE_
435  row = ",";
436  }
437  }
438  DTRACE_
439  // get JSON formatted log record
440  row += r->ToJson(" ");
441 
442  // send it to file
443  ToFile(row, true);
444 
445  // dealloc memory
446  delete r;
447 
448  if (isfirstrecord_m) {
449  DTRACE_
450  isfirstrecord_m = false;
451  }
452  DTRACE_
453  // return OK
454  return 0;
455 }

587 int rvs::logger::init_log_file() {
588  isfirstrecord_m = true;
589  bStop = false;
590  stop_flags = 0;
591 
592  std::string row;
593  std::string logfile(log_file);
594 
595  // if no logg to file requested, just return
596  if (logfile == "")
597  return 0;
598 
599  if (append()) {
600  // appnd to file, replace the closing "]" with "," in order to
601  // have well formed JSON after appending
602  int patch_status = -1;
603 
604  if (to_json()) {
605  int sts = JsonPatchAppend(&patch_status);
606  if (sts) {
607  return -1;
608  }
609  }
610  } else {
611  // logging but not appending - just truncate the file.
612  std::fstream fs;
613  fs.open(logfile, std::fstream::out);
614  bool berror = fs.fail();
615  fs.close();
616  if (berror) {
617  return -1;
618  }
619  if (to_json()) {
620  row = "[";
621  }
622  }
623 
624  // print to log file if requested
625  ToFile(row);
626 
627  return 0;
