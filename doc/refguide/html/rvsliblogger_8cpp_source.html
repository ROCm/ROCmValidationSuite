172 int rvs::logger::LogExt(const char* Message, const int LogLevel,
173  const unsigned int Sec, const unsigned int uSec) {
174  // lock cout_mutex for the duration of this block
175  std::lock_guard<std::mutex> lk(cout_mutex);
176 
177  // stop logging requested?
178  if (bStop) {
179  if (stop_flags)
180  // just return
181  return 0;
182  }
183 
184  if (LogLevel < lognone || LogLevel > logtrace) {
185  cerr << "ERROR: unknown logging level: " << LogLevel << '\n';
186  return -1;
187  }
188 
189  // log level too high?
190  if (LogLevel > loglevel_m)
191  return 0;
192 
193  uint32_t secs;
194  uint32_t usecs;
195 
196  if (Sec|uSec) {
197  secs = Sec;
198  usecs = uSec;
199  } else {
200  get_ticks(&secs, &usecs);
201  }
202 
203  char buff[64];
204  snprintf(buff, sizeof(buff), "%6d.%-6d", secs, usecs);
205 
206  std::string row("[");
207  row += loglevelname[LogLevel];
208  row +="] [";
209  row += buff;
210  row +="] ";
211  row += Message;
212 
213  // if no quiet option given, output to cout
214  if (!rvs::options::has_option("-q")) {
215  cout << row << '\n';
216  }
217 
218  // this stream does not output JSON
219  if (to_json())
220  return 0;
221 
222  // send to file if requested
223  if (isfirstrecord_m) {
224  isfirstrecord_m = false;
225  } else {
226  row = RVSENDL + row;
227  }
228  ToFile(row);
229 
230  return 0;
231 }

246 void* rvs::logger::LogRecordCreate(const char* Module, const char* Action,
247  const int LogLevel, const unsigned int Sec,
248  const unsigned int uSec) {
249  uint32_t sec;
250  uint32_t usec;
251 
252  if ((Sec|uSec)) {
253  sec = Sec;
254  usec = uSec;
255  } else {
256  get_ticks(&sec, &usec);
257  }
258 
259  rvs::LogNodeRec* rec = new LogNodeRec(Action, LogLevel, sec, usec);
260  AddString(rec, "action", Action);
261  AddString(rec, "module", Module);
262  AddString(rec, "loglevelname", (LogLevel >= lognone && LogLevel < logtrace) ?
263  loglevelname[LogLevel] : "UNKNOWN");
264 
265  return static_cast<void*>(rec);
266 }

277 int rvs::logger::LogRecordFlush(void* pLogRecord) {
278  std::string val;
279  DTRACE_
280 
281  LogNodeRec* r = static_cast<LogNodeRec*>(pLogRecord);
282  // no JSON loggin requested
283  if (!to_json()) {
284  DTRACE_
285  delete r;
286  return 0;
287  }
288 
289  // assert log levl
290  int level = r->LogLevel();
291  if (level < lognone || level > logtrace) {
292  cerr << "ERROR: unknown logging level: " << level << '\n';
293  delete r;
294  return -1;
295  }
296 
297  // if too high, ignore record
298  if (level > loglevel_m) {
299  DTRACE_
300  delete r;
301  return 0;
302  }
303 
304  // do not pre-pend "," separator for the first row
305  std::string row;
306  if (isfirstrecord_m) {
307  DTRACE_
308  isfirstrecord_m = false;
309  } else {
310  DTRACE_
311  row = ",";
312  }
313 
314  // get JSON formatted log record
315  row += r->ToJson(" ");
316 
317  // send it to file
318  ToFile(row);
319 
320  // dealloc memory
321  delete r;
322 
323  DTRACE_
324  // return OK
325  return 0;
326 }

370 int rvs::logger::JsonPatchAppend() {
371  std::string logfile;
372  if (!rvs::options::has_option("-l", &logfile))
373  return -1;
374 
375  FILE * pFile;
376  pFile = fopen(logfile.c_str() , "r+");

381  fputs("," , pFile);
382  fclose(pFile);

457 int rvs::logger::initialize() {
458  isfirstrecord_m = true;
459  bStop = false;
460  stop_flags = 0;
461 
462  std::string row;
463  std::string logfile;
464 
465  // if no logg to file requested, just return
466  if (!rvs::options::has_option("-l", &logfile))
467  return 0;
468 
469  if (append()) {
470  // appnd to file, replace the closing "]" with "," in order to
471  // have well formed JSON after appending
472  int patch_status = -1;
473 
474  if (to_json()) {
475  patch_status = JsonPatchAppend();
476  if (patch_status) {
477  row += "[";
478  }
479  }
480  } else {
481  // logging but not appending - just truncate the file.
482  std::fstream fs;
483  fs.open(logfile, std::fstream::out);
484  fs.close();
485  if (to_json()) {
486  row = "[";
487  }
488  }
489 
490  // print to log file if requested
491  ToFile(row);
492 
493  return 0;
494 }
