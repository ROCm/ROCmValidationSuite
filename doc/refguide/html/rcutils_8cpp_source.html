28 OSType searchos(std::string os_name){
29  std::string lowcasename{os_name};
30  std::transform( lowcasename.begin(), lowcasename.end(), lowcasename.begin(),
31  [](unsigned char c){ return std::tolower(c); });
32  for( auto itr = op_systems.begin();itr !=op_systems.end();++itr){
33  if(lowcasename.find(itr->first) != std::string::npos){
34  return itr->second;

36  }
37  return OSType::None;

40 OSType getOS(){
41  std::ifstream rel_file(os_release_file.c_str());
42  if(!rel_file.good()){
43  std::cout << "No /etc/os-release file, cant fetch details " << std::endl;
44  return OSType::None;
45  }
46  std::string line;
47  while (std::getline(rel_file, line))
48  {
49  auto found = line.find(name_key) ;
50  if (found!=std::string::npos){
51  found = line.find('\"');
52  auto endquote = line.find_last_of('\"');
53  if(found == std::string::npos || endquote == std::string::npos)
54  return OSType::None;
55  std::string osame = line.substr(found+1, endquote-found-1 );
56  return searchos(osame);
57  }
58  }
59 }

88 bool getPackageInfo(const std::string& package,
89  const std::string& packagemgr,
90  const std::string& command,
91  const std::string& option,
92  std::stringstream &ss){
93 
94  int read_pipe[2]; // From child to parent
95  if(pipe(read_pipe) == -1){
96  perror("Pipe");
97  return false;
98  }
99 
100  pid_t process_id = fork();
101  if(process_id < 0){
102  perror("Fork");
103  return false;
104  }
105  else if(process_id == 0) {
106 
107  /* Child process */
108  dup2(read_pipe[1], 1);
109  close(read_pipe[0]);
110  close(read_pipe[1]);
111 
112  /* Apply options in command only if present */
113  if(option.empty()) {
114  execlp(packagemgr.c_str(), packagemgr.c_str(), command.c_str(), package.c_str(), NULL);
115  }
116  else {
117  execlp(packagemgr.c_str(), packagemgr.c_str(), command.c_str(), option.c_str(), package.c_str(), NULL);
118  }
119  }
120  else {
121  /* Parent process */
122  int status;
123  waitpid(process_id, &status,0);
124  close(read_pipe[1]);
125  {
126  char arr[8192];
127  int n = read(read_pipe[0], arr, sizeof(arr));
128  ss.write(arr, n);
129 
130  }
131  close(read_pipe[0]);
132  return true;
133  }
134 }
