95 int pebb_action::run() {
96  string msg;
97  std::chrono::time_point<std::chrono::system_clock> pebb_start_time;
98  std::chrono::time_point<std::chrono::system_clock> pebb_end_time;
99 
100  RVSTRACE_
101  if (property.find("cli.-j") != property.end()) {
102  bjson = true;
103  }
104 
105  if (!get_all_common_config_keys())
106  return -1;
107  if (!get_all_pebb_config_keys())
108  return -1;
109 
110  // log_interval must be less than duration
111  if (property_log_interval > 0 && property_duration > 0) {
112  if (property_log_interval > property_duration) {
113  msg = "log_interval must be less than duration";
114  rvs::lp::Err(msg, MODULE_NAME_CAPS, action_name);
115  return -1;
116  }
117  }
118 
119  test_duration = property_duration;
120  if(bjson){
121  json_add_primary_fields();
122  }
123  int sts = create_threads();
124 
125  if (sts != 0) {
126  return sts;
127  }
128  // define timers
129  rvs::timer<pebb_action> timer_running(&pebb_action::do_running_average, this);
130  rvs::timer<pebb_action> timer_final(&pebb_action::do_final_average, this);
131 
132  unsigned int iter = property_count > 0 ? property_count : 1;
133  unsigned int step = 1;
134  int count = 0;
135 
136  do {
137  // let the test run in this iteration
138  brun = true;
139  count = 0;
140 
141  // start timers
142  if (property_duration) {
143  RVSTRACE_
144  timer_final.start(property_duration, true); // ticks only once
145  }
146 
147  if (property_log_interval) {
148  RVSTRACE_
149  timer_running.start(property_log_interval); // ticks continuously
150  }
151 
152  RVSTRACE_
153  pebb_start_time = std::chrono::system_clock::now();
154 
155  do {
156  if (property_parallel) {
157  sts = run_parallel();
158  } else {
159  sts = run_single();
160  }
161 
162  pebb_end_time = std::chrono::system_clock::now();
163  uint64_t test_time = time_diff(pebb_end_time, pebb_start_time) ;
164  if(test_time >= property_duration) {
165  pebb_action::do_final_average();
166  break;
167  }
168  } while(brun);
169 
170  RVSTRACE_
171  timer_running.stop();
172  timer_final.stop();
173 
174  std::cout << "\n Iteration value : " << iter;
175  iter -= step;
176 
177  // insert wait between runs if needed
178  if (iter > 0 && property_wait > 0) {
179  RVSTRACE_
180  sleep(property_wait);
181  }
182  } while (iter && !rvs::lp::Stopping());
183 
184  RVSTRACE_
185  sts = rvs::lp::Stopping() ? -1 : 0;
186 
187  print_final_average();
188 
189  std::cout << " \n =========================================================================================================================";
190  for(auto it = resultBandwidth.begin(); it != resultBandwidth.end(); it++) {
191  std::cout << " \n\n PCIE Bandwidth from , CPU::" << it->CPUId<< " to GPU Id::" << it->GPUId << " is " << it->finalBandwith << "\n\n";
192  }
193  std::cout << " ========================================================================================================================= \n";
194 
195  destroy_threads();
196  //bjson = true;
197  if(bjson){
198  rvs::lp::JsonActionEndNodeCreate();
199  }
200  return sts;
201 }

214  // iterate through test array and invoke tests one by one

230  }
