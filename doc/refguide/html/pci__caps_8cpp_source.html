53 unsigned int pci_dev_find_cap_offset(struct pci_dev *dev, unsigned char cap,
54  unsigned char type) {
55  struct pci_cap * pcap = dev->first_cap;
56  while (pcap != NULL) {
57  if (pcap->id == cap && pcap->type == type)
58  return pcap->addr;
59  pcap = pcap->next;
60  }
61 
62  return 0;
63 }

71 void get_link_cap_max_speed(struct pci_dev *dev, char *buff) {
72  const char *link_max_speed;
73 
74  // get pci dev capabilities offset
75  unsigned int cap_offset = pci_dev_find_cap_offset(dev, PCI_CAP_ID_EXP,
76  PCI_CAP_NORMAL);
77 
78  if (cap_offset != 0) {
79  unsigned int pci_dev_lnk_cap = pci_read_long(dev,
80  cap_offset + PCI_EXP_LNKCAP);
81 
82  // using 1,2,3 & 4 instead of the dedicated constants
83  // (that can be found in pci_regs.h) is the ugly way of doing it
84  // however, this is because in some linux versions the #define stops
85  // at PCI_EXP_LNKCAP_SLS_5_0GB (no 8, 16)
86  switch (pci_dev_lnk_cap & PCI_EXP_LNKCAP_SLS) {
87  case 1:
88  link_max_speed = "2.5 GT/s";
89  break;
90  case 2:
91  link_max_speed = "5 GT/s";
92  break;
93  case 3:
94  link_max_speed = "8 GT/s";
95  break;
96  case 4:
97  link_max_speed = "16 GT/s";
98  break;
99  default:
100  link_max_speed = "Unknown speed";
101  }
102 
103  snprintf(buff, PCI_CAP_DATA_MAX_BUF_SIZE, "%s", link_max_speed);
104  } else {
105  snprintf(buff, PCI_CAP_DATA_MAX_BUF_SIZE, "%s", PCI_CAP_NOT_SUPPORTED);
106  }
107 }

198 void get_slot_pwr_limit_value(struct pci_dev *dev, char *buff) {
199  // get pci dev capabilities offset
200  unsigned int cap_offset = pci_dev_find_cap_offset(dev, PCI_CAP_ID_EXP,
201  PCI_CAP_NORMAL);
202  float pwr;
203 
204  if (cap_offset != 0) {
205  unsigned int slot_cap = pci_read_long(dev, cap_offset + PCI_EXP_SLTCAP);
206  unsigned char slot_pwr_limit_scale = (slot_cap & PCI_EXP_SLTCAP_SPLS)
207  >> 15;
208  u16 slot_pwr_limit_value = (slot_cap & PCI_EXP_SLTCAP_SPLV) >> 7;
209 
210  if (slot_pwr_limit_value > 0xEF) {
211  switch (slot_pwr_limit_value) {
212  // according to the PCI Express Base Specification Revision 3.0
213  case 0xF0:
214  pwr = 250.0;
215  break;
216  case 0xF1:
217  pwr = 270.0;
218  break;
219  case 0xF2:
220  pwr = 300.0;
221  break;
222  default:
223  pwr = -1.0;
224  // (F3h to FFh = Reserved for Slot Power Limit
225  // values above 300W)
226  }
227  } else {
228  // according to the PCI Express Base Specification Revision 3.0
229  pwr = static_cast<float>(slot_pwr_limit_value)
230  * pow(10, -slot_pwr_limit_scale);
231  }
232 
233  snprintf(buff, PCI_CAP_DATA_MAX_BUF_SIZE, "%0.3fW", pwr);
234 
235  } else {
236  snprintf(buff, PCI_CAP_DATA_MAX_BUF_SIZE, "%s", PCI_CAP_NOT_SUPPORTED);
237  }
238 }

286 void get_kernel_driver(struct pci_dev *dev, char *buff) {
287  char name[1024], *drv, *base;
288  int n;
289 
290  buff[0] = '\0';
291 
292  if (dev->access->method != PCI_ACCESS_SYS_BUS_PCI) {
293  return;
294  }
295 
296  base = pci_get_param(dev->access, const_cast<char *>("sysfs.path"));
297  if (!base || !base[0]) {
298  return;
299  }
300 
301  n = snprintf(name, sizeof(name), "%s/devices/%04x:%02x:%02x.%d/driver",
302  base, dev->domain, dev->bus, dev->dev, dev->func);
303  if (n < 0 || n >= static_cast<int>(sizeof(name))) {
304  return;
305  }
306 
307  n = readlink(name, buff, PCI_CAP_DATA_MAX_BUF_SIZE);
308  if (n < 0) {
309  return;
310  }
311 
312  if (n >= PCI_CAP_DATA_MAX_BUF_SIZE) {
313  return;
314  }
315 
316  buff[n] = 0;
317 
318  if ((drv = strrchr(buff, '/')) != NULL)
319  snprintf(buff, PCI_CAP_DATA_MAX_BUF_SIZE, "%s", drv + 1);
320 }

327 void get_dev_serial_num(struct pci_dev *dev, char *buff) {
328  unsigned int cap_offset_dsn = pci_dev_find_cap_offset(dev,
329  PCI_EXT_CAP_ID_DSN, PCI_CAP_EXTENDED);
330 
331  if (cap_offset_dsn != 0) {
332  unsigned int t1, t2;
333  t1 = pci_read_long(dev, cap_offset_dsn + 4);
334  t2 = pci_read_long(dev, cap_offset_dsn + 8);
335  snprintf(buff, PCI_CAP_DATA_MAX_BUF_SIZE,
336  "%02x-%02x-%02x-%02x-%02x-%02x-%02x-%02x", t2 >> 24,
337  (t2 >> 16) & 0xff, (t2 >> 8) & 0xff, t2 & 0xff, t1 >> 24,
338  (t1 >> 16) & 0xff, (t1 >> 8) & 0xff, t1 & 0xff);

352 void get_pwr_budgeting(struct pci_dev *dev, uint8_t pb_pm_state,
353  uint8_t pb_type, uint8_t pb_power_rail, char *buff) {
354  u16 i, w = 0;
355  u16 base, scale;
356  uint8_t pb_act_pm_state, pb_act_type, pb_act_power_rail;
357 
358  unsigned int cap_offset_pwbgd = pci_dev_find_cap_offset(dev,
359  PCI_EXT_CAP_ID_PWR, PCI_CAP_EXTENDED);
360 
361  snprintf(buff, PCI_CAP_DATA_MAX_BUF_SIZE, "%s", PCI_CAP_NOT_SUPPORTED);
362 
363  if (cap_offset_pwbgd != 0) {
364  i = 0;
365 
366  do {
367  pci_write_byte(dev, cap_offset_pwbgd + PCI_PWR_DSR, i);
368  w = pci_read_word(dev, cap_offset_pwbgd + PCI_PWR_DATA);
369 
370  if (!w)
371  return;
372 
373  pb_act_pm_state = PCI_PWR_DATA_PM_STATE(w);
374  pb_act_type = PCI_PWR_DATA_TYPE(w);
375  pb_act_power_rail = PCI_PWR_DATA_RAIL(w);
376 
377  if (pb_act_pm_state == pb_pm_state && pb_act_type == pb_type &&
378  pb_act_power_rail == pb_power_rail) {
379  base = PCI_PWR_DATA_BASE(w);
380  scale = PCI_PWR_DATA_SCALE(w);
381  snprintf(buff, PCI_CAP_DATA_MAX_BUF_SIZE, "%.3fW",
382  base * pow(10, -scale));
383  return;
384  }
385 
386  i++;
387  } while (1);
388  }
389 }

396 void get_pwr_curr_state(struct pci_dev *dev, char *buff) {
397  u16 pmcsr;
398  const char *type_s;
399 
400  // init output buffer with "not supported" message
401  snprintf(buff, PCI_CAP_DATA_MAX_BUF_SIZE, "%s", PCI_CAP_NOT_SUPPORTED);
402 
403  // fetch capability offset
404  unsigned int cap_offset = pci_dev_find_cap_offset(dev,
405  PCI_CAP_ID_PM, PCI_CAP_NORMAL);
406 
407  if (cap_offset == 0)
408  return;
409 
410  pmcsr = pci_read_word(dev, cap_offset + PCI_PM_CTRL);
411 
412  switch (pmcsr & PCI_PM_CTRL_STATE_MASK) {
413  case 0:
414  type_s = "D0";
415  break;
416  case 1:
417  type_s = "D1";
418  break;
419  case 2:
420  type_s = "D2";
421  break;
422  case 3:
423  type_s = "D3";
424  break;
425  }
426 
427  snprintf(buff, PCI_CAP_DATA_MAX_BUF_SIZE, "%s", type_s);
428 }

435 void get_atomic_op_routing(struct pci_dev *dev, char *buff) {
436  bool atomic_op_routing_enable = false;
437 
438  // get pci dev capabilities offset
439  unsigned int cap_offset = pci_dev_find_cap_offset(dev, PCI_CAP_ID_EXP,
440  PCI_CAP_NORMAL);
441 
442  if (cap_offset != 0) {
443  // get Capability version via
444  // PCI Express Capabilities Register (offset 02h)
445  u16 cap_flags = pci_read_word(dev, cap_offset + 2);
446 
447  // check if it's capability version 2
448  if (!((cap_flags & PCI_EXP_FLAGS_VERS) < 2)) {
449  u16 dev_ctl2_reg_val = pci_read_word(dev,
450  cap_offset + PCI_EXP_DEVCTL2);
451 
452  // hardcoded 0x0040 because PCI_EXP_DEVCTL2_ATOMIC_REQ
453  // is not present on all versions of pci_regs.h
454  atomic_op_routing_enable = static_cast<bool>(dev_ctl2_reg_val
455  & 0x0040);
456 
457  snprintf(buff, PCI_CAP_DATA_MAX_BUF_SIZE, "%s",
458  atomic_op_routing_enable ? "TRUE" : "FALSE");
459  } else {
460  snprintf(buff, PCI_CAP_DATA_MAX_BUF_SIZE, "%s",
461  PCI_CAP_NOT_SUPPORTED);
462  }
463  } else {
464  snprintf(buff, PCI_CAP_DATA_MAX_BUF_SIZE, "%s", PCI_CAP_NOT_SUPPORTED);
465  }
466 }

472 int64_t get_atomic_op_register_value(struct pci_dev *dev) {
473  unsigned char i, has_memory_bar = 0;
474 
475  // get pci dev capabilities offset
476  unsigned int cap_offset = pci_dev_find_cap_offset(dev, PCI_CAP_ID_EXP,
477  PCI_CAP_NORMAL);
478 
479  if (cap_offset != 0) {
480  // get Capability version via
481  // PCI Express Capabilities Register (offset 02h)
482  u16 cap_flags = pci_read_word(dev, cap_offset + 2);
483 
484  // check if it's capability version 2
485  if (!((cap_flags & PCI_EXP_FLAGS_VERS) < 2)) {
486  // check if the device has memory space BAR
487  // (basically it should have but let us be sure about it)
488  for (i = 0; i < MEM_BAR_MAX_INDEX + 1; i++)
489  if (dev->base_addr[i] && dev->size[i]) {
490  if (!(dev->base_addr[i] & PCI_BASE_ADDRESS_SPACE_IO)) {
491  has_memory_bar = 1;
492  break;
493  }
494  }
495 
496  if (has_memory_bar) {
497  return pci_read_long(dev, cap_offset + PCI_EXP_DEVCAP2);
498 
499  } else {
500  return -1;
501  }
502  } else {
503  return -1;
504  }
505  } else {
506  return -1;
507  }
508 }
