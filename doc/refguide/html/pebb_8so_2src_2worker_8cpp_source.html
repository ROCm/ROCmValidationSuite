145 int pebbworker::do_transfer() {
146  double duration;
147  int sts;
148  unsigned int startsec;
149  unsigned int startusec;
150  unsigned int endsec;
151  unsigned int endusec;
152  std::string msg;
153 
154  msg = "[" + action_name + "] pebb transfer " + std::to_string(src_node) + " "
155  + std::to_string(dst_node) + " ";
156 
157  rvs::lp::get_ticks(&startsec, &startusec);
158 
159  if (block_size.size() == 0) {
160  block_size = pHsa->size_list;
161  }
162 
163  for (size_t i = 0; brun && i < block_size.size(); i++) {
164  current_size = block_size[i];
165 
166  if (rvs::lp::Stopping()) {
167  return -1;
168  }
169  // if needed, swap source and destination
170  if (!prop_h2d && prop_d2h) {
171  sts = pHsa->SendTraffic(dst_node, src_node, current_size,
172  bidirect, &duration);
173  } else {
174  sts = pHsa->SendTraffic(src_node, dst_node, current_size,
175  bidirect, &duration);
176  }
177  if (sts) {
178  std::cerr << "RVS-PEBB: internal error, src: " << src_node
179  << " dst: " << dst_node
180  << " current size: " << current_size << " status "<< sts <<std::endl;
181  return sts;
182  }
183 
184  {
185  std::lock_guard<std::mutex> lk(cntmutex);
186  running_size += current_size;
187  running_duration += duration;
188  }
189  }
190 
191  rvs::lp::get_ticks(&endsec, &endusec);
192  rvs::lp::Log(msg + "start", rvs::logdebug, startsec, startusec);
193  rvs::lp::Log(msg + "finish", rvs::logdebug, endsec, endusec);
194 
195  return 0;
196 }
