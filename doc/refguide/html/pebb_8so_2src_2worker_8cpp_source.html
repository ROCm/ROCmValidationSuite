74 void pebbworker::run() {
75  std::chrono::time_point<std::chrono::system_clock> pebb_start_time;
76  std::chrono::time_point<std::chrono::system_clock> pebb_end_time;
77  std::string msg;

79  msg = "[" + action_name + "] pebb thread " + std::to_string(src_node) + " "
80  + std::to_string(dst_node) + " has started";
81  rvs::lp::Log(msg, rvs::logdebug);
82 
83  brun = true;
84 
85  pebb_start_time = std::chrono::system_clock::now();
86  do{
87  do_transfer();

91  uint64_t test_time = time_diff(pebb_end_time, pebb_start_time) ;
92

98  msg = "[" + action_name + "] pebb thread " + std::to_string(src_node) + " "

158 int pebbworker::do_transfer() {
159  double duration;
160  int sts = -1;
161  unsigned int startsec;
162  unsigned int startusec;
163  unsigned int endsec;
164  unsigned int endusec;
165 
166  RVSTRACE_
167 
168  brun = true;
169  if (loglevel >= rvs::logdebug)
170  rvs::lp::get_ticks(&startsec, &startusec);
171 
172  if (block_size.size() == 0) {
173  RVSTRACE_
174  block_size = pHsa->size_list;
175  }
176 
177  for (size_t i = 0; brun && i < block_size.size(); i++) {
178  RVSTRACE_
179  current_size = block_size[i];
180 
181  if (rvs::lp::Stopping()) {
182  RVSTRACE_
183  return -1;
184  }
185  // if needed, swap source and destination
186  if (!prop_h2d && prop_d2h) {
187  RVSTRACE_
188  sts = pHsa->SendTraffic(dst_node, src_node, current_size,
189  bidirect, &duration);
190  } else {
191  RVSTRACE_
192  sts = pHsa->SendTraffic(src_node, dst_node, current_size,
193  bidirect, &duration);
194  }
195  if (sts) {
196  std::string msg = "internal error, src: " + std::to_string(src_node)
197  + " dst: " +std::to_string(dst_node)
198  + " current size: " + std::to_string(current_size)
199  + " status "+ std::to_string(sts);
200  rvs::lp::Err(msg, MODULE_NAME, action_name);
201  return sts;
202  }
203 
204  {
205  RVSTRACE_
206  std::lock_guard<std::mutex> lk(cntmutex);
207  running_size += current_size;
208  running_duration += duration;
209  }
210  }
211 
212  RVSTRACE_
213  if (loglevel >= rvs::logdebug) {
214  RVSTRACE_
215  std::string msg;
216  msg = "[" + action_name + "] pebb transfer " + std::to_string(src_node)
217  + " " + std::to_string(dst_node) + " ";
218 
219  rvs::lp::get_ticks(&endsec, &endusec);
220  rvs::lp::Log(msg + "start", rvs::logdebug, startsec, startusec);
221  rvs::lp::Log(msg + "finish", rvs::logdebug, endsec, endusec);
222  }
223 
224  return 0;
225 }
