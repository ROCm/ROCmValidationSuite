139  return -1;
140  }
141  } else {
142  cerr << "RVS-PESM: action: " << property["name"] <<
143  " invalide 'deviceid' key value: " << sdevid << std::endl;
144  return -1;
145  }
146  }
147 
148  // check if GPU id filtering is requied
149  string sdev;
150  if (has_property("device", &sdev)) {
151  pworker->set_strgpuids(sdev);
152  if (sdev != "all") {
153  vector<string> sarr = str_split(sdev, YAML_DEVICE_PROP_DELIMITER);
154  vector<int> iarr;
155  int sts = rvs_util_strarr_to_intarr(sarr, &iarr);
156  if (sts < 0) {
157  cerr << "RVS-PESM: action: " << property["name"] <<
158  " invalide 'device' key value: " << sdev << std::endl;
159  return -1;
160  }
161  pworker->set_gpuids(iarr);
162  }
163  } else {
164  cerr << "RVS-PESM: action: " << property["name"] <<
165  " key 'device' not found" << std::endl;
166  return -1;
167  }
168 
169  // start worker thread
170  rvs::lp::Log("[" + property["name"]+ "] pesm starting Worker",
171  rvs::logtrace);
172  pworker->start();
173  sleep(2);
174 
175  rvs::lp::Log("[" + property["name"]+ "] pesm Monitoring started",
176  rvs::logtrace);
177  } else {
178  rvs::lp::Log("[" + property["name"]+
179  "] pesm property[\"monitor\"] != \"true\"", rvs::logtrace);
180  if (pworker) {
181  // (give thread chance to start)
182  sleep(2);
183  pworker->set_stop_name(property["name"]);
184  pworker->stop();
185  delete pworker;
186  pworker = nullptr;
187  }
188  rvs::lp::Log("[" + property["name"]+ "] pesm Monitoring stopped",
189  rvs::logtrace);
190  }
191  return 0;

204 int action::do_gpu_list() {
205  log("pesm in do_gpu_list()", rvs::logtrace);
206 
207  std::map<string, string>::iterator it;
208 
209  struct device_info {
210  std::string bus;
211  std::string name;
212  int32_t node_id;
213  int32_t gpu_id;
214  int32_t device_id;
215  };
216 
217  std::vector<struct device_info> gpu_info_list;
218 
219  struct pci_access* pacc;
220  struct pci_dev* dev;
221  char buff[1024];
222  char devname[1024];
223 
224  // get the pci_access structure
225  pacc = pci_alloc();
226  // initialize the PCI library
227  pci_init(pacc);
228  // get the list of devices
229  pci_scan_bus(pacc);
230 
231  int ix = 0;
232  // iterate over devices
233  for (dev = pacc->devices; dev; dev = dev->next) {
234  // fil in the info
235  pci_fill_info(dev, PCI_FILL_IDENT | PCI_FILL_BASES | PCI_FILL_CLASS
236  | PCI_FILL_EXT_CAPS | PCI_FILL_CAPS | PCI_FILL_PHYS_SLOT);
237 
238  // computes the actual dev's location_id (sysfs entry)
239  uint16_t dev_location_id =
240  ((((uint16_t)(dev->bus)) << 8) | (dev->func));
241 
242  // if not and AMD GPU just continue
243  int32_t node_id = rvs::gpulist::GetNodeIdFromLocationId(dev_location_id);
244  if (node_id < 0)
245  continue;
246 
247  int32_t gpu_id = rvs::gpulist::GetGpuId(dev_location_id);
248  if (gpu_id < 0)
249  continue;
250 
251  snprintf(buff, sizeof(buff), "%02X:%02X.%d", dev->bus, dev->dev, dev->func);
252 
253  string name;
254  name = pci_lookup_name(pacc, devname, sizeof(devname), PCI_LOOKUP_DEVICE,
255  dev->vendor_id, dev->device_id);
256 
257  struct device_info info;
258  info.bus = buff;
259  info.name = name;
260  info.node_id = node_id;
261  info.gpu_id = gpu_id;
262  info.device_id = dev->device_id;
263  gpu_info_list.push_back(info);
264 
265  ++ix;
266  }
267 
268  std::sort(gpu_info_list.begin(), gpu_info_list.end(),
269  [](const struct device_info& a, const struct device_info& b) {
270  return a.node_id < b.node_id; });
271 
272  if (!gpu_info_list.empty()) {
273  cout << "Supported GPUs available:\n";
274  for (const auto& info : gpu_info_list) {
275  cout << info.bus << " - GPU[" << std::setw(2) << info.node_id
276  << " - " << std::setw(5) << info.gpu_id << "] " << info.name
277  << " (Device " << info.device_id << ")\n";
278  }
279  } else {
280  cout << endl << "No supported GPUs available.\n";
281  }
282 
283  pci_cleanup(pacc);
284 
285  return 0;
286 }
