77 bool pbqt_action::property_get_peers(int *error) {
78  *error = 0; // init with 'no error'
79  auto it = property.find("peers");
80  if (it != property.end()) {
81  if (it->second == "all") {
82  return true;
83  } else {
84  // split the list of gpu_id
85  prop_peers = str_split(it->second,
86  YAML_DEVICE_PROP_DELIMITER);
87  if (prop_peers.empty()) {
88  *error = 1; // list of gpu_id cannot be empty
89  } else {
90  for (vector<string>::iterator it_gpu_id =
91  prop_peers.begin();
92  it_gpu_id != prop_peers.end(); ++it_gpu_id)
93  if (!is_positive_integer(*it_gpu_id)) {
94  *error = 1;
95  break;
96  }
97  }
98  return false;
99  }
100 
101  } else {
102  *error = 1;
103  // when error is set, it doesn't really matter whether the method
104  // returns true or false
105  return false;
106  }
107 }

394 int pbqt_action::create_threads() {
395  std::string msg;
396 
397  std::vector<uint16_t> gpu_id;
398  std::vector<uint16_t> gpu_device_id;
399  uint16_t transfer_ix = 0;
400  bool bmatch_found = false;
401 
402  gpu_get_all_gpu_id(&gpu_id);
403  gpu_get_all_device_id(&gpu_device_id);
404 
405  for (size_t i = 0; i < gpu_id.size()-1; i++) { // all possible sources
406  // filter out by source device id
407  if (property_device_id > 0) {
408  if (property_device_id != gpu_device_id[i]) {
409  continue;
410  }
411  }
412 
413  // filter out by listed sources
414  if (!property_device_all) {
415  const auto it = std::find(property_device.cbegin(),
416  property_device.cend(),
417  gpu_id[i]);
418  if (it == property_device.cend()) {
419  continue;
420  }
421  }
422 
423  for (size_t j = i+1; j < gpu_id.size(); j++) { // all possible peers
424  RVSTRACE_
425  // filter out by peer id
426  if (prop_peer_deviceid > 0) {
427  RVSTRACE_
428  if (prop_peer_deviceid != gpu_device_id[j]) {
429  RVSTRACE_
430  continue;
431  }
432  }
433 
434  RVSTRACE_
435  // filter out by listed peers
436  if (!prop_peer_device_all_selected) {
437  RVSTRACE_
438  const auto it = std::find(prop_peers.cbegin(),
439  prop_peers.cend(),
440  std::to_string(gpu_id[j]));
441  if (it == prop_peers.cend()) {
442  RVSTRACE_
443  continue;
444  }
445  }
446 
447  RVSTRACE_
448  // signal that at lease one matching src-dst combination
449  // has been found:
450  bmatch_found = true;
451 
452  // get NUMA nodes
453  uint16_t srcnode;
454  if (rvs::gpulist::gpu2node(gpu_id[i], &srcnode)) {
455  msg + "no node found for GPU ID " + std::to_string(gpu_id[i]);
456  rvs::lp::Err(msg, MODULE_NAME_CAPS, action_name);
457  return -1;
458  }
459 
460  uint16_t dstnode;
461  if (rvs::gpulist::gpu2node(gpu_id[j], &dstnode)) {
462  RVSTRACE_
463  msg = "no node found for GPU ID " + std::to_string(gpu_id[j]);
464  rvs::lp::Err(msg, MODULE_NAME_CAPS, action_name);
465  return -1;
466  }
467 
468  RVSTRACE_
469  uint32_t distance = 0;
470  std::vector<rvs::linkinfo_t> arr_linkinfo;
471  rvs::hsa::Get()->GetLinkInfo(srcnode, dstnode,
472  &distance, &arr_linkinfo);
473 
474  // perform peer check
475  if (is_peer(gpu_id[i], gpu_id[j])) {
476  RVSTRACE_
477  msg = "[" + action_name + "] p2p "
478  + std::to_string(gpu_id[i]) + " "
479  + std::to_string(gpu_id[j]) + " peers:true ";
480 
481  if (distance == rvs::hsa::NO_CONN) {
482  msg += "distance:-1";
483  } else {
484  msg += "distance:" + std::to_string(distance);
485  }
486  // iterate through individual hops
487  for (auto it = arr_linkinfo.begin(); it != arr_linkinfo.end(); it++) {
488  msg += " " + it->strtype + ":";
489  if (it->distance == rvs::hsa::NO_CONN) {
490  msg += "-1";
491  } else {
492  msg +=std::to_string(it->distance);
493  }
494  }
495  rvs::lp::Log(msg, rvs::logresults);
496  if(distance == rvs::hsa::NO_CONN) {
497  continue; // no point if no connection
498  }
499  if (0 != arr_linkinfo.size()) {
500  /* Log link type */
501  log_json_data(std::to_string(srcnode), std::to_string(gpu_id[j]), rvs::logresults,
502  pbqt_json_data_t::PBQT_LINK_TYPE, arr_linkinfo[0].strtype);
503  /* Note: Assuming link type for all hops between GPUs are the same */
504  }
505 
506  RVSTRACE_
507  // GPUs are peers, create transaction for them
508  if (prop_test_bandwidth) {
509  RVSTRACE_
510  pbqtworker* p = nullptr;
511 
512  transfer_ix += 1;
513 
514  p = new pbqtworker;
515  if (p == nullptr) {
516  RVSTRACE_
517  msg = "internal error";
518  rvs::lp::Err(msg, MODULE_NAME_CAPS, action_name);
519  return -1;
520  }
521  p->initialize(srcnode, dstnode, prop_bidirectional);
522  RVSTRACE_
523  p->set_name(action_name);
524  p->set_stop_name(action_name);
525  p->set_transfer_ix(transfer_ix);
526  p->set_block_sizes(block_size);
527  test_array.push_back(p);
528  }
529  }
530  else {
531  RVSTRACE_
532  msg = "[" + action_name + "] p2p "
533  + std::to_string(gpu_id[i]) + " "
534  + std::to_string(gpu_id[j]) + " peers:false ";
535 
536  if (distance == rvs::hsa::NO_CONN) {
537  msg += "distance:-1";
538  } else {
539  msg += "distance:" + std::to_string(distance);
540  }
541  // iterate through individual hops
542  for (auto it = arr_linkinfo.begin(); it != arr_linkinfo.end(); it++) {
543  msg += " " + it->strtype + ":";
544  if (it->distance == rvs::hsa::NO_CONN) {
545  msg += "-1";
546  } else {
547  msg +=std::to_string(it->distance);
548  }
549  }
550 
551  rvs::lp::Log(msg, rvs::logresults);
552  }
553  }
554  }
555 
556  RVSTRACE_
557  if (prop_test_bandwidth && test_array.size() < 1) {
558  RVSTRACE_
559  std::string diag;
560  if (bmatch_found) {
561  RVSTRACE_
562  diag = "No peers found";
563  } else {
564  RVSTRACE_
565  diag = "No devices match criteria from the test configuration";
566  }
567  RVSTRACE_
568  msg = "[" + action_name + "] p2p-bandwidth " + diag;
569  rvs::lp::Log(msg, rvs::logerror);
570  if (bjson) {
571  RVSTRACE_
572  unsigned int sec;
573  unsigned int usec;
574  rvs::lp::get_ticks(&sec, &usec);
575  void* pjson = rvs::lp::LogRecordCreate("p2p-bandwidth",
576  action_name.c_str(), rvs::logerror, sec, usec);
577  if (pjson != NULL) {
578  RVSTRACE_
579  rvs::lp::AddString(pjson,
580  "message",
581  diag);
582  rvs::lp::LogRecordFlush(pjson);
583  }
584  }
585  RVSTRACE_
586  return 0;
587  }
588 
589  RVSTRACE_
590  for (auto it = test_array.begin(); it != test_array.end(); ++it) {
591  RVSTRACE_
592  (*it)->set_transfer_num(test_array.size());
593  }
594 
595  RVSTRACE_
596  return 0;
597 }

693

695  bandwidth = current_size/duration/1000 / 1000 / 1000;
696  if (bidir) {
697  bandwidth *=2;
698  }
699  snprintf( buff, sizeof(buff), "%.3f GBps", bandwidth);

702  // (do not reset final totals as this is just intermediate query)

707  if (bidir) {
708  bandwidth *=2;
709  }

720  RVSTRACE_

739

741  + std::to_string(transfer_ix) + "/" + std::to_string(transfer_num)
742  + "] " + std::to_string(src_id) + " " + std::to_string(dst_id)
743  + " bidirectional: " + std::string(bidir ? "true" : "false")

746 
747 #if 0
