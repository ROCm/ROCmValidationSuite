151 int gpup_action::property_get_value(uint16_t gpu_id) {
152  uint16_t node_id;
153  char path[CHAR_MAX_BUFF_SIZE];
154  void *json_gpuprop_node = NULL;
155  string prop_name, prop_val, msg;
156  std::ifstream f_prop;
157 
158  RVSTRACE_
159  if (rvs::gpulist::gpu2node(gpu_id, &node_id)) {
160  RVSTRACE_
161  return -1;
162  }
163 
164  // cache property names to validate for existance
165  property_name_validate = property_name;
166 
167  snprintf(path, CHAR_MAX_BUFF_SIZE, "%s/%d/properties",
168  KFD_SYS_PATH_NODES, node_id);
169 
170  if (bjson) {
171  RVSTRACE_
172  if (json_root_node == NULL) {
173  RVSTRACE_
174  return -1;
175  }
176  json_gpuprop_node = rvs::lp::CreateNode(json_root_node,
177  JSON_PROP_NODE_NAME);
178  if (json_gpuprop_node == NULL) {
179  RVSTRACE_
180  // log the error
181  msg = std::string(JSON_CREATE_NODE_ERROR);
182  rvs::lp::Err(msg, MODULE_NAME_CAPS, action_name);
183  return -1;
184  }
185  rvs::lp::AddNode(json_root_node, json_gpuprop_node);
186  }
187 
188  RVSTRACE_
189  f_prop.open(path);
190  while (f_prop >> prop_name) {
191  RVSTRACE_
192  f_prop >> prop_val;
193 
194  validate_property_name(prop_name);
195  // check if filtering by property is needed
196  if (io_link_property_name.size() > 0) {
197  auto it = std::find(property_name.begin(),
198  property_name.end(),
199  prop_name);
200  // not found - skip to next property
201  if (it == property_name.end()) {
202  continue;
203  }
204  }
205  msg = "["+action_name + "] " + MODULE_NAME +
206  " " + std::to_string(gpu_id) +
207  " " + prop_name + " " + prop_val;
208  rvs::lp::Log(msg, rvs::logresults);
209  if (bjson && json_gpuprop_node != NULL) {
210  rvs::lp::AddString(json_gpuprop_node, prop_name, prop_val);
211  }
212  }
213  RVSTRACE_
214  f_prop.close();
215 
216  if (property_name_validate.size() > 0) {
217  RVSTRACE_
218  msg = "Properties not found for GPU " + std::to_string(gpu_id) + ":";
219  for (auto it = property_name_validate.begin();
220  it != property_name_validate.end(); it++) {
221  msg += " " + *it;
222  }
223  rvs::lp::Err(msg, MODULE_NAME_CAPS, action_name);
224  return -1;
225  }
226 
227  RVSTRACE_
228  return 0;
229 }

324 int gpup_action::run(void) {
325  std::string msg;
326  int sts = 0;
327 
328  // get the action name
329  if (property_get(RVS_CONF_NAME_KEY, &action_name)) {
330  rvs::lp::Err("Action name missing", MODULE_NAME_CAPS);
331  return -1;
332  }
333 
334  // get <device> property value (a list of gpu id)
335  if (int sts = property_get_device()) {
336  switch (sts) {
337  case 1:
338  msg = "Invalid 'device' key value.";
339  break;
340  case 2:
341  msg = "Missing 'device' key.";
342  break;
343  }
344  rvs::lp::Err(msg, MODULE_NAME_CAPS, action_name);
345  return -1;
346  }
347 
348  // get the <deviceid> property value if provided
349  if (property_get_int<uint16_t>(RVS_CONF_DEVICEID_KEY,
350  &property_device_id, 0u)) {
351  msg = "Invalid 'deviceid' key value.";
352  rvs::lp::Err(msg, MODULE_NAME_CAPS, action_name);
353  return -1;
354  }
355 
356  // extract properties and io_links properties names
357  property_split(JSON_PROP_NODE_NAME);
358  property_split(JSON_IO_LINK_PROP_NODE_NAME);
359 
360  bjson = false; // already initialized in the default constructor
361 
362  // check for -j flag (json logging)
363  if (has_property("cli.-j")) {
364  bjson = true;
365  }
366 
367  // get all AMD GPUs
368  vector<uint16_t> gpu;
369  gpu_get_all_gpu_id(&gpu);
370  bool b_gpu_found = false;
371 
372  // iterate over AMD GPUs
373  for (auto it = gpu.begin(); it !=gpu.end(); ++it) {
374  // filter by gpu_id if needed
375  if (property_device_id > 0) {
376  uint16_t dev_id;
377  if (!rvs::gpulist::gpu2device(*it, &dev_id)) {
378  if (dev_id != property_device_id) {
379  continue;
380  }
381  } else {
382  msg = "Device ID not found for GPU " + std::to_string(*it);
383  rvs::lp::Err(msg, MODULE_NAME, action_name);
384  return -1;
385  }
386  }
387 
388  // filter by device if needed
389  if (!property_device_all) {
390  if (std::find(property_device.begin(), property_device.end(), *it) ==
391  property_device.end()) {
392  continue;
393  }
394  }
395 
396  b_gpu_found = true;
397 
398  // if JSON required
399  if (bjson) {
400  unsigned int sec;
401  unsigned int usec;
402  rvs::lp::get_ticks(&sec, &usec);
403 
404  json_root_node = rvs::lp::LogRecordCreate(MODULE_NAME,
405  action_name.c_str(), rvs::logresults, sec, usec);
406  if (json_root_node == nullptr) {
407  // log the error
408  msg = JSON_CREATE_NODE_ERROR;
409  rvs::lp::Err(msg, MODULE_NAME, action_name);
410  return -1;
411  }
412 
413  // Add GPU ID
414  rvs::lp::AddInt(json_root_node, RVS_JSON_LOG_GPU_ID_KEY, *it);
415  }
416 
417  // properties values
418  sts = property_get_value(*it);
419 
420  // so far so good?
421  if (sts == 0) {
422  RVSTRACE_
423  // do io_links properties
424  sts = property_io_links_get_value(*it);
425  }
426 
427  if (bjson) { // json logging stuff
428  RVSTRACE_
429  rvs::lp::LogRecordFlush(json_root_node);
430  json_root_node = nullptr;
431  }
432 
433  if (sts) {
434  RVSTRACE_
435  break;
436  }
437  } // for all gpu_id
438 
439  if (!b_gpu_found) {
440  msg = "No device matches criteria from configuration. ";
441  rvs::lp::Err(msg, MODULE_NAME, action_name);
442  return -1;
443  }
444  return sts;
445 }
