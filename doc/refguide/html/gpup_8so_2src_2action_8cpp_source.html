98 bool action::property_get_device(int *error, int num_nodes) {
99  map<string, string>::iterator it;
100  std::ifstream f_id;
101  char path[CHAR_MAX_BUFF_SIZE];
102  string gpu_id;
103  *error = 0; // init with 'no error'
104  it = property.find(RVS_CONF_DEVICE_KEY);
105  if (it != property.end()) {
106  if (it->second == "all") {
107  for (int node_id = 0; node_id < num_nodes; node_id++) {
108  snprintf(path, CHAR_MAX_BUFF_SIZE, "%s/%d/gpu_id",
109  KFD_SYS_PATH_NODES, node_id);
110  f_id.open(path);
111  f_id >> gpu_id;
112  if (gpu_id != "0") {
113  gpus_id.push_back(gpu_id);
114  }
115  f_id.close();
116  }
117  property.erase(it);
118  return true;
119  } else {
120  // split the list of gpu_id
121  gpus_id = str_split(it->second, YAML_DEVICE_PROP_DELIMITER);
122  property.erase(it);
123  if (gpus_id.empty()) {
124  *error = 1; // list of gpu_id cannot be empty
125  } else {
126  for (vector<string>::iterator it_gpu_id =
127  gpus_id.begin();
128  it_gpu_id != gpus_id.end(); ++it_gpu_id) {
129  if (!is_positive_integer(*it_gpu_id)) {
130  *error = 1;
131  break;
132  }}
133  }
134  return false;
135  }
136  } else {
137  *error = 1;
138  return false;
139  }
140 }

195 bool action::property_split(string props) {
196  map<string, string>::iterator it;
197  string s;
198  auto prop_length = std::end(gpu_prop_names) - std::begin(gpu_prop_names);
199  auto io_prop_length = std::end(gpu_io_link_prop_names) -
200  std::begin(gpu_io_link_prop_names);
201 
202  for (it = property.begin(); it != property.end(); ++it) {
203  s = it->first;
204  if (s.find(".") != std::string::npos && s.substr(0, s.find(".")) ==
205  props) {
206  if (!(s.substr(s.find(".")+1) == "all")) {
207  if (props == JSON_PROP_NODE_NAME)
208  property_name.push_back(s.substr(s.find(".")+1));
209  else
210  io_link_property_name.push_back(s.substr(s.find(".")+1));
211  } else {
212  if (props == JSON_PROP_NODE_NAME) {
213  for (int i = 0; i < prop_length; i++)
214  property_name.push_back(gpu_prop_names[i]);
215  } else {
216  for (int i = 0; i < io_prop_length; i++)
217  io_link_property_name.push_back(gpu_io_link_prop_names[i]);
218  }
219  return 1;
220  }
221  }
222  }
223  return 0;
224 }

231 void action::property_get_value(string gpu_id, int node_id) {
232  char path[CHAR_MAX_BUFF_SIZE];
233  void *json_gpuprop_node = NULL;
234  string prop_name, prop_val, msg;
235  std::ifstream f_prop;
236 
237  snprintf(path, CHAR_MAX_BUFF_SIZE, "%s/%d/properties",
238  KFD_SYS_PATH_NODES, node_id);
239  f_prop.open(path);
240 
241  if (bjson) {
242  if (json_root_node != NULL) {
243  json_gpuprop_node = rvs::lp::CreateNode(json_root_node,
244  JSON_PROP_NODE_NAME);
245  if (json_gpuprop_node == NULL) {
246  // log the error
247  msg = action_name + " " + MODULE_NAME + " " +
248  JSON_CREATE_NODE_ERROR;
249  cerr << "RVS-GPUP: " << msg;
250  }
251  }
252  }
253 
254  if (bjson && json_gpuprop_node != NULL) { // json logging stuff
255  rvs::lp::AddString(json_gpuprop_node, RVS_JSON_LOG_GPU_ID_KEY,
256  gpu_id);
257  }
258 
259  // properties
260  for (vector<string>::iterator it_prop_name=property_name.begin();
261  it_prop_name != property_name.end(); ++it_prop_name) {
262  while (f_prop >> prop_name) {
263  if (prop_name == *it_prop_name) {
264  f_prop >> prop_val;
265  msg = "["+action_name + "] " + MODULE_NAME +
266  " " + gpu_id +
267  " " + prop_name + " " + prop_val;
268  log(msg.c_str(), rvs::logresults);
269  if (bjson && json_gpuprop_node != NULL) {
270  rvs::lp::AddString(json_gpuprop_node, prop_name,
271  prop_val);
272  }
273  break;
274  }
275  f_prop >> prop_val;
276  }
277  f_prop.clear();
278  f_prop.seekg(0, std::ios::beg);
279  }
280  f_prop.close();
281 
282  if (bjson && json_gpuprop_node != NULL) // json logging stuff
283  rvs::lp::AddNode(json_root_node, json_gpuprop_node);
284 }

371 int action::run(void) {
372  string gpu_id, msg;
373  int num_nodes;
374  bool dev_id_corr;
375  int error = 0;
376 
377  // discover the number of nodes: Inside nodes folder there are only folders
378  // that represent the node number
379  num_nodes = gpu_num_subdirs(const_cast<char*>(KFD_SYS_PATH_NODES),
380  const_cast<char*>(""));
381 
382  // get the action name
383  rvs::actionbase::property_get_action_name(&error);
384  if (error == 2) {
385  msg = "action field is missing in gst module";
386  cerr << "RVS-GPUP: " << msg;
387  return -1;
388  }
389 
390  // get <device> property value (a list of gpu id)
391  property_get_device(&error, num_nodes);
392 
393  // get the <deviceid> property value if provided
394  int dev_id = property_get_deviceid(&error);
395 
396  // extract properties and io_links properties names
397  property_split(JSON_PROP_NODE_NAME);
398  property_split(JSON_IO_LINK_PROP_NODE_NAME);
399 
400  bjson = false; // already initialized in the default constructor
401 
402  // check for -j flag (json logging)
403  if (property.find("cli.-j") != property.end()) {
404  unsigned int sec;
405  unsigned int usec;
406  rvs::lp::get_ticks(&sec, &usec);
407 
408  bjson = true;
409 
410  json_root_node = rvs::lp::LogRecordCreate(MODULE_NAME,
411  action_name.c_str(), rvs::loginfo, sec, usec);
412  if (json_root_node == NULL) {
413  // log the error
414  msg = action_name + " " + MODULE_NAME + " " +
415  JSON_CREATE_NODE_ERROR;
416  }
417  }
418 
419  for (vector<string>::iterator it_gpu_id = gpus_id.begin(); it_gpu_id !=
420  gpus_id.end(); ++it_gpu_id) {
421  for (int node_id = 0; node_id < num_nodes; node_id++) {
422  dev_id_corr = device_id_correct(node_id, dev_id);
423 
424  gpu_id = property_get_gpuid(node_id);
425 
426  if (gpu_id == *it_gpu_id) {
427  if (dev_id_corr) {
428  // properties values
429  property_get_value(gpu_id, node_id);
430  // io_links properties
431  property_io_links_get_value(gpu_id, node_id);
432  } else {
433  cerr << "RVS-GPUP: action: " << property["name"] <<
434  " invalid 'deviceid' key value: " <<
435  dev_id << std::endl;
436  return -1;
437  }
438  }
439  }
440  }
441 
442  if (bjson && json_root_node != NULL) { // json logging stuff
443  rvs::lp::LogRecordFlush(json_root_node);
444  }
445 
446  return 0;
447 }
