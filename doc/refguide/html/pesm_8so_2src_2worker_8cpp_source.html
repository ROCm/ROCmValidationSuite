75 void Worker::run() {
76  brun = true;
77  char buff[1024];
78 
79  map<string, string>::iterator it;
80  vector<uint16_t> gpus_location_id;
81  map<uint16_t, string> old_val;
82  map<uint16_t, string> old_pwr_val;
83 
84  struct pci_access *pacc;
85  struct pci_dev *dev;
86 
87  unsigned int sec;
88  unsigned int usec;
89  void* r;
90 
91  // get timestamp
92  rvs::lp::get_ticks(&sec, &usec);
93 
94  // add string output
95  string msg("[" + action_name + "] pesm " + strgpuids + " started");
96  rvs::lp::Log(msg, rvs::logresults, sec, usec);
97 
98  // add JSON output
99  r = rvs::lp::LogRecordCreate("pesm", action_name.c_str(), rvs::logresults,
100  sec, usec);
101  rvs::lp::AddString(r, "msg", "started");
102  rvs::lp::AddString(r, "device", strgpuids);
103  rvs::lp::LogRecordFlush(r);
104 
105  // worker thread has started
106  while (brun) {
107  rvs::lp::Log("[" + action_name + "] pesm worker thread is running...",
108  rvs::logtrace);
109 
110  // get the pci_access structure
111  pacc = pci_alloc();
112  // initialize the PCI library
113  pci_init(pacc);
114  // get the list of devices
115  pci_scan_bus(pacc);
116 
117  // iterate over devices
118  for (dev = pacc->devices; dev; dev = dev->next) {
119  pci_fill_info(dev, PCI_FILL_IDENT | PCI_FILL_BASES | PCI_FILL_CLASS
120  | PCI_FILL_EXT_CAPS | PCI_FILL_CAPS
121  | PCI_FILL_PHYS_SLOT); // fil in the info
122 
123  // computes the actual dev's location_id (sysfs entry)
124  uint16_t dev_location_id =
125  ((((uint16_t)(dev->bus)) << 8) | ((uint16_t)(dev->dev)) << 3);
126 
127  uint16_t gpu_id;
128  // if not and AMD GPU just continue
129  if (rvs::gpulist::location2gpu(dev_location_id, &gpu_id))
130  continue;
131 
132  // device_id filtering
133  if ( device_id != 0 && dev->device_id != device_id)
134  continue;
135 
136  // gpu id filtering
137  if (bfiltergpu) {
138  auto itgpuid = find(gpuids.begin(), gpuids.end(), gpu_id);
139  if (itgpuid == gpuids.end())
140  continue;
141  }
142 
143  rvs::lp::get_ticks(&sec, &usec);
144 
145  // get current speed for the link
146  get_link_stat_cur_speed(dev, buff);
147  string new_val(buff);
148 
149  // get current power state for GPU
150  get_pwr_curr_state(dev, buff);
151  string new_pwr_val(buff);
152 
153  // link speed changed?
154  if (old_val[gpu_id] != new_val) {
155  // new value is different, so store it;
156  old_val[gpu_id] = new_val;
157 
158  string msg("[" + action_name + "] " + "pesm "
159  + std::to_string(gpu_id) + " link speed change " + new_val);
160  rvs::lp::Log(msg, rvs::loginfo, sec, usec);
161 
162  r = rvs::lp::LogRecordCreate("pesm ", action_name.c_str(), rvs::loginfo,
163  sec, usec);
164  rvs::lp::AddString(r, "msg", "link speed change");
165  rvs::lp::AddString(r, "val", new_val);
166  rvs::lp::LogRecordFlush(r);
167  }
168 
169  // power state changed
170  if (old_pwr_val[gpu_id] != new_pwr_val) {
171  // new value is different, so store it;
172  old_pwr_val[gpu_id] = new_pwr_val;
173 
174  string msg("[" + action_name + "] " + "pesm "
175  + std::to_string(gpu_id) +
176  " power state change " + new_pwr_val);
177  rvs::lp::Log(msg, rvs::loginfo, sec, usec);
178 
179  r = rvs::lp::LogRecordCreate("pesm", action_name.c_str(), rvs::loginfo,
180  sec, usec);
181  rvs::lp::AddString(r, "msg", "power state change");
182  rvs::lp::AddString(r, "val", new_pwr_val);
183  rvs::lp::LogRecordFlush(r);
184  }
185  }
186 
187  pci_cleanup(pacc);
188 
189  sleep(1);
190  }
191 
192  // get timestamp
193  rvs::lp::get_ticks(&sec, &usec);
194 
195  // add string output
196  msg = "[" + stop_action_name + "] pesm all stopped";
197  rvs::lp::Log(msg, rvs::logresults, sec, usec);
198 
199  // add JSON output
200  r = rvs::lp::LogRecordCreate("PESM",
201  stop_action_name.c_str(), rvs::logresults,
202  sec, usec);
203  rvs::lp::AddString(r, "msg", "stopped");
204  rvs::lp::LogRecordFlush(r);
205 
206  rvs::lp::Log("[" + stop_action_name + "] pesm worker thread has finished",
207  rvs::logdebug);
208 }
