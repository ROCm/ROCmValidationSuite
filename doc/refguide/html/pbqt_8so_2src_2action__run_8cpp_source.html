92 int pbqt_action::run() {
93  int sts;
94  string msg;
95  std::chrono::time_point<std::chrono::system_clock> pbqt_start_time;
96  std::chrono::time_point<std::chrono::system_clock> pbqt_end_time;
97 
98  rvs::lp::Log("int pbqt_action::run()", rvs::logtrace);
99 
100  if (property.find("cli.-j") != property.end()) {
101  bjson = true;
102  }
103 
104  if (!get_all_common_config_keys()) {
105  msg = "Error in get_all_common_config_keys()";
106  rvs::lp::Err(msg, MODULE_NAME_CAPS, action_name);
107  return -1;
108  }
109  if (!get_all_pbqt_config_keys()) {
110  msg = "Error in get_all_pbqt_config_keys()";
111  rvs::lp::Err(msg, MODULE_NAME_CAPS, action_name);
112  return -1;
113  }
114 
115  // log_interval must be less than duration
116  if (property_log_interval > 0 && property_duration > 0) {
117  if (static_cast<uint64_t>(property_log_interval) > property_duration) {
118  msg = "log_interval must be less than duration";
119  rvs::lp::Err(msg, MODULE_NAME_CAPS, action_name);
120  return -1;
121  }
122  }
123 
124  test_duration = property_duration;
125 
126  if(bjson){
127  json_add_primary_fields();
128  }
129 
130  sts = create_threads();
131  if (sts) {
132  RVSTRACE_
133  return sts;
134  }
135 
136  if (!prop_test_bandwidth || test_array.size() < 1) {
137  RVSTRACE_
138  // do cleanup
139  destroy_threads();
140  return 0;
141  }
142 
143  RVSTRACE_
144  // define timers
145  rvs::timer<pbqt_action> timer_running(&pbqt_action::do_running_average, this);
146  rvs::timer<pbqt_action> timer_final(&pbqt_action::do_final_average, this);
147 
148  unsigned int iter = property_count > 0 ? property_count : 1;
149  unsigned int step = 1;
150 
151  do {
152  RVSTRACE_
153  // let the test run in this iteration
154  brun = true;
155 
156  // start timers
157  if (property_duration) {
158  RVSTRACE_
159  timer_final.start(property_duration, true); // ticks only once
160  }
161 
162  if (property_log_interval) {
163  RVSTRACE_
164  timer_running.start(property_log_interval); // ticks continuously
165  }
166 
167  pbqt_start_time = std::chrono::system_clock::now();
168 
169  RVSTRACE_
170  do {
171  if (property_parallel) {
172  sts = run_parallel();
173  } else {
174  sts = run_single();
175  }
176  pbqt_end_time = std::chrono::system_clock::now();
177  uint64_t test_time = time_diff(pbqt_end_time, pbqt_start_time) ;
178  if(test_time >= property_duration) {
179  pbqt_action::do_final_average();
180  break;
181  }
182  } while (brun);
183 
184  RVSTRACE_
185  timer_running.stop();
186  timer_final.stop();
187 
188  iter -= step;
189 
190  // insert wait between runs if needed
191  if (iter > 0 && property_wait > 0) {
192  RVSTRACE_
193  sleep(property_wait);
194  }
195  } while (iter && !rvs::lp::Stopping());
196 
197  RVSTRACE_
198  sts = rvs::lp::Stopping() ? -1 : 0;
199 
200  print_final_average();
201 
202  // do cleanup
203  destroy_threads();
204 
205  if(bjson){
206  rvs::lp::JsonActionEndNodeCreate();
207  }
208 
209  return sts;
210 }

220 int pbqt_action::run_single() {
221  RVSTRACE_
222  int sts = 0;
223 
224  // iterate through test array and invoke tests one by one
225  for (auto it = test_array.begin(); brun && it != test_array.end(); ++it) {
226  RVSTRACE_
227  (*it)->do_transfer();
228 
229  // if log interval is zero, print current results immediately
230  if (property_log_interval == 0) {
231  print_running_average(*it);
232  }
233 
234  if (rvs::lp::Stopping()) {
235  RVSTRACE_
236  brun = false;
237  sts = -1;
238  break;
239  }
240  }
241 
242  return sts;
243 }

252 int pbqt_action::run_parallel() {
253  RVSTRACE_
254 
255  // start all worker threads
256  for (auto it = test_array.begin(); it != test_array.end(); ++it) {
257  (*it)->start();
258  }
259 
260  // join all worker threads
261  for (auto it = test_array.begin(); it != test_array.end(); ++it) {

264 
265  return rvs::lp::Stopping() ? -1 : 0;
