116 void rvs::actionbase::property_get_uint_list(const std::string& key,
117  const std::string& delimiter,
118  std::vector<uint32_t>* pval,
119  bool* pball,
120  int *error) {
121  bool bfound = false;
122  std::string strval;
123 
124  // init with 'no error'
125  *error = 0;
126 
127  // fetch key value if any
128  bfound = has_property(key, &strval);
129 
130  // key not found - return
131  if (!bfound) {
132  *error = 2;
133  return;
134  }
135 
136  // found and is "all" - set flag and return
137  if (strval == "all") {
138  *pball = true;
139  pval->clear();
140  return;
141  } else {
142  *pball = false;
143  }
144 
145  // parse key value into std::vector<std::string>
146  auto strarray = str_split(strval, delimiter);
147 
148  // convert str arary into uint16_t array
149  int sts = rvs_util_strarr_to_uintarr(strarray, pval);
150 
151  if (sts < 0) {
152  *error = 1;
153  pval->clear();
154  }
155 }

180 bool rvs::actionbase::property_get_device(int *error) {
181  *error = 0; // init with 'no error'
182  auto it = property.find(RVS_CONF_DEVICE_KEY);
183  if (it != property.end()) {
184  if (it->second == "all") {
185  return true;
186  } else {
187  // split the list of gpu_id
188  device_prop_gpu_id_list = str_split(it->second,
189  YAML_DEVICE_PROP_DELIMITER);
190 
191  if (device_prop_gpu_id_list.empty()) {
192  *error = 1; // list of gpu_id cannot be empty
193  } else {
194  for (vector<string>::iterator it_gpu_id =
195  device_prop_gpu_id_list.begin();
196  it_gpu_id != device_prop_gpu_id_list.end(); ++it_gpu_id)
197  if (!is_positive_integer(*it_gpu_id)) {
198  *error = 1;
199  break;
200  }
201  }
202  return false;
203  }
204 
205  } else {
206  *error = 1;
207  // when error is set, it doesn't really matter whether the method
208  // returns true or false
209  return false;
210  }
211 }
