127 bool action::get_gpu_all_pcie_capabilities(struct pci_dev *dev,
128  uint16_t gpu_id) {
129  char buff[CHAR_BUFF_MAX_SIZE];
130  string prop_name, msg;
131  bool pci_infra_qual_result = true;
132  map<string, string>::iterator it; // module's properties map iterator
133  void *json_pcaps_node = NULL;
134  uint8_t i;
135 
136  if (bjson) {
137  if (json_root_node != NULL) {
138  json_pcaps_node = rvs::lp::CreateNode(json_root_node,
139  JSON_CAPS_NODE_NAME);
140  if (json_pcaps_node == NULL) {
141  // log the error
142  msg = action_name + " " + MODULE_NAME + " "
143  + JSON_CREATE_NODE_ERROR;
144  cerr << "RVS-PEQT: " << msg;
145  }
146  }
147  }
148 
149  if (bjson && json_pcaps_node != NULL) {
150  rvs::lp::AddString(json_pcaps_node, RVS_JSON_LOG_GPU_ID_KEY,
151  std::to_string(gpu_id));
152  }
153 
154  for (it = property.begin(); it != property.end(); ++it) {
155  // skip the "capability."
156  string prop_name = it->first.substr(it->first.find_last_of(".") + 1);
157  bool prop_found = false;
158  for (i = 0; i < PCI_DEV_NUM_CAPABILITIES; i++) {
159  if (prop_name == pcie_cap_names[i]) {
160  prop_found = true;
161  // call the capability's corresponding function
162  (*arr_prop_pfunc_names[i])(dev, buff);
163 
164  // log the capability's value
165  msg = "[" + action_name + "] " + MODULE_NAME + " " +
166  pcie_cap_names[i] + " " + buff;
167  log(msg.c_str(), rvs::loginfo);
168 
169  if (bjson && json_pcaps_node != NULL) {
170  rvs::lp::AddString(json_pcaps_node, pcie_cap_names[i],
171  buff);
172  }
173 
174  // check for regex match
175  if (it->second != "") {
176  try {
177  regex prop_regex(it->second);
178  if (!regex_match(buff, prop_regex)) {
179  pci_infra_qual_result = false;
180  }
181  } catch (const std::regex_error& e) {
182  // log the regex error
183  msg = action_name + " " + MODULE_NAME + " "
184  + YAML_REGULAR_EXPRESSION_ERROR + " at '"
185  + it->second + "'";
186  cerr << "RVS-PEQT: " << msg;
187  }
188  }
189  break;
190  }
191  }
192 
193  if (!prop_found &&
194  it->first.find(YAML_CAPABILITY_TAG) != string::npos) {
195  // the property was not found among those that
196  // have fixed/constant name => check whether it's
197  // a dynamic Power Budgeting capability
198  if (regex_match(prop_name, pb_dynamic_regex)) {
199  // no additional checks are needed (itetator != .end() && npos)
200  // because the prop_name already matched the regular expression
201 
202  std::size_t pos_pb_pm_state =
203  prop_name.find_first_of(PB_OP_COND_DYN_DELIMITER);
204  map<string, uint8_t>::iterator it_pb_pm_state =
205  pb_op_pm_states_encodings_map.find
206  (prop_name.substr(0, pos_pb_pm_state));
207  uint8_t pb_op_pm_state = it_pb_pm_state->second;
208 
209  std::size_t pos_pb_type =
210  prop_name.find(PB_OP_COND_DYN_DELIMITER,
211  pos_pb_pm_state + 1);
212  map<string, uint8_t>::iterator it_pb_type =
213  pb_op_pm_types_encodings_map.find
214  (prop_name.substr(pos_pb_pm_state + 1,
215  pos_pb_type - pos_pb_pm_state - 1));
216  uint8_t pb_op_pm_type = it_pb_type->second;
217 
218  map<string, uint8_t>::iterator it_pb_power_rail =
219  pb_op_pm_power_rails_encodings_map.find
220  (prop_name.substr(pos_pb_type + 1));
221  uint8_t pb_op_power_rail = it_pb_power_rail->second;
222  // query for power budgeting capabilities
223  get_pwr_budgeting(dev, pb_op_pm_state, pb_op_pm_type,
224  pb_op_power_rail, buff);
225 
226  // log the capability's value
227  msg = "[" + action_name + "] " + MODULE_NAME + " " + prop_name
228  + " " + buff;
229  log(msg.c_str(), rvs::loginfo);
230 
231  if (bjson && json_pcaps_node != NULL) {
232  rvs::lp::AddString(json_pcaps_node, prop_name,
233  buff);
234  }
235 
236  // check for regex match
237  if (it->second != "") {
238  try {
239  regex prop_regex(it->second);
240  if (!regex_match(buff, prop_regex)) {
241  pci_infra_qual_result = false;
242  }
243  } catch (const std::regex_error& e) {
244  // log the regex error
245  msg = action_name + " " + MODULE_NAME + " "
246  + YAML_REGULAR_EXPRESSION_ERROR + " at '"
247  + it->second + "'";
248  cerr << "RVS-PEQT: " << msg;
249  }
250  }
251  }
252  }
253  }
254 
255  if (bjson && json_pcaps_node != NULL)
256  rvs::lp::AddNode(json_root_node, json_pcaps_node);
257 
258  return pci_infra_qual_result;
259 }

266 int action::run(void) {
267  string msg;
268  map<string, string>::iterator it; // module's properties map iterator
269  bool pci_infra_qual_result = true; // PCI qualification result
270  bool amd_gpus_found = false;
271  bool device_all_selected = false;
272  bool device_id_filtering = false;
273  int error = 0;
274  uint16_t deviceid;
275  vector<uint16_t> gpus_location_id;
276  vector<uint16_t> gpus_id;
277  uint8_t i;
278 
279  struct pci_access *pacc;
280  struct pci_dev *dev;
281 
282  // get the action name
283  rvs::actionbase::property_get_action_name(&error);
284  if (error == 2) {
285  msg = "action field is missing in gst module";
286  cerr << "RVS-PEQT: " << msg;
287  return -1;
288  }
289 
290  bjson = false; // already initialized in the default constructor
291 
292  // check for -j flag (json logging)
293  if (property.find("cli.-j") != property.end()) {
294  unsigned int sec;
295  unsigned int usec;
296  rvs::lp::get_ticks(&sec, &usec);
297 
298  bjson = true;
299 
300  json_root_node = rvs::lp::LogRecordCreate(MODULE_NAME,
301  action_name.c_str(), rvs::loginfo, sec, usec);
302  if (json_root_node == NULL) {
303  // log the error
304  msg = action_name + " " + MODULE_NAME + " " +
305  JSON_CREATE_NODE_ERROR;
306  cerr << "RVS-PEQT: " << msg;
307  }
308  }
309 
310  // get <device> property value (a list of gpu id)
311  device_all_selected = property_get_device(&error);
312  if (error) {
313  // log the error
314  msg = action_name + " " + MODULE_NAME + " " +
315  YAML_DEVICE_PROPERTY_ERROR;
316  cerr << "RVS-PEQT: " << msg;
317 
318  // log the module's result (FALSE) and abort PCIe qualification check
319  // (<device> parameter is mandatory)
320  msg = action_name + " " + MODULE_NAME + " " + PEQT_RESULT_FAIL_MESSAGE;
321  cerr << "RVS-PEQT: " << msg;
322 
323  return -1; // PCIe qualification check cannot continue
324  }
325 
326  // get the <deviceid> property value
327  int devid = property_get_deviceid(&error);
328  if (!error) {
329  if (devid != -1) {
330  deviceid = static_cast<uint16_t>(devid);
331  device_id_filtering = true;
332  }
333  } else {
334  // log the error
335  msg = action_name + " " + MODULE_NAME + " " +
336  YAML_DEVICEID_PROPERTY_ERROR;
337  cerr << "RVS-PEQT: " << msg;
338 
339  // log the module's result (FALSE) and abort PCIe qualification check
340  // (<device> parameter is mandatory)
341  msg = action_name + " " + MODULE_NAME + " " + PEQT_RESULT_FAIL_MESSAGE;
342  cerr << "RVS-PEQT: " << msg;
343 
344  return -1; // PCIe qualification check cannot continue
345  }
346 
347  // get the pci_access structure
348  pacc = pci_alloc();
349 
350  if (pacc == NULL) {
351  // log the error
352  msg = action_name + " " + MODULE_NAME + " " + PCI_ALLOC_ERROR;
353  cerr << "RVS-PEQT: " << msg;
354 
355  // log the module's result (FALSE)
356  msg = action_name + " " + MODULE_NAME + " " + PEQT_RESULT_FAIL_MESSAGE;
357  cerr << "RVS-PEQT: " << msg;
358  return 1; // PCIe qualification check cannot continue
359  }
360 
361  // compose Power Budgeting dynamic regex
362  string dyn_pb_regex_str = "^(";
363  for (i = 0; i < PB_NUM_OP_STATES; i++) {
364  dyn_pb_regex_str += pb_op_pm_states_list[i];
365  pb_op_pm_states_encodings_map.insert(std::pair<string, uint8_t>
366  (pb_op_pm_states_list[i], pb_op_pm_states_encoding[i]));
367  if (i < PB_NUM_OP_STATES - 1)
368  dyn_pb_regex_str += "|";
369  }
370  dyn_pb_regex_str += ")_(";
371  for (i = 0; i < PB_NUM_OP_TYPES; i++) {
372  dyn_pb_regex_str += pb_op_types_list[i];
373  pb_op_pm_types_encodings_map.insert(std::pair<string, uint8_t>
374  (pb_op_types_list[i], pb_op_types_encoding[i]));
375  if (i < PB_NUM_OP_TYPES - 1)
376  dyn_pb_regex_str += "|";
377  }
378  dyn_pb_regex_str += ")_(";
379  for (i = 0; i < PN_NUM_OP_POWER_RAILS; i++) {
380  dyn_pb_regex_str += pb_op_power_rails_list[i];
381  pb_op_pm_power_rails_encodings_map.insert(std::pair<string, uint8_t>
382  (pb_op_power_rails_list[i], pb_op_power_rails_encoding[i]));
383  if (i < PN_NUM_OP_POWER_RAILS - 1)
384  dyn_pb_regex_str += "|";
385  }
386 
387  dyn_pb_regex_str += ")$";
388  pb_dynamic_regex.assign(dyn_pb_regex_str);
389 
390  // initialize the PCI library
391  pci_init(pacc);
392  // get the list of devices
393  pci_scan_bus(pacc);
394 
395  // iterate over devices
396  for (dev = pacc->devices; dev; dev = dev->next) {
397  // fill in the info
398  pci_fill_info(dev,
399  PCI_FILL_IDENT | PCI_FILL_BASES | PCI_FILL_CLASS
400  | PCI_FILL_EXT_CAPS | PCI_FILL_CAPS | PCI_FILL_PHYS_SLOT);
401 
402  // computes the actual dev's location_id (sysfs entry)
403  uint16_t dev_location_id = ((((uint16_t) (dev->bus)) << 8)
404  | (dev->func));
405 
406  // check if this pci_dev corresponds to one of the AMD GPUs
407  int32_t gpu_id = rvs::gpulist::GetGpuId(dev_location_id);
408 
409  if (-1 != gpu_id) {
410  // that should be an AMD GPU
411 
412  // check for deviceid filtering
413  if (!device_id_filtering
414  || (device_id_filtering && dev->device_id == deviceid)) {
415  // check if the GPU is part of the PCIe check
416  // (either device: all or the gpu_id is in
417  // the device: <gpu id> list
418 
419  bool cur_gpu_selected = false;
420 
421  if (device_all_selected) {
422  cur_gpu_selected = true;
423  } else {
424  // search for this gpu in the list
425  // provided under the <device> property
426  vector<string>::iterator it_gpu_id = find(
427  device_prop_gpu_id_list.begin(),
428  device_prop_gpu_id_list.end(),
429  std::to_string(gpu_id));
430 
431  if (it_gpu_id != device_prop_gpu_id_list.end())
432  cur_gpu_selected = true;
433  }
434 
435  if (cur_gpu_selected) {
436  amd_gpus_found = true;
437  if (!get_gpu_all_pcie_capabilities(dev,
438  static_cast<uint16_t>(gpu_id)))
439  pci_infra_qual_result = false;
440  }
441  }
442  }
443  }
444 
445  pci_cleanup(pacc);
446 
447  if (!amd_gpus_found)
448  pci_infra_qual_result = false;
449 
450  msg = "[" + action_name + "] " + MODULE_NAME + " "
451  + (pci_infra_qual_result ?
452  PEQT_RESULT_PASS_MESSAGE : PEQT_RESULT_FAIL_MESSAGE);
453  log(msg.c_str(), rvs::logresults);
454 
455  if (bjson && json_root_node != NULL) {
456  rvs::lp::AddString(json_root_node, "RESULT",
457  (pci_infra_qual_result ?
458  PEQT_RESULT_PASS_MESSAGE : PEQT_RESULT_FAIL_MESSAGE));
459  rvs::lp::LogRecordFlush(json_root_node);
460  }
461 
462  return 0;
463 }
