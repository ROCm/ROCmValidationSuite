128 bool peqt_action::get_all_common_config_keys(void) {
129  string msg, sdevid, sdev;
130  int error;
131  bool res;
132  res = true;
133 
134  // get the action name
135  if (property_get(RVS_CONF_NAME_KEY, &action_name)) {
136  rvs::lp::Err("Action name missing", MODULE_NAME_CAPS);
137  res = false;
138  }
139 
140  // get <device> property value (a list of gpu id)
141  if ((error = property_get_device())) {
142  switch (error) {
143  case 1:
144  msg = "Invalid 'device' key value.";
145  break;
146  case 2:
147  msg = "Missing 'device' key.";
148  break;
149  }
150  rvs::lp::Err(msg, MODULE_NAME_CAPS, action_name);
151  res = false;
152  }
153 
154  // get the <deviceid> property value if provided
155  if (property_get_int<uint16_t>(RVS_CONF_DEVICEID_KEY,
156  &property_device_id, 0u)) {
157  msg = "Invalid 'deviceid' key value.";
158  rvs::lp::Err(msg, MODULE_NAME_CAPS, action_name);
159  res = false;
160  }
161 
162  return res;
163 }

173 bool peqt_action::get_gpu_all_pcie_capabilities(struct pci_dev *dev,
174  uint16_t gpu_id) {
175  char buff[CHAR_BUFF_MAX_SIZE];
176  string prop_name, msg;
177  bool pci_infra_qual_result = true;
178  map<string, string>::iterator it; // module's properties map iterator
179  void *json_pcaps_node = NULL;
180  uint8_t i;
181 
182  if (bjson) {
183  unsigned int sec;
184  unsigned int usec;
185  rvs::lp::get_ticks(&sec, &usec);
186 
187  json_pcaps_node = rvs::lp::LogRecordCreate(MODULE_NAME,
188  action_name.c_str(), rvs::loginfo, sec, usec);
189 
190  if (json_pcaps_node == NULL) {
191  // log the error
192  msg = JSON_CREATE_NODE_ERROR;
193  rvs::lp::Err(msg, MODULE_NAME, action_name);
194  return false;
195  }
196  }
197 
198  if (bjson && json_pcaps_node != NULL) {
199  rvs::lp::AddString(json_pcaps_node, RVS_JSON_LOG_GPU_ID_KEY,
200  std::to_string(gpu_id));
201  }
202 
203  for (it = property.begin(); it != property.end(); ++it) {
204  // skip the "capability."
205  string prop_name = it->first.substr(it->first.find_last_of(".") + 1);
206  bool prop_found = false;
207  for (i = 0; i < PCI_DEV_NUM_CAPABILITIES; i++) {
208  if ((prop_name == pcie_cap_names[i]) &&
209  ( dev != NULL )){
210  prop_found = true;
211  // call the capability's corresponding function
212  (*arr_prop_pfunc_names[i])(dev, buff);
213 
214  // log the capability's value
215  msg = "[" + action_name + "] " + MODULE_NAME + " " +
216  pcie_cap_names[i] + " " + buff;
217  rvs::lp::Log(msg, rvs::loginfo);
218 
219  if (bjson && json_pcaps_node != NULL) {
220  rvs::lp::AddString(json_pcaps_node, pcie_cap_names[i],
221  buff);
222  }
223 
224  // check for regex match
225  if (it->second != "") {
226  try {
227  regex prop_regex(it->second);
228  if (!regex_match(buff, prop_regex)) {
229  pci_infra_qual_result = false;
230  }
231  } catch (const std::regex_error& e) {
232  // log the regex error
233  msg = std::string(YAML_REGULAR_EXPRESSION_ERROR)
234  + " at '"
235  + it->second + "'";
236  rvs::lp::Err(msg, MODULE_NAME, action_name);;
237  }
238  }
239  break;
240  }
241  }
242 
243  if (!prop_found &&
244  it->first.find(YAML_CAPABILITY_TAG) != string::npos) {
245  // the property was not found among those that
246  // have fixed/constant name => check whether it's
247  // a dynamic Power Budgeting capability
248  if (regex_match(prop_name, pb_dynamic_regex)) {
249  // no additional checks are needed (itetator != .end() && npos)
250  // because the prop_name already matched the regular expression
251 
252  std::size_t pos_pb_pm_state =
253  prop_name.find_first_of(PB_OP_COND_DYN_DELIMITER);
254  map<string, uint8_t>::iterator it_pb_pm_state =
255  pb_op_pm_states_encodings_map.find
256  (prop_name.substr(0, pos_pb_pm_state));
257  uint8_t pb_op_pm_state = it_pb_pm_state->second;
258 
259  std::size_t pos_pb_type =
260  prop_name.find(PB_OP_COND_DYN_DELIMITER,
261  pos_pb_pm_state + 1);
262  map<string, uint8_t>::iterator it_pb_type =
263  pb_op_pm_types_encodings_map.find
264  (prop_name.substr(pos_pb_pm_state + 1,
265  pos_pb_type - pos_pb_pm_state - 1));
266  uint8_t pb_op_pm_type = it_pb_type->second;
267 
268  map<string, uint8_t>::iterator it_pb_power_rail =
269  pb_op_pm_power_rails_encodings_map.find
270  (prop_name.substr(pos_pb_type + 1));
271  uint8_t pb_op_power_rail = it_pb_power_rail->second;
272  // query for power budgeting capabilities
273  get_pwr_budgeting(dev, pb_op_pm_state, pb_op_pm_type,
274  pb_op_power_rail, buff);
275 
276  // log the capability's value
277  msg = "[" + action_name + "] " + MODULE_NAME + " " + prop_name
278  + " " + buff;
279  rvs::lp::Log(msg, rvs::loginfo);
280 
281  if (bjson && json_pcaps_node != NULL) {
282  rvs::lp::AddString(json_pcaps_node, prop_name,
283  buff);
284  }
285 
286  // check for regex match
287  if (it->second != "") {
288  try {
289  regex prop_regex(it->second);
290  if (!regex_match(buff, prop_regex)) {
291  pci_infra_qual_result = false;
292  }
293  } catch (const std::regex_error& e) {
294  pci_infra_qual_result = false;
295  // log the regex error
296  msg = action_name + " " + MODULE_NAME + " "
297  + YAML_REGULAR_EXPRESSION_ERROR + " at '"
298  + it->second + "'";
299  rvs::lp::Err(msg, MODULE_NAME, action_name);
300  }
301  }
302  }
303  }
304  }
305 
306  rvs::lp::LogRecordFlush(json_pcaps_node);
307 
308  return pci_infra_qual_result;
309 }

316 int peqt_action::run(void) {
317  string msg;
318  map<string, string>::iterator it; // module's properties map iterator
319  bool pci_infra_qual_result = true; // PCI qualification result
320  bool amd_gpus_found = false;
321  uint8_t i;
322  unsigned int sec;
323  unsigned int usec;
324 
325  struct pci_access *pacc;
326  struct pci_dev *dev;
327 
328  RVSTRACE_
329  bjson = false; // already initialized in the default constructor
330 
331  // check for -j flag (json logging)
332  if (property.find("cli.-j") != property.end()) {
333  bjson = true;
334  }
335 
336  if (!get_all_common_config_keys()) {
337  msg = "Error in get_all_common_config_keys()";
338  rvs::lp::Err(msg, MODULE_NAME_CAPS, action_name);
339  return -1;
340  }
341 
342  // get the pci_access structure
343  pacc = pci_alloc();
344 
345  if (pacc == NULL) {
346  // log the error
347  msg = PCI_ALLOC_ERROR;
348  rvs::lp::Err(msg, MODULE_NAME, action_name);
349  return 1; // PCIe qualification check cannot continue
350  }
351 
352  // compose Power Budgeting dynamic regex
353  string dyn_pb_regex_str = "^(";
354  for (i = 0; i < PB_NUM_OP_STATES; i++) {
355  dyn_pb_regex_str += pb_op_pm_states_list[i];
356  pb_op_pm_states_encodings_map.insert(std::pair<string, uint8_t>
357  (pb_op_pm_states_list[i], pb_op_pm_states_encoding[i]));
358  if (i < PB_NUM_OP_STATES - 1)
359  dyn_pb_regex_str += "|";
360  }
361  dyn_pb_regex_str += ")_(";
362  for (i = 0; i < PB_NUM_OP_TYPES; i++) {
363  dyn_pb_regex_str += pb_op_types_list[i];
364  pb_op_pm_types_encodings_map.insert(std::pair<string, uint8_t>
365  (pb_op_types_list[i], pb_op_types_encoding[i]));
366  if (i < PB_NUM_OP_TYPES - 1)
367  dyn_pb_regex_str += "|";
368  }
369  dyn_pb_regex_str += ")_(";
370  for (i = 0; i < PN_NUM_OP_POWER_RAILS; i++) {
371  dyn_pb_regex_str += pb_op_power_rails_list[i];
372  pb_op_pm_power_rails_encodings_map.insert(std::pair<string, uint8_t>
373  (pb_op_power_rails_list[i], pb_op_power_rails_encoding[i]));
374  if (i < PN_NUM_OP_POWER_RAILS - 1)
375  dyn_pb_regex_str += "|";
376  }
377 
378  dyn_pb_regex_str += ")$";
379  pb_dynamic_regex.assign(dyn_pb_regex_str);
380 
381  RVSTRACE_
382  // initialize the PCI library
383  pci_init(pacc);
384  // get the list of devices
385  pci_scan_bus(pacc);
386 
387  RVSTRACE_
388  // iterate over devices
389  for (dev = pacc->devices; dev; dev = dev->next) {
390  RVSTRACE_
391 
392  if(dev == NULL) {
393  msg = "[" + action_name + "] " + MODULE_NAME + " "
394  + "Device is null ";
395  rvs::lp::Log(msg, rvs::logresults);
396  return false;
397  }
398 
399  // fill in the info
400  pci_fill_info(dev,
401  PCI_FILL_IDENT | PCI_FILL_BASES | PCI_FILL_CLASS
402  | PCI_FILL_EXT_CAPS | PCI_FILL_CAPS | PCI_FILL_PHYS_SLOT);
403 
404  // computes the actual dev's location_id (sysfs entry)
405  uint16_t dev_location_id = ((((uint16_t) (dev->bus)) << 8)
406  | ((uint16_t) (dev->dev)) << 3);
407 
408  // check if this pci_dev corresponds to one of the AMD GPUs
409  uint16_t gpu_id;
410  // if not and AMD GPU just continue
411  if (rvs::gpulist::location2gpu(dev_location_id, &gpu_id)) {
412  RVSTRACE_
413  continue;
414  }
415 
416  // check for deviceid filtering
417  if (property_device_id > 0 && dev->device_id != property_device_id) {
418  RVSTRACE_
419  continue;
420  }
421 
422  if (!property_device_all) {
423  RVSTRACE_
424  if (find(property_device.begin(), property_device.end(), gpu_id) ==
425  property_device.end()) {
426  RVSTRACE_
427  continue;
428  }
429  }
430  RVSTRACE_
431 
432  amd_gpus_found = true;
433  if (!get_gpu_all_pcie_capabilities(dev, gpu_id)) {
434  RVSTRACE_
435  pci_infra_qual_result = false;
436  }
437  RVSTRACE_
438  }
439 
440  RVSTRACE_
441  pci_cleanup(pacc);
442 
443  if (!amd_gpus_found) {
444  msg = "No matching GPUs found";
445  rvs::lp::Err(msg, MODULE_NAME, action_name);
446  return -1;
447  }
448 
449  RVSTRACE_
450  msg = "[" + action_name + "] " + MODULE_NAME + " "
451  + (pci_infra_qual_result ?
452  PEQT_RESULT_PASS_MESSAGE : PEQT_RESULT_FAIL_MESSAGE);
453  rvs::lp::Log(msg, rvs::logresults);
454 
455  if (bjson) {
456  RVSTRACE_
457  rvs::lp::get_ticks(&sec, &usec);
458  json_root_node = rvs::lp::LogRecordCreate(MODULE_NAME,
459  action_name.c_str(), rvs::logresults, sec, usec);
460  if (json_root_node == NULL) {
461  // log the error
462  msg = JSON_CREATE_NODE_ERROR;
463  rvs::lp::Err(msg, MODULE_NAME, action_name);
464  return -1;
465  }
466 
467  if (pci_infra_qual_result) {
468  rvs::lp::AddInt(json_root_node, "Sts", 1);
469  rvs::lp::AddString(json_root_node, "pass", PEQT_RESULT_PASS_MESSAGE);
470  } else {
471  rvs::lp::AddInt(json_root_node, "Sts", 0);
472  rvs::lp::AddString(json_root_node, "pass", PEQT_RESULT_FAIL_MESSAGE);
473  }
474 
475  rvs::lp::LogRecordFlush(json_root_node);
476  }
477 
478  RVSTRACE_
479  return 0;
480 }
