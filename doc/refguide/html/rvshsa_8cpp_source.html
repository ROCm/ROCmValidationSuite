116 void rvs::hsa::print_hsa_status(string message, hsa_status_t st) {
117  string log_msg = message;
118  // skip successfull messages
119  if (st == HSA_STATUS_SUCCESS) {
120  return;
121  }
122  switch (st) {
123  case HSA_STATUS_SUCCESS : {
124  log_msg = message + " The function has been executed successfully.";
125  break;
126  };
127  case HSA_STATUS_INFO_BREAK : {
128  log_msg = message + " A traversal over a list of elements has been " +
129  "interrupted by the application before completing.";
130  break;
131  };
132  case HSA_STATUS_ERROR : {
133  log_msg = message + " A generic error has occurred.";
134  break;
135  };
136  case HSA_STATUS_ERROR_INVALID_ARGUMENT : {
137  log_msg = message + " One of the actual arguments does not meet a " +
138  "precondition stated in the documentation of the corresponding formal " +
139  "argument.";
140  break;
141  };
142  case HSA_STATUS_ERROR_INVALID_QUEUE_CREATION : {
143  log_msg = message + " The requested queue creation is not valid.";
144  break;
145  };
146  case HSA_STATUS_ERROR_INVALID_ALLOCATION : {
147  log_msg = message + " The requested allocation is not valid.";
148  break;
149  };
150  case HSA_STATUS_ERROR_INVALID_AGENT : {
151  log_msg = message + " The agent is invalid.";
152  break;
153  };
154  case HSA_STATUS_ERROR_INVALID_REGION : {
155  log_msg = message + " The memory region is invalid.";
156  break;
157  };
158  case HSA_STATUS_ERROR_INVALID_SIGNAL : {
159  log_msg = message + " The signal is invalid.";
160  break;
161  };
162  case HSA_STATUS_ERROR_INVALID_QUEUE : {
163  log_msg = message + " The queue is invalid.";
164  break;
165  };
166  case HSA_STATUS_ERROR_OUT_OF_RESOURCES : {
167  log_msg = message + " The HSA runtime failed to allocate the necessary " +
168  "resources. This error may also occur when the HSA runtime needs to " +
169  "spawn threads or create internal OS-specific events.";
170  break;
171  };
172  case HSA_STATUS_ERROR_INVALID_PACKET_FORMAT : {
173  log_msg = message + " The AQL packet is malformed.";
174  break;
175  };
176  case HSA_STATUS_ERROR_RESOURCE_FREE : {
177  log_msg = message +
178  " An error has been detected while releasing a resource.";
179  break;
180  };
181  case HSA_STATUS_ERROR_NOT_INITIALIZED : {
182  log_msg = message +
183  " An API other than ::hsa_init has been invoked while the reference " +
184  "count of the HSA runtime is 0.";
185  break;
186  };
187  case HSA_STATUS_ERROR_REFCOUNT_OVERFLOW : {
188  log_msg = message + " The maximum reference count for the object has " +
189  "been reached.";
190  break;
191  };
192  case HSA_STATUS_ERROR_INCOMPATIBLE_ARGUMENTS : {
193  log_msg = message + " The arguments passed to a functions are not " +
194  "compatible.";
195  break;
196  };
197  case HSA_STATUS_ERROR_INVALID_INDEX : {
198  log_msg = message + " The index is invalid.";
199  break;
200  };
201  case HSA_STATUS_ERROR_INVALID_ISA : {
202  log_msg = message + " The instruction set architecture is invalid.";
203  break;
204  };
205  case HSA_STATUS_ERROR_INVALID_ISA_NAME : {
206  log_msg = message + " The instruction set architecture name is invalid.";
207  break;
208  };
209  case HSA_STATUS_ERROR_INVALID_CODE_OBJECT : {
210  log_msg = message + " The code object is invalid.";
211  break;
212  };
213  case HSA_STATUS_ERROR_INVALID_EXECUTABLE : {
214  log_msg = message + " The executable is invalid.";
215  break;
216  };
217  case HSA_STATUS_ERROR_FROZEN_EXECUTABLE : {
218  log_msg = message + " The executable is frozen.";
219  break;
220  };
221  case HSA_STATUS_ERROR_INVALID_SYMBOL_NAME : {
222  log_msg = message + " There is no symbol with the given name.";
223  break;
224  };
225  case HSA_STATUS_ERROR_VARIABLE_ALREADY_DEFINED : {
226  log_msg = message + " The variable is already defined.";
227  break;
228  };
229  case HSA_STATUS_ERROR_VARIABLE_UNDEFINED : {
230  log_msg = message + " The variable is undefined.";
231  break;
232  };
233  case HSA_STATUS_ERROR_EXCEPTION : {
234  log_msg = message +
235  " An HSAIL operation resulted on a hardware exception.";
236  break;
237  };
238  default : {
239  log_msg = message + " Unknown error.";
240  break;
241  }
242  }
243  rvs::lp::Log(log_msg, rvs::logdebug);
244 }

257 void rvs::hsa::InitAgents() {
258  hsa_status_t status;
259  string log_msg;
260 
261  RVSTRACE_
262  // Initialize Roc Runtime
263  status = hsa_init();
264  print_hsa_status(__FILE__, __LINE__, __func__, "hsa_init()", status);
265 
266  // Initialize profiling
267  status = hsa_amd_profiling_async_copy_enable(true);
268  print_hsa_status(__FILE__, __LINE__, __func__,
269  "hsa_amd_profiling_async_copy_enable()", status);
270 
271  // Populate the lists of agents
272  status = hsa_iterate_agents(ProcessAgent, &agent_list);
273  print_hsa_status(__FILE__, __LINE__, __func__,
274  "hsa_iterate_agents()", status);
275 
276  for (uint32_t i = 0; i < agent_list.size(); i++) {
277  rvs::lp::Log("[RVSHSA] ============================", rvs::logdebug);
278  log_msg = "[RVSHSA] InitAgents - agent with name = " +
279  agent_list[i].agent_name + " and device_type = " +
280  agent_list[i].agent_device_type;
281  rvs::lp::Log(log_msg.c_str(), rvs::logtrace);
282  rvs::lp::Log("[RVSHSA] ============================", rvs::logdebug);
283 
284  // Populate the list of memory pools
285  RVSTRACE_
286  status = hsa_amd_agent_iterate_memory_pools(agent_list[i].agent,
287  ProcessMemPool, &agent_list[i]);
288  print_hsa_status(__FILE__, __LINE__, __func__,
289  "hsa_amd_agent_iterate_memory_pools()", status);
290 
291  // separate the lists
292  if (agent_list[i].agent_device_type == "CPU") {
293  cpu_list.push_back(agent_list[i]);
294  } else if (agent_list[i].agent_device_type == "GPU") {
295  gpu_list.push_back(agent_list[i]);
296  } else {
297  log_msg = "Unexpected agent type: " + agent_list[i].agent_device_type;
298  rvs::lp::Log(log_msg, rvs::logdebug);
299  }
300  }
301 
302  // Initialize the list of buffer sizes to use in copy/read/write operations
303  // For All Copy operations use only one buffer size
304  if (size_list.size() == 0) {
305  uint32_t size_len = sizeof(DEFAULT_SIZE_LIST)/sizeof(uint32_t);
306  for (uint32_t idx = 0; idx < size_len; idx++) {
307  size_list.push_back(DEFAULT_SIZE_LIST[idx]);
308  }
309  }
310 
311  std::sort(size_list.begin(), size_list.end());
312 }

324 hsa_status_t rvs::hsa::ProcessAgent(hsa_agent_t agent, void* data) {
325  hsa_status_t status;
326  char agent_name[64];
327  hsa_device_type_t device_type;
328  string log_msg, log_agent_name;
329  uint32_t node;
330  AgentInformation agent_info;
331 
332  // get agent list
333  vector<AgentInformation>* agent_l =
334  reinterpret_cast<vector<AgentInformation>*>(data);
335 
336  RVSTRACE_;
337 
338  // Get the name of the agent
339  status = hsa_agent_get_info(agent, HSA_AGENT_INFO_NAME, agent_name);
340  print_hsa_status(__FILE__, __LINE__, __func__,
341  "[HSA_AGENT_INFO_NAME", status);
342  rvs::lp::Log(string("agent_name: ") + agent_name, rvs::logdebug);
343 
344  // Get device type
345  status = hsa_agent_get_info(agent, HSA_AGENT_INFO_DEVICE, &device_type);
346  print_hsa_status(__FILE__, __LINE__, __func__,
347  "[RVSHSA] HSA_AGENT_INFO_DEVICE", status);
348 
349  status = hsa_agent_get_info(agent, HSA_AGENT_INFO_NODE, &node);
350  print_hsa_status(__FILE__, __LINE__, __func__,
351  "[RVSHSA] HSA_AGENT_INFO_NODE", status);
352  agent_info.node = node;
353  rvs::lp::Log("node: " + std::to_string(node), rvs::logdebug);
354 
355  log_agent_name = agent_name;
356  log_msg = "[RVSHSA] Found agent with name = " + log_agent_name +
357  " and device_type = ";
358  switch (device_type) {
359  case HSA_DEVICE_TYPE_CPU : {
360  agent_info.agent_device_type = "CPU";
361  log_msg = log_msg + "CPU.";
362  break;
363  };
364  case HSA_DEVICE_TYPE_GPU : {
365  agent_info.agent_device_type = "GPU";
366  log_msg = log_msg + "GPU.";
367  break;
368  };
369  case HSA_DEVICE_TYPE_DSP : {
370  agent_info.agent_device_type = "DSP";
371  log_msg = log_msg + "DSP.";
372  break;
373  };
374  }
375  rvs::lp::Log(log_msg.c_str(), rvs::logdebug);
376 
377  // add agent to list
378  agent_info.agent = agent;
379  agent_info.agent_name = log_agent_name;
380  agent_l->push_back(agent_info);
381 
382  return HSA_STATUS_SUCCESS;
383 }

395 hsa_status_t rvs::hsa::ProcessMemPool(hsa_amd_memory_pool_t pool, void* data) {
396  hsa_status_t status;
397 
398  RVSTRACE_
399  // get current agents memory pools
400  AgentInformation* agent_info = reinterpret_cast<AgentInformation*>(data);
401 
402  // Query pools' segment, report only pools from global segment
403  hsa_amd_segment_t segment;
404  status = hsa_amd_memory_pool_get_info(pool,
405  HSA_AMD_MEMORY_POOL_INFO_SEGMENT,
406  &segment);
407  print_hsa_status(__FILE__, __LINE__, __func__,
408  "hsa_amd_memory_pool_get_info()", status);
409  if (HSA_AMD_SEGMENT_GLOBAL != segment) {
410  RVSTRACE_
411  return HSA_STATUS_SUCCESS;
412  }
413 
414  RVSTRACE_
415  // Determine if allocation is allowed in this pool
416  // Report only pools that allow an alloction by user
417  bool alloc = false;
418  status = hsa_amd_memory_pool_get_info(pool,
419  HSA_AMD_MEMORY_POOL_INFO_RUNTIME_ALLOC_ALLOWED,
420  &alloc);
421  print_hsa_status(__FILE__, __LINE__, __func__,
422  "HSA_AMD_MEMORY_POOL_INFO_RUNTIME_ALLOC_ALLOWED", status);
423  if (alloc != true) {
424  RVSTRACE_
425  return HSA_STATUS_SUCCESS;
426  }
427 
428  RVSTRACE_
429  // Query the max allocatable size
430  size_t max_size = 0;
431  status = hsa_amd_memory_pool_get_info(pool,
432  HSA_AMD_MEMORY_POOL_INFO_SIZE,
433  &max_size);
434  print_hsa_status(__FILE__, __LINE__, __func__,
435  "HSA_AMD_MEMORY_POOL_INFO_SIZE", status);
436  agent_info->max_size_list.push_back(max_size);
437 
438  // Determine if the pools is accessible to all agents
439  bool access_to_all = false;
440  status = hsa_amd_memory_pool_get_info(pool,
441  HSA_AMD_MEMORY_POOL_INFO_ACCESSIBLE_BY_ALL,
442  &access_to_all);
443  print_hsa_status(__FILE__, __LINE__, __func__,
444  "HSA_AMD_MEMORY_POOL_INFO_ACCESSIBLE_BY_ALL", status);
445 
446  // Determine type of access to owner agent
447  hsa_amd_memory_pool_access_t owner_access;
448  hsa_agent_t agent = agent_info->agent;
449  status = hsa_amd_agent_memory_pool_get_info(agent, pool,
450  HSA_AMD_AGENT_MEMORY_POOL_INFO_ACCESS,
451  &owner_access);
452  print_hsa_status(__FILE__, __LINE__, __func__, status);
453 
454  // Determine if the pool is fine-grained or coarse-grained
455  uint32_t flag = 0;
456  status = hsa_amd_memory_pool_get_info(pool,
457  HSA_AMD_MEMORY_POOL_INFO_GLOBAL_FLAGS,
458  &flag);
459  print_hsa_status(__FILE__, __LINE__, __func__,
460  "HSA_AMD_MEMORY_POOL_INFO_GLOBAL_FLAGS", status);
461  bool is_kernarg = (HSA_AMD_MEMORY_POOL_GLOBAL_FLAG_KERNARG_INIT & flag);
462 
463  // Update the pool handle for system memory if kernarg is true
464  rvs::lp::Log("[RVSHSA] ****************************************",
465  rvs::logdebug);
466  if (is_kernarg) {
467  agent_info->sys_pool = pool;
468  rvs::lp::Log("[RVSHSA] Found system memory region", rvs::logdebug);
469  } else if (owner_access != HSA_AMD_MEMORY_POOL_ACCESS_NEVER_ALLOWED) {
470  rvs::lp::Log("[RVSHSA] Found regular memory region", rvs::logdebug);
471  }
472  rvs::lp::Log("[RVSHSA] ****************************************",
473  rvs::logdebug);
474  agent_info->mem_pool_list.push_back(pool);
475 
476  return HSA_STATUS_SUCCESS;
477 }

527  double end = std::max(async_time_fwd.end, async_time_rev.end);
528  RVSTRACE_
529  return(end - start);

545 int rvs::hsa::Allocate(int SrcAgent, int DstAgent, size_t Size,
546  hsa_amd_memory_pool_t* pSrcPool, void** SrcBuff,
547  hsa_amd_memory_pool_t* pDstPool, void** DstBuff) {
548  hsa_status_t status;
549  void* srcbuff = nullptr;
550  void* dstbuff = nullptr;
551 
552  // iterate over src pools
553  for (size_t i = 0; i < agent_list[SrcAgent].mem_pool_list.size(); i++) {
554  RVSTRACE_
555  // size too small, continue
556  if (Size > agent_list[SrcAgent].max_size_list[i]) {
557  RVSTRACE_
558  continue;
559  }
560 
561  RVSTRACE_
562  // try allocating source buffer
563  status = hsa_amd_memory_pool_allocate(
564  agent_list[SrcAgent].mem_pool_list[i], Size, 0, &srcbuff);
565  if (status != HSA_STATUS_SUCCESS) {
566  print_hsa_status(__FILE__, __LINE__, __func__,
567  "hsa_amd_memory_pool_allocate()",
568  status);
569  continue;
570  }
571 
572  RVSTRACE_
573  // iterate over dst pools
574  for (size_t j = 0; j < agent_list[DstAgent].mem_pool_list.size(); j++) {
575  RVSTRACE_
576  // size too small, continue
577  if (Size > agent_list[DstAgent].max_size_list[j]) {
578  RVSTRACE_
579  continue;
580  }
581 
582  RVSTRACE_
583  // check if src agent has access to this dst agent's pool
584  hsa_amd_memory_pool_access_t access =
585  HSA_AMD_MEMORY_POOL_ACCESS_NEVER_ALLOWED;
586  if (agent_list[SrcAgent].agent_device_type == "CPU") {
587  RVSTRACE_
588  status = hsa_amd_agent_memory_pool_get_info(
589  agent_list[DstAgent].agent,
590  agent_list[SrcAgent].mem_pool_list[j],
591  HSA_AMD_AGENT_MEMORY_POOL_INFO_ACCESS,
592  &access);
593  } else {
594  RVSTRACE_
595  status = hsa_amd_agent_memory_pool_get_info(
596  agent_list[SrcAgent].agent,
597  agent_list[DstAgent].mem_pool_list[j],
598  HSA_AMD_AGENT_MEMORY_POOL_INFO_ACCESS,
599  &access);
600  }
601  print_hsa_status(__FILE__, __LINE__, __func__,
602  "hsa_amd_agent_memory_pool_get_info()",
603  status);
604 
605  if (access == HSA_AMD_MEMORY_POOL_ACCESS_NEVER_ALLOWED) {
606  RVSTRACE_
607  continue;
608  }
609  RVSTRACE_
610  // try allocating destination buffer
611  status = hsa_amd_memory_pool_allocate(
612  agent_list[DstAgent].mem_pool_list[j], Size, 0, &dstbuff);
613  print_hsa_status(__FILE__, __LINE__, __func__,
614  "hsa_amd_memory_pool_allocate()",
615  status);
616  if (status != HSA_STATUS_SUCCESS) {
617  RVSTRACE_
618  continue;
619  }
620 
621  RVSTRACE_
622  // destination buffer allocated,
623  // give access to agents
624 
625  RVSTRACE_
626 
627  // determine which one is a cpu and allow access on the other agent
628  if (agent_list[SrcAgent].agent_device_type == "CPU") {
629  RVSTRACE_
630  status = hsa_amd_agents_allow_access(1,
631  &agent_list[DstAgent].agent,
632  NULL,
633  srcbuff);
634  } else {
635  RVSTRACE_
636  status = hsa_amd_agents_allow_access(1,
637  &agent_list[SrcAgent].agent,
638  NULL,
639  dstbuff);
640  }
641  print_hsa_status(__FILE__, __LINE__, __func__,
642  "hsa_amd_agents_allow_access()",
643  status);
644  if (status != HSA_STATUS_SUCCESS) {
645  RVSTRACE_
646  // do cleanup
647  hsa_amd_memory_pool_free(dstbuff);
648  dstbuff = nullptr;
649  continue;
650  }
651 
652  RVSTRACE_
653  // all OK, set output parameters:
654  *pSrcPool = agent_list[SrcAgent].mem_pool_list[i];
655  *pDstPool = agent_list[DstAgent].mem_pool_list[j];
656  *SrcBuff = srcbuff;
657  *DstBuff = dstbuff;
658 
659  return 0;
660  } // end of dst agent pool loop
661 
662  RVSTRACE_
663  // suitable destination buffer not foud, deallocate src buff and exit
664  hsa_amd_memory_pool_free(srcbuff);
665  } // end of src agent pool loop
666 
667  RVSTRACE_
668  return -1;
669 }

682 int rvs::hsa::SendTraffic(uint32_t SrcNode, uint32_t DstNode,
683  size_t Size, bool bidirectional,
684  double* Duration) {
685  hsa_status_t status;
686  int sts;
687 
688  int32_t src_ix_fwd;
689  int32_t dst_ix_fwd;
690  hsa_amd_memory_pool_t src_pool_fwd;
691  hsa_amd_memory_pool_t dst_pool_fwd;
692  void* src_ptr_fwd = nullptr;
693  void* dst_ptr_fwd = nullptr;
694  hsa_signal_t signal_fwd;
695 
696  int32_t src_ix_rev;
697  int32_t dst_ix_rev;
698  hsa_amd_memory_pool_t src_pool_rev;
699  hsa_amd_memory_pool_t dst_pool_rev;
700  void* src_ptr_rev = nullptr;
701  void* dst_ptr_rev = nullptr;
702  hsa_signal_t signal_rev;
703 
704  RVSTRACE_
705 
706  // given NUMA nodes, find agent indexes
707  src_ix_fwd = FindAgent(SrcNode);
708  dst_ix_fwd = FindAgent(DstNode);
709  if (src_ix_fwd < 0 || dst_ix_fwd < 0) {
710  RVSTRACE_
711  return -1;
712  }
713 
714  // allocate buffers and grant permissions for forward transfer
715  sts = Allocate(src_ix_fwd, dst_ix_fwd, Size,
716  &src_pool_fwd, &src_ptr_fwd,
717  &dst_pool_fwd, &dst_ptr_fwd);
718  if (sts) {
719  RVSTRACE_
720  return -1;
721  }
722 
723  // Create a signal to wait on copy operation
724  status = hsa_signal_create(1, 0, NULL, &signal_fwd);
725  print_hsa_status(__FILE__, __LINE__, __func__,
726  "hsa_signal_create()",
727  status);
728  if (status != HSA_STATUS_SUCCESS) {
729  hsa_amd_memory_pool_free(src_ptr_fwd);
730  hsa_amd_memory_pool_free(dst_ptr_fwd);
731  RVSTRACE_

794  RVSTRACE_
795  while (hsa_signal_wait_acquire(signal_rev, HSA_SIGNAL_CONDITION_LT,
796  1, uint64_t(-1), HSA_WAIT_STATE_ACTIVE)) {}

802 
803  hsa_amd_memory_pool_free(src_ptr_fwd);
804  hsa_amd_memory_pool_free(dst_ptr_fwd);
805  hsa_signal_destroy(signal_fwd);
806 
807  if (bidirectional) {
808  RVSTRACE_
809  hsa_amd_memory_pool_free(src_ptr_rev);
810  hsa_amd_memory_pool_free(dst_ptr_rev);
811  hsa_signal_destroy(signal_rev);
812  }
813  RVSTRACE_
814
