152 void pebbworker_b2b::run() {
153  std::chrono::time_point<std::chrono::system_clock> pebb_start_time;
154  std::chrono::time_point<std::chrono::system_clock> pebb_end_time;
155  hsa_status_t status;
156  int sts;
157 
158  RVSTRACE_
159 
160  // enable test
161  brun = true;
162 
163  // allocate buffers and grant permissions for forward transfer
164  if (prop_h2d) {
165  sts = pHsa->Allocate(ctx_fwd.SrcAgentIx, ctx_fwd.DstAgentIx, b2b_block_size,
166  &ctx_fwd.SrcPool, &ctx_fwd.pSrcBuff,
167  &ctx_fwd.DstPool, &ctx_fwd.pDstBuff);
168  if (sts) {
169  RVSTRACE_
170  deinit();
171  return;
172  }
173 
174  // Create a signal to wait on forward copy operation
175  if (HSA_STATUS_SUCCESS !=
176  (status = hsa_signal_create(1, 0, NULL, &ctx_fwd.Sig))) {
177  rvs::hsa::print_hsa_status(__FILE__, __LINE__, __func__,
178  "hsa_signal_create()", status);
179  RVSTRACE_
180  deinit();
181  return;
182  }
183  }
184 
185  // allocate buffers and grant permissions for reverse transfer
186  if (prop_d2h) {
187  sts = pHsa->Allocate(ctx_rev.SrcAgentIx, ctx_rev.DstAgentIx, b2b_block_size,
188  &ctx_rev.SrcPool, &ctx_rev.pSrcBuff,
189  &ctx_rev.DstPool, &ctx_rev.pDstBuff);
190 
191  if (sts) {
192  RVSTRACE_
193  deinit();
194  return;
195  }
196 
197  // Create a signal to wait on reverse copy operation
198  if (HSA_STATUS_SUCCESS !=
199  (status = hsa_signal_create(1, 0, NULL, &ctx_rev.Sig))) {
200  rvs::hsa::print_hsa_status(__FILE__, __LINE__, __func__,
201  "hsa_signal_create()", status);
202  RVSTRACE_
203  deinit();
204  return;
205  }
206  }
207 
208 
209  pebb_start_time = std::chrono::system_clock::now();
210  while (brun) {
211  // initiate forward transfer
212  if (prop_h2d) {
213  RVSTRACE_
214  hsa_signal_store_relaxed(ctx_fwd.Sig, 1);
215  if (HSA_STATUS_SUCCESS !=
216  (status = hsa_amd_memory_async_copy(
217  ctx_fwd.pDstBuff, ctx_fwd.DstAgent,
218  ctx_fwd.pSrcBuff, ctx_fwd.SrcAgent,
219  b2b_block_size,
220  0, NULL, ctx_fwd.Sig))) {
221  rvs::hsa::print_hsa_status(__FILE__, __LINE__, __func__,
222  "hsa_amd_memory_async_copy()",
223  status);
224  break;
225  }
226  }
227 
228  if (prop_d2h) {
229  RVSTRACE_
230  // initiate reverse transfer
231  hsa_signal_store_relaxed(ctx_rev.Sig, 1);
232  if (HSA_STATUS_SUCCESS != (status = hsa_amd_memory_async_copy(
233  ctx_rev.pDstBuff, ctx_rev.DstAgent,
234  ctx_rev.pSrcBuff, ctx_rev.SrcAgent,
235  b2b_block_size,
236  0, NULL, ctx_rev.Sig))) {
237  rvs::hsa::print_hsa_status(__FILE__, __LINE__, __func__,
238  "hsa_amd_memory_async_copy()",
239  status);
240  break;
241  }
242  }
243 
244  // wait for transfer to complete
245  if (prop_h2d) {
246  RVSTRACE_
247  while (hsa_signal_wait_acquire(ctx_fwd.Sig, HSA_SIGNAL_CONDITION_LT,
248  1, uint64_t(-1), HSA_WAIT_STATE_ACTIVE)) {}
249  }
250 
251  // if bidirectional, also wait for reverse transfer to complete
252  if (prop_d2h) {
253  RVSTRACE_
254  while (hsa_signal_wait_acquire(ctx_rev.Sig, HSA_SIGNAL_CONDITION_LT,
255  1, uint64_t(-1), HSA_WAIT_STATE_ACTIVE)) {}
256  }
257 
258  RVSTRACE_
259  // get transfer duration
260  double duration = 0.0;
261  if (!prop_h2d && prop_d2h) {
262  duration = pHsa->GetCopyTime(bidirect,
263  ctx_rev.Sig, ctx_fwd.Sig)/1000000000;
264  } else {
265  duration = pHsa->GetCopyTime(bidirect,
266  ctx_fwd.Sig, ctx_rev.Sig)/1000000000;
267  }
268 
269  {
270  RVSTRACE_
271  std::lock_guard<std::mutex> lk(cntmutex);
272  running_size += b2b_block_size;
273  running_duration += duration;
274  }
275 
276  pebb_end_time = std::chrono::system_clock::now();
277 
278  uint64_t test_time = time_diff(pebb_end_time, pebb_start_time) ;
279 
280  if(test_time >= test_duration) {
281  break;
282  }
283  } // while(brun)
284 
285  RVSTRACE_
286  // deallocate buffers and signals
287  deinit();
288 }
