80 bool pqtaction::property_get_peers(int *error) {
81  *error = 0; // init with 'no error'
82  auto it = property.find("peers");
83  if (it != property.end()) {
84  if (it->second == "all") {
85  return true;
86  } else {
87  // split the list of gpu_id
88  prop_peers = str_split(it->second,
89  YAML_DEVICE_PROP_DELIMITER);
90  if (prop_peers.empty()) {
91  *error = 1; // list of gpu_id cannot be empty
92  } else {
93  for (vector<string>::iterator it_gpu_id =
94  prop_peers.begin();
95  it_gpu_id != prop_peers.end(); ++it_gpu_id)
96  if (!is_positive_integer(*it_gpu_id)) {
97  *error = 1;
98  break;
99  }
100  }
101  return false;
102  }
103 
104  } else {
105  *error = 1;
106  // when error is set, it doesn't really matter whether the method
107  // returns true or false
108  return false;
109  }
110 }

331 int pqtaction::create_threads() {
332  std::string msg;
333 
334  std::vector<uint16_t> gpu_id;
335  std::vector<uint16_t> gpu_device_id;
336  uint16_t transfer_ix = 0;
337  bool bmatch_found = false;
338 
339  gpu_get_all_gpu_id(&gpu_id);
340  gpu_get_all_device_id(&gpu_device_id);
341 
342  for (size_t i = 0; i < gpu_id.size(); i++) { // all possible sources
343  // filter out by source device id
344  if (prop_device_id_filtering) {
345  if (prop_deviceid != gpu_device_id[i]) {
346  continue;
347  }
348  }
349 
350  // filter out by listed sources
351  if (!prop_device_all_selected) {
352  const auto it = std::find(device_prop_gpu_id_list.cbegin(),
353  device_prop_gpu_id_list.cend(),
354  std::to_string(gpu_id[i]));
355  if (it == device_prop_gpu_id_list.cend()) {
356  continue;
357  }
358  }
359 
360  for (size_t j = 0; j < gpu_id.size(); j++) { // all possible peers
361  RVSTRACE_
362  // filter out by peer id
363  if (prop_peer_deviceid > 0) {
364  RVSTRACE_
365  if (prop_peer_deviceid != gpu_device_id[j]) {
366  RVSTRACE_
367  continue;
368  }
369  }
370 
371  RVSTRACE_
372  // filter out by listed peers
373  if (!prop_peer_device_all_selected) {
374  RVSTRACE_
375  const auto it = std::find(prop_peers.cbegin(),
376  prop_peers.cend(),
377  std::to_string(gpu_id[j]));
378  if (it == prop_peers.cend()) {
379  RVSTRACE_
380  continue;
381  }
382  }
383 
384  RVSTRACE_
385  // signal that at lease one matching src-dst combination
386  // has been found:
387  bmatch_found = true;
388 
389  // get NUMA nodes
390  int srcnode = rvs::gpulist::GetNodeIdFromGpuId(gpu_id[i]);
391  if (srcnode < 0) {
392  std::cerr << "RVS-PQT: no node found for GPU ID "
393  << std::to_string(gpu_id[i]);
394  return -1;
395  }
396 
397  int dstnode = rvs::gpulist::GetNodeIdFromGpuId(gpu_id[j]);
398  if (srcnode < 0) {
399  RVSTRACE_
400  std::cerr << "RVS-PQT: no node found for GPU ID "
401  << std::to_string(gpu_id[j]);
402  return -1;
403  }
404 
405  RVSTRACE_
406  uint32_t distance = 0;
407  std::vector<rvs::linkinfo_t> arr_linkinfo;
408  rvs::hsa::Get()->GetLinkInfo(srcnode, dstnode,
409  &distance, &arr_linkinfo);
410 
411  // perform peer check
412  if (is_peer(gpu_id[i], gpu_id[j])) {
413  RVSTRACE_
414  msg = "[" + action_name + "] p2p "
415  + std::to_string(gpu_id[i]) + " "
416  + std::to_string(gpu_id[j]) + " peers:true ";
417 
418  if (distance == rvs::hsa::NO_CONN) {
419  msg += "distance:-1";
420  } else {
421  msg += "distance:" + std::to_string(distance);
422  }
423  // iterate through individual hops
424  for (auto it = arr_linkinfo.begin(); it != arr_linkinfo.end(); it++) {
425  msg += " " + it->strtype + ":";
426  if (it->distance == rvs::hsa::NO_CONN) {
427  msg += "-1";
428  } else {
429  msg +=std::to_string(it->distance);
430  }
431  }
432  rvs::lp::Log(msg, rvs::logresults);
433 
434  if (bjson) {
435  RVSTRACE_
436  unsigned int sec;
437  unsigned int usec;
438  rvs::lp::get_ticks(&sec, &usec);
439  void* pjson = rvs::lp::LogRecordCreate(MODULE_NAME,
440  action_name.c_str(), rvs::logresults, sec, usec);
441  if (pjson != NULL) {
442  RVSTRACE_
443  rvs::lp::AddString(pjson, "src",
444  std::to_string(gpu_id[i]));
445  rvs::lp::AddString(pjson, "dst",
446  std::to_string(gpu_id[j]));
447  rvs::lp::AddString(pjson, "p2p", "true");
448  if (distance == rvs::hsa::NO_CONN) {
449  rvs::lp::AddInt(pjson, "distance", -1);
450  } else {
451  rvs::lp::AddInt(pjson, "distance", distance);
452  }
453 
454  void* phops = rvs::lp::CreateNode(pjson, "hops");
455  rvs::lp::AddNode(pjson, phops);
456 
457  // iterate through individual hops
458  for (uint i = 0; i < arr_linkinfo.size(); i++) {
459  char sbuff[64];
460  snprintf(sbuff, sizeof(sbuff), "hop%d", i);
461  void* phop = rvs::lp::CreateNode(phops, sbuff);
462  rvs::lp::AddString(phop, "type", arr_linkinfo[i].strtype);
463  if (arr_linkinfo[i].distance == rvs::hsa::NO_CONN) {
464  rvs::lp::AddInt(phop, "distance", -1);
465  } else {
466  rvs::lp::AddInt(phop, "distance", arr_linkinfo[i].distance);
467  }
468  rvs::lp::AddNode(phops, phop);
469  }
470 
471  rvs::lp::LogRecordFlush(pjson);
472  }
473  }
474 
475  RVSTRACE_
476  // GPUs are peers, create transaction for them
477  if (prop_test_bandwidth) {
478  RVSTRACE_
479  transfer_ix += 1;
480  pqtworker* p = new pqtworker;
481  p->initialize(srcnode, dstnode, prop_bidirectional);
482  p->set_name(action_name);
483  p->set_stop_name(action_name);
484  p->set_transfer_ix(transfer_ix);
485  p->set_block_sizes(block_size);
486  test_array.push_back(p);
487  }
488 
489  } else {
490  RVSTRACE_
491  msg = "[" + action_name + "] p2p "
492  + std::to_string(gpu_id[i]) + " "
493  + std::to_string(gpu_id[j]) + " peers:false ";
494 
495  if (distance == rvs::hsa::NO_CONN) {
496  msg += "distance:-1";
497  } else {
498  msg += "distance:" + std::to_string(distance);
499  }
500  // iterate through individual hops
501  for (auto it = arr_linkinfo.begin(); it != arr_linkinfo.end(); it++) {
502  msg += " " + it->strtype + ":";
503  if (it->distance == rvs::hsa::NO_CONN) {
504  msg += "-1";
505  } else {
506  msg +=std::to_string(it->distance);
507  }
508  }
509 
510  rvs::lp::Log(msg, rvs::logresults);
511 
512  if (bjson) {
513  RVSTRACE_
514  unsigned int sec;
515  unsigned int usec;
516  rvs::lp::get_ticks(&sec, &usec);
517  void* pjson = rvs::lp::LogRecordCreate(MODULE_NAME,
518  action_name.c_str(), rvs::logresults, sec, usec);
519  if (pjson != NULL) {
520  RVSTRACE_
521  rvs::lp::AddString(pjson,
522  "src", std::to_string(gpu_id[i]));
523  rvs::lp::AddString(pjson,
524  "dst", std::to_string(gpu_id[j]));
525  rvs::lp::AddString(pjson,
526  "p2p", "false");
527  if (distance == rvs::hsa::NO_CONN) {
528  rvs::lp::AddInt(pjson, "distance", -1);
529  } else {
530  rvs::lp::AddInt(pjson, "distance", distance);
531  }
532 
533  void* phops = rvs::lp::CreateNode(pjson, "hops");
534  rvs::lp::AddNode(pjson, phops);
535 
536  // iterate through individual hops
537  for (uint i = 0; i < arr_linkinfo.size(); i++) {
538  char sbuff[64];
539  snprintf(sbuff, sizeof(sbuff), "hop%d", i);
540  void* phop = rvs::lp::CreateNode(phops, sbuff);
541  rvs::lp::AddString(phop, "type", arr_linkinfo[i].strtype);
542  if (arr_linkinfo[i].distance == rvs::hsa::NO_CONN) {
543  rvs::lp::AddInt(phop, "distance", -1);
544  } else {
545  rvs::lp::AddInt(phop, "distance", arr_linkinfo[i].distance);
546  }
547  rvs::lp::AddNode(phops, phop);
548  }
549 
550  rvs::lp::LogRecordFlush(pjson);
551  }
552  }
553  }
554  }
555  }
556 
557  RVSTRACE_
558  if (prop_test_bandwidth && test_array.size() < 1) {
559  RVSTRACE_
560  std::string diag;
561  if (bmatch_found) {
562  RVSTRACE_
563  diag = "No peers found";
564  } else {
565  RVSTRACE_
566  diag = "No devices match criteria from the test configuation";
567  }
568  RVSTRACE_
569  msg = "[" + action_name + "] p2p-bandwidth " + diag;
570  rvs::lp::Log(msg, rvs::logerror);
571  if (bjson) {
572  RVSTRACE_
573  unsigned int sec;
574  unsigned int usec;
575  rvs::lp::get_ticks(&sec, &usec);
576  void* pjson = rvs::lp::LogRecordCreate("p2p-bandwidth",
577  action_name.c_str(), rvs::logerror, sec, usec);
578  if (pjson != NULL) {
579  RVSTRACE_
580  rvs::lp::AddString(pjson,
581  "message",
582  diag);
583  rvs::lp::LogRecordFlush(pjson);
584  }
585  }
586  RVSTRACE_
587  return 0;
588  }
589 
590  RVSTRACE_
591  for (auto it = test_array.begin(); it != test_array.end(); ++it) {
592  RVSTRACE_
593  (*it)->set_transfer_num(test_array.size());
594  }
595 
596  RVSTRACE_
597  return 0;
598 }

622 int pqtaction::run() {
623  int sts;
624  string msg;
625 
626  rvs::lp::Log("int pqtaction::run()", rvs::logtrace);
627 
628  if (property.find("cli.-j") != property.end()) {
629  bjson = true;
630  }
631 
632  if (!get_all_common_config_keys()) {
633  cerr << "RVS-PQT: " << "Error in get_all_common_config_keys()"<< std::endl;
634  return -1;
635  }
636  if (!get_all_pqt_config_keys()) {
637  cerr << "RVS-PQT: " << "Error in get_all_pqt_config_keys()"<< std::endl;
638  return -1;
639  }
640 
641  // log_interval must be less than duration
642  if (prop_log_interval > 0 && gst_run_duration_ms > 0) {
643  if (static_cast<uint64_t>(prop_log_interval) > gst_run_duration_ms) {
644  cerr << "RVS-PQT: action: " << action_name <<
645  " log_interval must be less than duration" << std::endl;
646  return -1;
647  }
648  }
649 
650  sts = create_threads();
651  if (sts)
652  return sts;
653 
654  if (prop_test_bandwidth && test_array.size() > 0) {
655  if (gst_runs_parallel) {
656  sts = run_parallel();
657  } else {
658  sts = run_single();
659  }
660  }
661 
662  // do cleanup
663  destroy_threads();
664 
665  return sts;
666 }

677 int pqtaction::is_peer(uint16_t Src, uint16_t Dst) {
679  rvs::hsa* pHsa;
680 
681  if (Src == Dst) {
682  return 0;
683  }
684  pHsa = rvs::hsa::Get();
685 
686  // GPUs are peers, create transaction for them
687  int srcnode = rvs::gpulist::GetNodeIdFromGpuId(Src);
688  if (srcnode < 0) {
689  std::cerr << "RVS-PQT: no node found for GPU ID "
690  << std::to_string(Src);
691  return 0;
692  }
693 
694  int dstnode = rvs::gpulist::GetNodeIdFromGpuId(Dst);
695  if (srcnode < 0) {
696  std::cerr << "RVS-PQT: no node found for GPU ID "
697  << std::to_string(Dst);
698  return 0;
699  }
700 
701  return pHsa->rvs::hsa::GetPeerStatus(srcnode, dstnode);
702 }

711 int pqtaction::run_single() {
712  RVSTRACE_
713  // define timers
714  rvs::timer<pqtaction> timer_running(&pqtaction::do_running_average, this);
715  rvs::timer<pqtaction> timer_final(&pqtaction::do_final_average, this);
716 
717  unsigned int iter = gst_run_count > 0 ? gst_run_count : 1;
718  unsigned int step = gst_run_count == 0 ? 0 : 1;
719 
720  // let the test run
721  brun = true;
722 
723  // start timers
724  if (gst_run_duration_ms) {
725  RVSTRACE_
726  timer_final.start(gst_run_duration_ms, true); // ticks only once
727  }
728 
729  if (prop_log_interval) {
730  RVSTRACE_
731  timer_running.start(prop_log_interval); // ticks continuously
732  }
733 
734  // iterate through test array and invoke tests one by one
735  RVSTRACE_
736  do {
737  RVSTRACE_
738  for (auto it = test_array.begin(); brun && it != test_array.end(); ++it) {
739  RVSTRACE_
740  (*it)->do_transfer();
741 
742  // if log interval is zero, print current results immediately
743  if (prop_log_interval == 0) {
744  print_running_average(*it);
745  }
746  sleep(1);
747 
748  if (rvs::lp::Stopping()) {
749  RVSTRACE_
750  brun = false;
751  break;
752  }
753  }
754  RVSTRACE_
755 
756  iter -= step;
757
