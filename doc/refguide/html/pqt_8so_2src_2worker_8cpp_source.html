137 int pqtworker::do_transfer() {
138  double duration;
139  int sts;
140  unsigned int startsec;
141  unsigned int startusec;
142  unsigned int endsec;
143  unsigned int endusec;
144  std::string msg;
145 
146  msg = "[" + action_name + "] pqt transfer " + std::to_string(src_node) + " "
147  + std::to_string(dst_node) + " ";
148 
149  rvs::lp::get_ticks(&startsec, &startusec);
150 
151  if (block_size.size() == 0) {
152  block_size = pHsa->size_list;
153  }
154 
155  for (size_t i = 0; brun && i < block_size.size(); i++) {
156  current_size = block_size[i];
157  sts = pHsa->SendTraffic(src_node, dst_node, current_size,
158  bidirect, &duration);
159 
160  if (sts) {
161  std::cerr << "RVS-PQT: internal error, src: " << src_node
162  << " dst: " << dst_node
163  << " current size: " << current_size << std::endl;
164  return sts;
165  }
166 
167  {
168  std::lock_guard<std::mutex> lk(cntmutex);
169  running_size += current_size;
170  running_duration += duration;
171  }
172  }
173 
174  rvs::lp::get_ticks(&endsec, &endusec);
175  rvs::lp::Log(msg + "start", rvs::logdebug, startsec, startusec);
176  rvs::lp::Log(msg + "finish", rvs::logdebug, endsec, endusec);
177 
178  return 0;
179 }
