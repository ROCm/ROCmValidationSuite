69 void pbqtworker::run() {
70  std::string msg;
71  std::chrono::time_point<std::chrono::system_clock> pbqt_start_time;
72  std::chrono::time_point<std::chrono::system_clock> pbqt_end_time;
73 
74  msg = "[" + action_name + "] pbqt thread " + std::to_string(src_node) + " "
75  + std::to_string(dst_node) + " has started";
76  rvs::lp::Log(msg, rvs::logdebug);
77 
78  brun = true;
79 
80  pbqt_start_time = std::chrono::system_clock::now();
81  do {
82  do_transfer();
83 
84  pbqt_end_time = std::chrono::system_clock::now();
85 
86  uint64_t test_time = time_diff(pbqt_end_time, pbqt_start_time) ;
87 
88  if(test_time >= test_duration) {
89  break;
90  }
91  } while (brun);
92 
93  msg = "[" + action_name + "] pbqt thread " + std::to_string(src_node) + " "
94  + std::to_string(dst_node) + " has finished";
95  rvs::lp::Log(msg, rvs::logdebug);

148 int pbqtworker::do_transfer() {
149  double duration;
150  int sts;
151  unsigned int startsec;
152  unsigned int startusec;
153  unsigned int endsec;
154  unsigned int endusec;
155  std::string msg;
156 
157  msg = "[" + action_name + "] pbqt transfer " + std::to_string(src_node) + " "
158  + std::to_string(dst_node) + " ";
159 
160  rvs::lp::get_ticks(&startsec, &startusec);
161 
162  if (block_size.size() == 0) {
163  block_size = pHsa->size_list;
164  }
165  for (size_t i = 0; brun && i < block_size.size(); i++) {
166  current_size = block_size[i];
167  sts = pHsa->SendTraffic(src_node, dst_node, current_size,
168  bidirect, &duration);
169 
170  if (sts) {
171  msg = "internal error, src: " + std::to_string(src_node)
172  + " dst: " + std::to_string(dst_node)
173  + " current size: " + std::to_string(current_size);
174  rvs::lp::Err(msg, MODULE_NAME, action_name);
175  return sts;
176  }
177 
178  {
179  std::lock_guard<std::mutex> lk(cntmutex);
180  running_size += current_size;
181  running_duration += duration;
182  }
183  }
184 
185  rvs::lp::get_ticks(&endsec, &endusec);
186  rvs::lp::Log(msg + "start", rvs::logdebug, startsec, startusec);
187  rvs::lp::Log(msg + "finish", rvs::logdebug, endsec, endusec);
188 
189  return 0;
190 }
