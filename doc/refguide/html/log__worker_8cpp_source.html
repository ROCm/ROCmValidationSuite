132 void log_worker::run() {
133  std::chrono::time_point<std::chrono::system_clock> start_time, end_time;
134  float cur_power_value, avg_power = 0;
135  uint64_t power_sampling_iters = 0, cur_milis;
136  string msg;
137 
138  {
139  std::lock_guard<std::mutex> lck(mtx_brun);
140  brun = true;
141  }
142 
143  start_time = std::chrono::system_clock::now();
144  for (;;) {
145  // check if stop signal was received
146  if (rvs::lp::Stopping())
147  break;
148 
149  {
150  std::lock_guard<std::mutex> lck(mtx_brun);
151  if (!brun)
152  break;
153  }
154 
155  {
156  std::lock_guard<std::mutex> lck(mtx_bpaused);
157  if (bpaused)
158  continue;
159  }
160 
161  // get GPU's current average power
162  cur_power_value = get_power_data(gpu_hwmon_entry);
163  if (cur_power_value != 0) {
164  avg_power += cur_power_value;
165  power_sampling_iters++;
166  }
167 
168  end_time = std::chrono::system_clock::now();
169  cur_milis = time_diff(end_time, start_time);
170  if (cur_milis >= log_interval) {
171  if (power_sampling_iters != 0) {
172  avg_power /= power_sampling_iters;
173  msg = "[" + action_name + "] " + MODULE_NAME + " " +
174  std::to_string(gpu_id) + " " +
175  IET_LOGGER_CURRENT_POWER_MSG + " " +
176  std::to_string(avg_power);
177  log(msg.c_str(), rvs::loginfo);
178  log_to_json(IET_LOGGER_CURRENT_POWER_MSG,
179  std::to_string(avg_power), rvs::loginfo);
180  }
181 
182  avg_power = 0;
183  power_sampling_iters = 0;
184  start_time = std::chrono::system_clock::now();
185  } else {
186  usleep(POWER_PROCESS_DELAY);
187  }
188  }
189 }
