85 int action::run(void) {
86  map<string, string>::iterator it; // module's properties map iterator
87 
88  string msg;
89  int sample_interval = 1000, log_interval = 0;
90  int error = 0;
91  bool metric_true, metric_bound;
92  int metric_min, metric_max;
93  bool terminate = false;
94  uint64_t duration = 1;
95  std::vector<uint16_t> gpu_id;
96 
97  gpu_get_all_gpu_id(&gpu_id);
98 
99  if (rvs::actionbase::has_property("sample_interval")) {
100  sample_interval =
101  rvs::actionbase::property_get_sample_interval(&error);
102  }
103 
104  if (rvs::actionbase::has_property("log_interval")) {
105  log_interval = rvs::actionbase::property_get_log_interval(&error);
106  if ( log_interval < sample_interval ) {
107  msg = "Log interval has the lower value than the sample interval ";
108  cerr << "RVS-GM: action: " << property["name"] << msg << endl;
109  return -1;
110  }
111  }
112 
113  if (rvs::actionbase::has_property("terminate")) {
114  terminate = rvs::actionbase::property_get_terminate(&error);
115  }
116 
117  // start of monitoring?
118  if (property["monitor"] == "true") {
119  if (pworker) {
120  rvs::lp::Log("[" + property["name"]+ "] gm monitoring already started",
121  rvs::logresults);
122  return 0;
123  }
124 
125  pworker = new Worker();
126  pworker->set_name(property["name"]);
127  pworker->set_sample_int(sample_interval);
128  pworker->set_log_int(log_interval);
129  pworker->set_terminate(terminate);
130  if (property["force"] == "true")
131  pworker->set_force(true);
132 
133  if (rvs::actionbase::has_property("duration")) {
134  rvs::actionbase::property_get_run_duration(&error);
135  duration = rvs::actionbase::gst_run_duration_ms;
136  }
137 
138  for (it = property.begin(); it != property.end(); ++it) {
139  metric_bound = false;
140  string word;
141  string s = it->first;
142  if (s.find(".") != std::string::npos && s.substr(0, s.find(".")) ==
143  "metrics") {
144  string metric = s.substr(s.find(".")+1);
145  s = it->second;
146  vector<string> values = str_split(s, YAML_DEVICE_PROP_DELIMITER);
147 
148  if (values.size() == 3) {
149  metric_true = (values[0] == "true") ? true : false;
150  metric_max = std::stoi(values[1]);
151  metric_min = std::stoi(values[2]);
152  metric_bound = true;
153  } else {
154  msg = " Wrong number of metric parameters ";
155  cerr << "RVS-GM: action: " << property["name"] << msg << endl;
156  return -1;
157  }
158 
159  pworker->set_metr_mon(metric, metric_true);
160  pworker->set_bound(metric, metric_bound, metric_max, metric_min);
161  values.clear();
162  }
163  }
164 
165  // check if -j flag is passed
166  if (has_property("cli.-j")) {
167  bjson = true;
168  pworker->json(true);
169  }
170 
171  // checki if deviceid filtering is required
172  string sdevid;
173  if (has_property("deviceid", &sdevid)) {
174  if (::is_positive_integer(sdevid)) {
175  try {
176  pworker->set_deviceid(std::stoi(sdevid));
177  }
178  catch(...) {
179  cerr << "RVS-GM: action: " << property["name"] <<
180  " invalide 'deviceid' key value: " << sdevid << std::endl;
181  return -1;
182  }
183  } else {
184  cerr << "RVS-GM: action: " << property["name"] <<
185  " invalide 'deviceid' key value: " << sdevid << std::endl;
186  return -1;
187  }
188  }
189 
190  // check if GPU id filtering is requied
191  string sdev;
192  if (has_property("device", &sdev)) {
193  vector<uint16_t> iarr;
194  if (sdev != "all") {
195  vector<string> sarr = str_split(sdev, YAML_DEVICE_PROP_DELIMITER);
196  int sts = rvs_util_strarr_to_uintarr(sarr, &iarr);
197  if (sts < 0) {
198  cerr << "RVS-GM: action: " << property["name"] <<
199  " invalide 'device' key value: " << sdev << std::endl;
200  return -1;
201  }
202  pworker->set_gpuids(iarr);
203  } else {
204  pworker->set_gpuids(gpu_id);
205  }
206  } else {
207  cerr << "RVS-GM: action: " << property["name"] <<
208  " key 'device' not found" << std::endl;
209  return -1;
210  }
211  // set stop name before start
212  pworker->set_stop_name(property["name"]);
213  // start worker thread
214  rvs::lp::Log("[" + property["name"]+ "] gm starting Worker",
215  rvs::logtrace);
216  pworker->start();
217  sleep(duration);
218 
219  rvs::lp::Log("[" + property["name"]+ "] gm Monitoring started",
220  rvs::logtrace);
221  } else {
222  rvs::lp::Log("[" + property["name"]+
223  "] gm property[\"monitor\"] != \"true\"", rvs::logtrace);
224  if (pworker) {
225  // (give thread chance to start)
226  sleep(2);
227  pworker->set_stop_name(property["name"]);
228  pworker->stop();
229  delete pworker;
230  pworker = nullptr;
231  }
232  }
233 
234  if (bjson && json_root_node != NULL) { // json logging stuff
235  rvs::lp::LogRecordFlush(json_root_node);
236  }
237 
238  return 0;
239 }
