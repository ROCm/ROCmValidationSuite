86 bool gm_action::get_all_common_config_keys(void) {
87  string msg;
88  int error;
89 
90  bool sts = true;
91  // check if -j flag is passed
92  if (has_property("cli.-j")) {
93  bjson = true;
94  }
95 
96  if (property_get(RVS_CONF_NAME_KEY, &action_name)) {
97  rvs::lp::Err("Action name missing", MODULE_NAME_CAPS);
98  return false;
99  }
100 
101  // get <device> property value (a list of gpu id)
102  if (int ists = property_get_device()) {
103  switch (ists) {
104  case 1:
105  msg = "Invalid 'device' key value.";
106  break;
107  case 2:
108  msg = "Missing 'device' key.";
109  break;
110  }
111  rvs::lp::Err(msg, MODULE_NAME_CAPS, action_name);
112  sts = false;
113  }
114 
115  // get the <deviceid> property value if provided
116  if (property_get_int<uint16_t>(RVS_CONF_DEVICEID_KEY,
117  &property_device_id, 0u)) {
118  msg = "Invalid 'deviceid' key value.";
119  rvs::lp::Err(msg, MODULE_NAME_CAPS, action_name);
120  sts = false;
121  }
122 
123  if (property_get_int<uint64_t>(RVS_CONF_DURATION_KEY,
124  &property_duration, 0u)) {
125  msg = "Invalid '" + std::string(RVS_CONF_DURATION_KEY) + "' key.";
126  rvs::lp::Err(msg, MODULE_NAME_CAPS, action_name);
127  sts = false;
128  }
129 
130  error = property_get_int<uint64_t>
131  (RVS_CONF_LOG_INTERVAL_KEY, &property_log_interval, DEFAULT_LOG_INTERVAL);
132  if (error == 1) {
133  msg = "Invalid '" +std::string(RVS_CONF_LOG_INTERVAL_KEY) + "' key.";
134  rvs::lp::Err(msg, MODULE_NAME_CAPS, action_name);
135  sts = false;
136  }
137 
138  if (property_get_int<uint64_t>(RVS_CONF_SAMPLE_INTERVAL_KEY,
139  &sample_interval, 500u)) {
140  msg = "Invalid '" +std::string(RVS_CONF_SAMPLE_INTERVAL_KEY) + "' key.";
141  rvs::lp::Err(msg, MODULE_NAME_CAPS, action_name);
142  sts = false;
143  }
144 
145  if (property_get(RVS_CONF_TERMINATE_KEY, &prop_terminate, false)) {
146  msg = "Invalid 'terminate' key.";
147  rvs::lp::Err(msg, MODULE_NAME_CAPS, action_name);
148  sts = false;
149  }
150 
151  if (property_get(GM_FORCE, &prop_force, false)) {
152  msg = "Invalid '" + std::string(GM_FORCE) + "' key.";
153  rvs::lp::Err(msg, MODULE_NAME_CAPS, action_name);
154  sts = false;
155  }
156 
157  if (property_log_interval < sample_interval) {
158  msg = "Log interval has the lower value than the sample interval.";
159  rvs::lp::Err(msg, MODULE_NAME_CAPS, action_name);
160  sts = false;
161  }
162 
163  return sts;
164 }

173 int gm_action::get_bounds(const char* pMetric) {
174  std::string smetric("metrics.");
175  smetric += pMetric;
176 
177  std::string sval;
178  if (!has_property(smetric, &sval)) {
179  return 2;
180  }
181 
182  Worker::Metric_bound bound_;
183  int error;
184  std::vector<string> values = str_split(sval, YAML_DEVICE_PROP_DELIMITER);
185  if (values.size() == 3) {
186  bound_.mon_metric = true;
187  bound_.check_bounds = (values[0] == "true") ? true : false;
188  error = rvs_util_parse<uint32_t>(values[1], &bound_.max_val);
189  if (error) {
190  return 1;
191  }
192  error = rvs_util_parse<uint32_t>(values[2], &bound_.min_val);
193  if (error) {
194  return 1;
195  }
196  property_bounds[std::string(pMetric)] = bound_;
197  } else {
198  return 1;
199  }
200 
201  return 0;
202 }

258 int gm_action::run(void) {
259  string msg;
260  rsmi_status_t status;
261 
262  // if monitoring is already running, stop it
263  // (it will be restarted if needed)
264  RVSTRACE_
265  if (pworker) {
266  RVSTRACE_
267  // (give thread chance to start)
268  sleep(2);
269  pworker->set_stop_name(property["name"]);
270  pworker->stop();
271  delete pworker;
272  pworker = nullptr;
273  }
274  // this action should stop monitoring?
275  if (property["monitor"] != "true") {
276  RVSTRACE_
277  // already done, just return
278  return 0;
279  }
280 
281  RVSTRACE_
282  // start new monitoring
283  if (!get_all_common_config_keys()) {
284  RVSTRACE_
285  return -1;
286  }
287 
288  if (!get_all_gm_config_keys()) {
289  RVSTRACE_
290  return -1;
291  }
292 
293  RVSTRACE_
294 
295  // if 'device: all' get all AMD GPU IDs
296  if (property_device_all) {
297  gpu_get_all_gpu_id(&property_device);
298  }
299 
300  // apply device_id filtering if needed
301  if (property_device_id > 0) {
302  RVSTRACE_
303  std::vector<uint16_t> gpu_id_filtered;
304  for (auto it = property_device.begin(); it != property_device.end(); it++) {
305  RVSTRACE_
306 
307  uint16_t _dev_id;
308  if (rvs::gpulist::gpu2device(*it, &_dev_id)) {
309  RVSTRACE_
310  // if not found just continue
311  continue;
312  }
313 
314  if (_dev_id == property_device_id) {
315  RVSTRACE_
316  gpu_id_filtered.push_back(*it);
317  }
318  }
319  property_device = gpu_id_filtered;
320  }
321 
322  RVSTRACE_
323 
324  // verify that the resulting array is not empty
325  if (property_device.size() < 1) {
326  rvs::lp::Err("No devices match filtering criteria.",
327  MODULE_NAME_CAPS, action_name);
328  return -1;
329  }
330 
331  // convert GPU ID into rocm_smi_lib device index
332  std::map<uint32_t, int32_t> dv_ind;
333  for (auto it = property_device.begin(); it != property_device.end(); it++) {
334  RVSTRACE_
335  uint16_t location_id;
336  if (rvs::gpulist::gpu2location(*it, &location_id)) {
337  msg = "Could not obtain BDF for GPU ID: ";
338  msg += std::to_string(*it);
339  rvs::lp::Err(msg, MODULE_NAME_CAPS, action_name);
340  return -1;
341  }
342  uint32_t ix;
343  status = rvs::rsmi_dev_ind_get(location_id, &ix);
344  if(status == RSMI_STATUS_SUCCESS) {
345  dv_ind.insert(std::pair<uint32_t, int32_t>(ix, *it));
346  }
347  }
348 
349  pworker = new Worker();
350  pworker->set_name(action_name);
351  pworker->json(bjson);
352  pworker->set_sample_int(sample_interval);
353  pworker->set_log_int(property_log_interval);
354  pworker->set_terminate(prop_terminate);
355  if (prop_force)
356  pworker->set_force(true);
357 
358  // set stop name before start
359  pworker->set_stop_name(action_name);
360  // set array of device indices to monitor
361  pworker->set_dv_ind(dv_ind);
362  // set bounds map
363  pworker->set_bound(property_bounds);
364 
365  RVSTRACE_
366  // start worker thread
367  pworker->start();
368 
369  // this should be used only for testing purposes
370  if (property_duration) {
371  RVSTRACE_
372  sleep(property_duration);
373  }
374 
375  RVSTRACE_
376 
377  return 0;
378 }
