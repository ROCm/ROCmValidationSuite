97 void rvs::cli::extract_path() {
98  char path[PATH_MAX];
99  char dest[PATH_MAX];
100  memset(dest, 0, sizeof(dest)); // readlink does not null terminate!
101 
102  pid_t pid = getpid();
103  snprintf(path, sizeof(path), "/proc/%d/exe", pid);
104  if (readlink(path, dest, PATH_MAX) == -1) {
105  std::cerr << "RVS-CLI: could not extract path to executable\n";
106  return;
107  }
108 
109  std::string argv0(dest);
110 
111  size_t found = argv0.find_last_of("/\\");
112  options::opt["pwd"] = argv0.substr(0, found) + "/";
113 }

202 int rvs::cli::parse(int Argc, char** Argv) {
203  init_grammar();
204 
205  extract_path();
206 
207  argc = Argc;
208  argv = Argv;
209  context.push(econtext::eof);
210  context.push(econtext::command);
211 
212  for (;;) {
213  std::string token = get_token();
214  bool token_done = false;
215  while (!token_done) {
216  econtext top = context.top();
217  context.pop();
218 
219  switch (top) {
220  case econtext::command:
221  token_done = try_command(token);
222  break;
223 
224  case econtext::value:
225  token_done = try_value(token);
226  if (!token_done) {
227  errstr = std::string("syntax error: value expected after ") +
228  current_option;
229  return -1;
230  }
231  break;
232 
233  case econtext::eof:
234  if (token == "") {
235  emit_option();
236  return 0;
237  } else {
238  errstr = "unexpected command line argument: " +
239  token;
240  return -1;
241  }
242 
243  default:
244  errstr = "syntax error: " + token;
245  return -1;
246  }
247  }
248  }
249 
250  return -1;
251 }

322 bool rvs::cli::try_command(const std::string& token) {
323  auto it = grammar.find(token);
324  if (it == grammar.end())
325  return false;
326 
327  // emit previous buffer contents (if any)
328  emit_option();
329 
330  // token identified as command, so store it:
331  current_option = it->second->name;
332 
333  // fill context std::stack with new possible continuations:
334  it->second->adjust_context(&context);
335 
336  return true;
337 }

349 bool rvs::cli::try_value(const std::string& token) {
350  if (token == "")
351  return false;
352 
353  // should not be one of command line options
354  auto it = grammar.find(token);
355  if (it != grammar.end())
356  return false;
357 
358  // token is value for previous command
359  current_value = token;
360 
361  // emit previous option-value pair:
362  emit_option();
363 
364  return true;
365 }
