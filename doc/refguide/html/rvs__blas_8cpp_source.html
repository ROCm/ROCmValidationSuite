49 rvs_blas::rvs_blas(int _gpu_device_index, int _m, int _n, int _k, int transA, int transB,
50  float alpha , float beta, rocblas_int lda, rocblas_int ldb, rocblas_int ldc, std::string _ops_type)
51  : gpu_device_index(_gpu_device_index)
52  , ops_type(_ops_type)
53  , m(_m), n(_n), k(_k)
54  , size_a(0), size_b(0), size_c(0), size_d(0)
55  , da(nullptr), db(nullptr), dc(nullptr)
56  , ha(nullptr), hb(nullptr), hc(nullptr)
57  , ddbla(nullptr), ddblb(nullptr), ddblc(nullptr)
58  , hdbla(nullptr), hdblb(nullptr), hdblc(nullptr)
59  , dhlfa(nullptr), dhlfb(nullptr), dhlfc(nullptr), dhlfd(nullptr)
60  , hhlfa(nullptr), hhlfb(nullptr), hhlfc(nullptr)
61  , hip_stream(nullptr)
62  , blas_handle(nullptr)
63  , is_handle_init(false)
64  , is_error(false)
65  {
66 
67  if(transA == 0) {
68  transa = rocblas_operation_none;
69  }else{
70  transa = rocblas_operation_transpose;
71  }
72 
73  if(transB == 0) {
74  transb = rocblas_operation_none;
75  }else{
76  transb = rocblas_operation_transpose;
77  }
78 
79  if(ops_type == "hgemm") {
80  //auto A_row = transA == rocblas_operation_none ? m : k;
81  auto A_col = transA == rocblas_operation_none ? k : m;
82  //auto B_row = transB == rocblas_operation_none ? k : n;
83  auto B_col = transB == rocblas_operation_none ? n : k;
84 
85  size_a = size_t(lda) * A_col;
86  size_b = size_t(ldb) * B_col;
87  size_c = size_t(ldc) * n;
88  size_d = size_t(ldc) * n;
89  }else{
90  size_a = size_t(k) * m;
91  size_b = size_t(k) * n;
92  size_c = size_t(n) * m;
93  }
94 
95  //setting alpha and beta val
96  blas_alpha_val = alpha;
97  blas_beta_val = beta;
98 
99  //setting lda offsets
100  //Leading data offsets
101  if(lda == 0 || ldb == 0 || ldc == 0) {
102  blas_lda_offset = m;
103  blas_ldb_offset = n;
104  blas_ldc_offset = k;
105  }else{
106  blas_lda_offset = lda;
107  blas_ldb_offset = ldb;
108  blas_ldc_offset = ldc;
109  }
110 
111  if (alocate_host_matrix_mem()) {
112  if (!init_gpu_device())
113  is_error = true;
114  } else {
115  is_error = true;
116  }
117 }

133  // select GPU device & allocate memory
134  if (hipSetDevice(gpu_device_index) != hipSuccess) {
135  // cannot select the given GPU device
136  return false;
137  } else {
138  //ROCBLAS Initialize

141  if (!allocate_gpu_matrix_mem())
142  return false;

144  is_handle_init = true;
145  if (rocblas_get_stream(blas_handle, &hip_stream)
146  != rocblas_status_success)
147  return false;
148  } else {
149  return false;
150  }
151  }
152  return true;

535 float rvs_blas::fast_pseudo_rand(uint64_t *nextr) {
536  *nextr = *nextr * 1103515245 + 12345;
537  return static_cast<float>(static_cast<uint32_t>
538  ((*nextr / 65536) % RANDOM_CT)) / RANDOM_DIV_CT;
