77 bool rvs_blas::init_gpu_device(void) {
78  // select GPU device & allocate memory
79  if (hipSetDevice(gpu_device_index) != hipSuccess) {
80  // cannot select the given GPU device
81  return false;
82  } else {
83  if (!allocate_gpu_matrix_mem())
84  return false;
85  if (rocblas_create_handle(&blas_handle) == rocblas_status_success) {
86  is_handle_init = true;
87  if (rocblas_get_stream(blas_handle, &hip_stream)
88  != rocblas_status_success)
89  return false;
90  } else {
91  return false;
92  }
93  }
94  return true;
95 }

137 bool rvs_blas::allocate_gpu_matrix_mem(void) {
