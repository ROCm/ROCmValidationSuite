125 static string get_hwmon_entry(const std::string &path) {
126  DIR *dirp;
127  struct dirent *dir;
128  dirp = opendir(path.c_str());
129  if (dirp) {
130  while ((dir = readdir(dirp)) != 0) {
131  if ((strcmp(dir->d_name, ".") == 0) ||
132  (strcmp(dir->d_name, "..") == 0))
133  continue;
134  if (strstr(dir->d_name, HWMON_FOLDER_BASE_NAME) != NULL) {
135  closedir(dirp);
136  return path + "/" + dir->d_name + "/" + GPU_POWER_DATA_FILE;
137  }
138  }
139  closedir(dirp);
140  return "";
141  } else {
142  return "";

448 bool action::do_edp_test(void) {
449  size_t k = 0;
450  for (;;) {
451  unsigned int i = 0;
452  if (gst_run_wait_ms != 0) // delay iet execution
453  sleep(gst_run_wait_ms);
454 
455  vector<IETWorker> workers(edpp_gpus.size());
456  vector<gpu_hwmon_info>::iterator it;
457 
458  // all worker instances have the same json settings
459  IETWorker::set_use_json(bjson);
460 
461  for (it = edpp_gpus.begin(); it != edpp_gpus.end(); ++it) {
462  // set worker thread params
463  workers[i].set_name(action_name);
464  workers[i].set_gpu_id((*it).gpu_id);
465  workers[i].set_gpu_device_index((*it).hip_gpu_deviceid);
466  workers[i].set_gpu_hwmon_entry((*it).gpu_hwmon_power_entry);
467  workers[i].set_run_wait_ms(gst_run_wait_ms);
468  workers[i].set_run_duration_ms(gst_run_duration_ms);
469  workers[i].set_ramp_interval(iet_ramp_interval);
470  workers[i].set_log_interval(iet_log_interval);
471  workers[i].set_sample_interval(iet_sample_interval);
472  workers[i].set_max_violations(iet_max_violations);
473  workers[i].set_target_power(iet_target_power);
474  workers[i].set_tolerance(iet_tolerance);
475  workers[i].set_matrix_size(iet_matrix_size);
476  i++;
477  }
478 
479  if (gst_runs_parallel) {
480  for (i = 0; i < edpp_gpus.size(); i++)
481  workers[i].start();
482 
483  // join threads
484  for (i = 0; i < edpp_gpus.size(); i++)
485  workers[i].join();
486  } else {
487  for (i = 0; i < edpp_gpus.size(); i++) {
488  workers[i].start();
489  workers[i].join();
490 
491  // check if stop signal was received
492  if (rvs::lp::Stopping())
493  return false;
494  }
495  }
496 
497  // check if stop signal was received
498  if (rvs::lp::Stopping())
499  return false;
500 
501  if (gst_run_count != 0) {
502  k++;
503  if (k == gst_run_count)
504  break;
505  }
506  }
507  return rvs::lp::Stopping() ? false : true;
508 }

575 bool action::add_gpu_to_edpp_list(uint16_t dev_location_id, uint16_t gpu_irq,
576  int32_t gpu_id, int hip_num_gpu_devices) {
577  for (int i = 0; i < hip_num_gpu_devices; i++) {
578  // get GPU device properties
579  hipDeviceProp_t props;
580  hipGetDeviceProperties(&props, i);
581 
582  // compute device location_id (needed to match this device
583  // with one of those found while querying the pci bus
584  uint16_t hip_dev_location_id =
585  ((((uint16_t) (props.pciBusID)) << 8) | (props.pciDeviceID));
586 
587  if (hip_dev_location_id == dev_location_id) {
588  // now try finding the hwmon entry which is
589  // needed for the power related data
590  for (auto cgpu_dev : monitor_devices) {
591  // get irq of device
592  string irq = get_irq(cgpu_dev ->path());
593  if (irq != "" && gpu_irq == std::stoi(irq)) {
594  // found the device
595  string base_folder_hwmon = cgpu_dev->path() +
596  "/" + SMI_DEVICE_FOLDER_BASE_NAME +
597  "/" + HWMON_FOLDER_BASE_NAME;
598  string cgpu_hwmon_power_entry =
599  get_hwmon_entry(base_folder_hwmon);
600  if (cgpu_hwmon_power_entry != "") {
601  gpu_hwmon_info cgpu_info;
602  cgpu_info.hip_gpu_deviceid = i;
603  cgpu_info.gpu_id = gpu_id;
604  cgpu_info.gpu_hwmon_power_entry =
605  cgpu_hwmon_power_entry;
606  edpp_gpus.push_back(cgpu_info);
607 
608  return true;
609  }
610  break;
611  }
612  }
613  }
614  }
615 
616  return false;
617 }

624 int action::get_all_selected_gpus(void) {
625  string msg;
626  bool amd_gpus_found = false;
627  int hip_num_gpu_devices;
628  struct pci_access *pacc;
629  struct pci_dev *pci_cdev;
630  amd::smi::RocmSMI hw;
631 
632  hip_num_gpu_devices = get_num_amd_gpu_devices();
633  if (hip_num_gpu_devices == 0)
634  return 0; // no AMD compatible GPU found!
635 
636  // get GPU devices via smi lib
637  hw.DiscoverDevices();
638  hw.IterateSMIDevices(smi_get_gpu_devices_list,
639  reinterpret_cast<void *>(&monitor_devices));
640 
641  // get the pci_access structure
642  pacc = pci_alloc();
643 
644  if (pacc == NULL) {
645  // log the error
646  msg = action_name + " " + MODULE_NAME + " " + PCI_ALLOC_ERROR;
647  cerr << "RVS-IET: " << msg;
648 
649  return -1; // EDPp test cannot continue
650  }
651 
652  // initialize the PCI library
653  pci_init(pacc);
654  // get the list of devices
655  pci_scan_bus(pacc);
656 
657  // iterate over devices
658  for (pci_cdev = pacc->devices; pci_cdev; pci_cdev = pci_cdev->next) {
659  // fill in the info
660  pci_fill_info(pci_cdev,
661  PCI_FILL_IDENT | PCI_FILL_BASES | PCI_FILL_CLASS);
662 
663  // computes the actual dev's location_id (sysfs entry)
664  uint16_t dev_location_id = ((((uint16_t) (pci_cdev->bus)) << 8)
665  | (pci_cdev->func));
666 
667  // check if this pci_dev corresponds to one of the AMD GPUs
668  int32_t gpu_id = rvs::gpulist::GetGpuId(dev_location_id);
669  if (gpu_id == -1)
670  continue;
671 
672  // that should be an AMD GPU
673  // check for deviceid filtering
674  if (!device_id_filtering || (device_id_filtering &&
675  pci_cdev->device_id == deviceid)) {
676  // check if the GPU is part of the EDPp test (either <device>: all
677  // or the gpu_id is in the device: <gpu id> list)
678  bool cur_gpu_selected = false;
679 
680  if (device_all_selected) {
681  cur_gpu_selected = true;
682  } else {
683  // search for this gpu in the list
684  // provided under the <device> property
685  vector<string>::iterator it_gpu_id = find(
686  device_prop_gpu_id_list.begin(),
687  device_prop_gpu_id_list.end(),
688  std::to_string(gpu_id));
689 
690  if (it_gpu_id != device_prop_gpu_id_list.end())
691  cur_gpu_selected = true;
692  }
693 
694  if (cur_gpu_selected)
695  if (add_gpu_to_edpp_list(dev_location_id, pci_cdev->irq,
696  gpu_id, hip_num_gpu_devices))
697  amd_gpus_found = true;
698  }
699  }
700 
701  pci_cleanup(pacc);
702 
703  if (amd_gpus_found) {
704  if (do_edp_test())
705  return 0;
706  return -1;
707  }
708 
709  return 0;
710 }

719
