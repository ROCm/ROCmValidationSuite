378 void iet_action::hip_to_smi_indices(void) {
379  int hip_num_gpu_devices;
380  hipGetDeviceCount(&hip_num_gpu_devices);
381  // map this to smi as only these are visible
382  uint32_t smi_num_devices;
383  uint64_t val_ui64;
384  std::map<uint64_t, int> smi_map;
385 
386  rsmi_status_t err = rsmi_num_monitor_devices(&smi_num_devices);
387  if( err == RSMI_STATUS_SUCCESS){
388  for(auto i = 0; i < smi_num_devices; ++i){
389  err = rsmi_dev_pci_id_get(i, &val_ui64);
390  smi_map.insert({val_ui64, i});
391  }
392  }
393 
394  for (int i = 0; i < hip_num_gpu_devices; i++) {
395  // get GPU device properties
396  hipDeviceProp_t props;
397  hipGetDeviceProperties(&props, i);
398 
399  // compute device location_id (needed to match this device
400  // with one of those found while querying the pci bus
401  uint16_t hip_dev_location_id =
402  ((((uint16_t) (props.pciBusID)) << 8) | (((uint16_t)(props.pciDeviceID)) << 3) );
403  if(smi_map.find(hip_dev_location_id) != smi_map.end()){
404  hip_to_smi_idxs.insert({i, smi_map[hip_dev_location_id]});
405  }
406  }
407 }

414 bool iet_action::do_edp_test(map<int, uint16_t> iet_gpus_device_index) {
415  std::string msg;
416  uint32_t dev_idx = 0;
417  size_t k = 0;
418  int gpuId;
419  bool gpu_masking = false; // if HIP_VISIBLE_DEVICES is set, this will be true
420  int hip_num_gpu_devices;
421  hipGetDeviceCount(&hip_num_gpu_devices);
422 
423  vector<IETWorker> workers(iet_gpus_device_index.size());
424  for (;;) {
425  unsigned int i = 0;
426  map<int, uint16_t>::iterator it;
427 
428  if (property_wait != 0) // delay iet execution
429  sleep(property_wait);
430 
431 
432  uint32_t smi_num_devices;
433  rsmi_status_t err = rsmi_num_monitor_devices(&smi_num_devices);
434  if(smi_num_devices != hip_num_gpu_devices)
435  gpu_masking = true;
436  if(gpu_masking){ // this is the case when using HIP_VISIBLE_DEVICES variable to modify GPU visibility
437  // smi output wont be affected by the flag and hence indices should be appropriately used.
438  hip_to_smi_indices();
439  }
440 
441  IETWorker::set_use_json(bjson);
442  for (it = iet_gpus_device_index.begin(); it != iet_gpus_device_index.end(); ++it) {
443  if(hip_to_smi_idxs.find(it->first) != hip_to_smi_idxs.end()){
444  workers[i].set_smi_device_index(hip_to_smi_idxs[it->first]);
445  } else{
446  workers[i].set_smi_device_index(it->first);
447  }
448  gpuId = it->second;
449  // set worker thread params
450  workers[i].set_name(action_name);
451  workers[i].set_gpu_id(it->second);
452  workers[i].set_gpu_device_index(it->first);
453  workers[i].set_pwr_device_id(dev_idx++);
454  workers[i].set_run_wait_ms(property_wait);
455  workers[i].set_run_duration_ms(property_duration);
456  workers[i].set_ramp_interval(iet_ramp_interval);
457  workers[i].set_log_interval(property_log_interval);
458  workers[i].set_sample_interval(iet_sample_interval);
459  workers[i].set_max_violations(iet_max_violations);
460  workers[i].set_target_power(iet_target_power);
461  workers[i].set_tolerance(iet_tolerance);
462  workers[i].set_matrix_size_a(iet_matrix_size_a);
463  workers[i].set_matrix_size_b(iet_matrix_size_b);
464  workers[i].set_matrix_size_c(iet_matrix_size_c);
465  workers[i].set_iet_ops_type(iet_ops_type);
466  workers[i].set_matrix_transpose_a(iet_trans_a);
467  workers[i].set_matrix_transpose_b(iet_trans_b);
468  workers[i].set_alpha_val(iet_alpha_val);
469  workers[i].set_beta_val(iet_beta_val);
470  workers[i].set_lda_offset(iet_lda_offset);
471  workers[i].set_ldb_offset(iet_ldb_offset);
472  workers[i].set_ldc_offset(iet_ldc_offset);
473  workers[i].set_tp_flag(iet_tp_flag);
474 
475  i++;
476  }
477 
478  if (property_parallel) {
479  for (i = 0; i < iet_gpus_device_index.size(); i++)
480  workers[i].start();
481  // join threads
482  for (i = 0; i < iet_gpus_device_index.size(); i++)
483  workers[i].join();
484 
485  } else {
486  for (i = 0; i < iet_gpus_device_index.size(); i++) {
487  workers[i].start();
488  workers[i].join();
489 
490  // check if stop signal was received
491  if (rvs::lp::Stopping()) {
492  return false;
493  }
494  }
495  }
496 
497 
498  msg = "[" + action_name + "] " + MODULE_NAME + " " + std::to_string(gpuId) + " Shutting down rocm-smi ";
499  rvs::lp::Log(msg, rvs::loginfo);
500 
501 
502  // check if stop signal was received
503  if (rvs::lp::Stopping())
504  return false;
505 
506  if (property_count == ++k) {
507  break;
508  }
509  }
510 
511 
512  msg = "[" + action_name + "] " + MODULE_NAME + " " + std::to_string(gpuId) + " Done with edp test ";
513  rvs::lp::Log(msg, rvs::loginfo);
514 
515  sleep(1000);
516 
517  return true;
518 }
