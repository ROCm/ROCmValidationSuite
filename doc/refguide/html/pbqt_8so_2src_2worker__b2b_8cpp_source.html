150 void pbqtworker_b2b::run() {
151  std::chrono::time_point<std::chrono::system_clock> pbqt_start_time;
152  std::chrono::time_point<std::chrono::system_clock> pbqt_end_time;
153  hsa_status_t status;
154  int sts;
155 
156  RVSTRACE_
157 
158  // enable test
159  brun = true;
160 
161  // allocate buffers and grant permissions for forward transfer
162  sts = pHsa->Allocate(ctx_fwd.SrcAgentIx, ctx_fwd.DstAgentIx, b2b_block_size,
163  &ctx_fwd.SrcPool, &ctx_fwd.pSrcBuff,
164  &ctx_fwd.DstPool, &ctx_fwd.pDstBuff);
165  if (sts) {
166  RVSTRACE_
167  deinit();
168  return;
169  }
170 
171  // Create a signal to wait on forward copy operation
172  if (HSA_STATUS_SUCCESS !=
173  (status = hsa_signal_create(1, 0, NULL, &ctx_fwd.Sig))) {
174  rvs::hsa::print_hsa_status(__FILE__, __LINE__, __func__,
175  "hsa_signal_create()", status);
176  RVSTRACE_
177  deinit();
178  return;
179  }
180 
181  // allocate buffers and grant permissions for reverse transfer
182  if (bidirect) {
183  sts = pHsa->Allocate(ctx_rev.SrcAgentIx, ctx_rev.DstAgentIx, b2b_block_size,
184  &ctx_rev.SrcPool, &ctx_rev.pSrcBuff,
185  &ctx_rev.DstPool, &ctx_rev.pDstBuff);
186 
187  if (sts) {
188  RVSTRACE_
189  deinit();
190  return;
191  }
192 
193  // Create a signal to wait on reverse copy operation
194  if (HSA_STATUS_SUCCESS !=
195  (status = hsa_signal_create(1, 0, NULL, &ctx_rev.Sig))) {
196  rvs::hsa::print_hsa_status(__FILE__, __LINE__, __func__,
197  "hsa_signal_create()", status);
198  RVSTRACE_
199  deinit();
200  return;
201  }
202  }
203 
204 
205  pbqt_start_time = std::chrono::system_clock::now();
206 
207  while (brun) {
208  // initiate forward transfer
209 
210  RVSTRACE_
211  hsa_signal_store_relaxed(ctx_fwd.Sig, 1);
212  if (HSA_STATUS_SUCCESS !=
213  (status = hsa_amd_memory_async_copy(
214  ctx_fwd.pDstBuff, ctx_fwd.DstAgent,
215  ctx_fwd.pSrcBuff, ctx_fwd.SrcAgent,
216  b2b_block_size,
217  0, NULL, ctx_fwd.Sig))) {
218  rvs::hsa::print_hsa_status(__FILE__, __LINE__, __func__,
219  "hsa_amd_memory_async_copy()",
220  status);
221  break;
222  }
223 
224  if (bidirect) {
225  RVSTRACE_
226  // initiate reverse transfer
227  hsa_signal_store_relaxed(ctx_rev.Sig, 1);
228  if (HSA_STATUS_SUCCESS != (status = hsa_amd_memory_async_copy(
229  ctx_rev.pDstBuff, ctx_rev.DstAgent,
230  ctx_rev.pSrcBuff, ctx_rev.SrcAgent,
231  b2b_block_size,
232  0, NULL, ctx_rev.Sig))) {
233  rvs::hsa::print_hsa_status(__FILE__, __LINE__, __func__,
234  "hsa_amd_memory_async_copy()",
235  status);
236  break;
237  }
238  }
239 
240  // wait for transfer to complete
241  RVSTRACE_
242  while (hsa_signal_wait_acquire(ctx_fwd.Sig, HSA_SIGNAL_CONDITION_LT,
243  1, uint64_t(-1), HSA_WAIT_STATE_ACTIVE)) {}
244 
245  // if bidirectional, also wait for reverse transfer to complete
246  if (bidirect) {
247  RVSTRACE_
248  while (hsa_signal_wait_acquire(ctx_rev.Sig, HSA_SIGNAL_CONDITION_LT,
249  1, uint64_t(-1), HSA_WAIT_STATE_ACTIVE)) {}
250  }
251 
252  RVSTRACE_
253  // get transfer duration
254  double duration = pHsa->GetCopyTime(bidirect,
255  ctx_fwd.Sig, ctx_rev.Sig)/1000000000;
256  {
257  RVSTRACE_
258  std::lock_guard<std::mutex> lk(cntmutex);
259  running_size += b2b_block_size;
260  running_duration += duration;
261  }
262  pbqt_end_time = std::chrono::system_clock::now();
263 
264  uint64_t test_time = time_diff(pbqt_end_time, pbqt_start_time) ;
265 
266  if(test_time >= test_duration) {
267  break;
268  }
269 
270  } // while(brun)
271 
272  RVSTRACE_
273  // deallocate buffers and signals
274  deinit();
275 }
