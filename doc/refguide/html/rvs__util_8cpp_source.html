44 std::vector<std::string> str_split(const std::string& str_val,
45  const std::string& delimiter) {
46  std::vector<std::string> str_tokens;
47  int prev_pos = 0, cur_pos = 0;
48  do {
49  cur_pos = str_val.find(delimiter, prev_pos);
50  if (cur_pos == std::string::npos)
51  cur_pos = str_val.length();
52  std::string token = str_val.substr(prev_pos, cur_pos - prev_pos);
53  if (!token.empty())
54  str_tokens.push_back(token);
55  prev_pos = cur_pos + delimiter.length();
56  } while (cur_pos < str_val.length() && prev_pos < str_val.length());
57  return str_tokens;
58 }

107 bool fetch_gpu_list(int hip_num_gpu_devices, map<int, uint16_t>& gpus_device_index,
108  const std::vector<uint16_t>& property_device,
109  const int& property_device_id, bool property_device_all, bool mcm_check){
110  bool amd_gpus_found = false;
111  bool mcm_die = false;
112  bool amd_mcm_gpu_found = false;
113  for (int i = 0; i < hip_num_gpu_devices; i++) {
114  // get GPU device properties
115  hipDeviceProp_t props;
116  hipGetDeviceProperties(&props, i);
117 
118  // compute device location_id (needed in order to identify this device
119  // in the gpus_id/gpus_device_id list
120  unsigned int dev_location_id =
121  ((((unsigned int) (props.pciBusID)) << 8) | ((unsigned int)(props.pciDeviceID)) << 3);
122  uint16_t dev_domain = props.pciDomainID;
123  uint16_t devId;
124  uint16_t gpu_id;
125  if (rvs::gpulist::domlocation2gpu(dev_domain, dev_location_id, &gpu_id)) {
126  continue;
127  }
128  if (rvs::gpulist::gpu2device(gpu_id, &devId)){
129  continue;
130  }
131  // filter by device id if needed
132  if (property_device_id > 0 && property_device_id != devId)
133  continue;
134 
135  // check if this GPU is part of the select ones as per config
136  // (device = "all" or the gpu_id is in the device: <gpu id> list)
137  bool cur_gpu_selected = false;
138 
139  if (property_device_all) {
140  cur_gpu_selected = true;
141  } else {
142  // search for this gpu in the list
143  // provided under the <device> property
144  auto it_gpu_id = find(property_device.begin(),
145  property_device.end(),
146  gpu_id);
147 
148  if (it_gpu_id != property_device.end())
149  cur_gpu_selected = true;
150  }
151  // if mcm check enabled, print message if device is MCM
152  if (mcm_check){
153  std::stringstream msg_stream;
154  mcm_die = gpu_check_if_mcm_die(i);
155  if (mcm_die) {
156  msg_stream.str("");
157  msg_stream << "GPU ID : " << std::setw(5) << gpu_id << " - " << "Device : " << std::setw(5) << devId <<
158  " - " << "GPU is a die/chiplet in Multi-Chip Module (MCM) GPU";
159  rvs::lp::Log(msg_stream.str(), rvs::logresults);
160 
161  amd_mcm_gpu_found = true;
162  }
163  }
164  // excluding secondary die from test list, drops power reading substantially
165  if (cur_gpu_selected) { // need not exclude secondary die, just log it out.
166  gpus_device_index.insert
167  (std::pair<int, uint16_t>(i, gpu_id));
168  amd_gpus_found = true;
169  }
170  mcm_die = false;
171  }
172  if (amd_mcm_gpu_found && mcm_check) {
173  std::stringstream msg_stream;
174  msg_stream.str("");
175  msg_stream << "Note: The system has Multi-Chip Module (MCM) GPU/s." << "\n"
176  << "In MCM GPU, primary GPU die shows total socket (primary + secondary) power information." << "\n"
177  << "Secondary GPU die does not have any power information associated with it independently."<< "\n"
178  << "So, expect power reading from Secondary GPU die as 0."<< "\n";
179  rvs::lp::Log(msg_stream.str(), rvs::logresults);
180  }
181  return amd_gpus_found;
182 }

187  struct device_info {
188  std::string bus;
189  std::string name;
190  int32_t node_id;
191  int32_t gpu_id;
192  int32_t device_id;
193  };
194 
195  char buff[1024];
196  int hip_num_gpu_devices;
197  std::string errmsg = " No supported GPUs available.";
198  std::vector<device_info> gpu_info_list;
199 
200  hipGetDeviceCount(&hip_num_gpu_devices);
201  if( hip_num_gpu_devices == 0){
202  std::cout << std::endl << errmsg << std::endl;
203  return 0;
204  }
205  for (int i = 0; i < hip_num_gpu_devices; i++) {
206  hipDeviceProp_t props;
207  hipGetDeviceProperties(&props, i);
208 
209  // compute device location_id (needed in order to identify this device
210  // in the gpus_id/gpus_device_id list
211  unsigned int dev_location_id =
212  ((((unsigned int) (props.pciBusID)) << 8) | ((unsigned int)(props.pciDeviceID)) << 3);
213  uint16_t dev_domain = props.pciDomainID;
214  uint16_t node_id;
215  if (rvs::gpulist::domlocation2node(dev_domain, dev_location_id, &node_id)) {
216  continue;
217  }
218  uint16_t gpu_id;
219  if (rvs::gpulist::domlocation2gpu(dev_domain, dev_location_id, &gpu_id)) {
220  continue;
221  }
222  uint16_t dev_id;
223  if (rvs::gpulist::gpu2device(gpu_id, &dev_id)){
224  continue;
225  }
226  snprintf(buff, sizeof(buff), "%04d:%02X:%02X.%d",props.pciDomainID, props.pciBusID, props.pciDeviceID, 0);
227  device_info info;
228  info.bus = buff;
229  info.name = props.name;
230  info.node_id = node_id;
231  info.gpu_id = gpu_id;
232  info.device_id = dev_id;
233  gpu_info_list.push_back(info);
234 
235  }
236  std::sort(gpu_info_list.begin(), gpu_info_list.end(),
237  [](const struct device_info& a, const struct device_info& b) {
238  return a.node_id < b.node_id; });
239  if (!gpu_info_list.empty()) {
240  std::cout << "Supported GPUs available:\n";
241  for (const auto& info : gpu_info_list) {
242  std::cout << info.bus << " - GPU[" << std::setw(2) << info.node_id
243  << " - " << std::setw(5) << info.gpu_id << "] " << info.name
244  << " (Device " << info.device_id << ")\n";
245  }
246  } else {
247  std::cout << std::endl << errmsg << std::endl;
248  }
249  return 0;
250 }
