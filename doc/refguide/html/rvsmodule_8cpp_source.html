79 int rvs::module::initialize(const char* pConfig) {
80  // Check if pConfig file exists
81  std::ifstream file(pConfig);
82 
83  if (!file.good()) {
84  char buff[1024];
85  snprintf(buff, sizeof(buff), "file does not exist: %s", pConfig);
86  rvs::logger::Err(buff, MODULE_NAME_CAPS);
87  return -1;
88  } else {
89  file.close();
90  }
91 
92  // load list of supported modules from config file
93  YAML::Node config = YAML::LoadFile(pConfig);
94 
95  // verify that that the file format is supported
96  YAML::const_iterator it = config.begin();
97  if (it == config.end()) {
98  rvs::logger::Err("unsupported file format. Version string not found.",
99  MODULE_NAME_CAPS);
100  return -1;
101  }
102 
103  std::string key = it->first.as<std::string>();
104  std::string value = it->second.as<std::string>();
105 
106  if (key != "version") {
107  rvs::logger::Err("unsupported file format. Version string not found.",
108  MODULE_NAME_CAPS);
109  return -1;
110  }
111 
112  if (value != "1") {
113  char buff[1024];
114  snprintf(buff, sizeof(buff), "file version is %s, expected 1",
115  value.c_str());
116  rvs::logger::Err(buff, MODULE_NAME_CAPS);
117  return -1;
118  }
119 
120  // load nam-file pairs:
121  for (it++; it != config.end(); ++it) {
122  key = it->first.as<std::string>();
123  value = it->second.as<std::string>();
124  filemap.insert(std::pair<string, string>(key, value));
125  }
126 
127  return 0;
128 }

139 rvs::module* rvs::module::find_create_module(const char* name) {
140  module* m = nullptr;
141 
142  // find module based on short name
143  auto it = modulemap.find(std::string(name));
144 
145  // not found...
146  if (it == modulemap.end()) {
147  // ... try opening .so
148 
149  // first find proper .so filename
150  auto it = filemap.find(std::string(name));
151 
152  // not found...
153  if (it == filemap.end()) {
154  // this should never happen if .config is OK
155  char buff[1024];
156  snprintf(buff, sizeof(buff),
157  "module '%s' not found in configuration.", name);
158  rvs::logger::Err(buff, MODULE_NAME_CAPS);
159  return NULL;
160  }
161 
162  // open .so
163  string libpath;
164  if (rvs::options::has_option("-m", &libpath)) {
165  libpath += "/";
166  } else {
167  rvs::options::has_option("pwd", &libpath); // has ending forward slash too
168  libpath += "../lib/rvs/";
169  }
170  string sofullname(libpath + it->second);
171  void* psolib = dlopen(sofullname.c_str(), RTLD_NOW);
172  // error?
173  if (!psolib) {
174  //Search libraries in current path for backward compatibility
175  rvs::options::has_option("pwd", &libpath); // has ending forward slash too
176  string sofullname(libpath + it->second);
177  psolib = dlopen(sofullname.c_str(), RTLD_NOW);
178  // error?
179  if (!psolib) {
180  char buff[1024];
181  snprintf(buff, sizeof(buff),
182  "could not load .so '%s'", sofullname.c_str());
183  rvs::logger::Err(buff, MODULE_NAME_CAPS);
184  snprintf(buff, sizeof(buff),
185  "reason: '%s'", dlerror());
186  rvs::logger::Err(buff, MODULE_NAME_CAPS);
187  return NULL; // fail
188  }
189  }
190  // create module object
191  m = new rvs::module(name, psolib);
192  if (!m) {
193  dlclose(psolib);
194  return NULL;
195  }
196 
197  // initialize API function pointers
198  if (m->init_interfaces()) {
199  char buff[1024];
200  snprintf(buff, sizeof(buff),
201  "could not init interfaces for '%s'", it->second.c_str());
202  rvs::logger::Err(buff, MODULE_NAME_CAPS);
203  dlclose(psolib);
204  delete m;
205  return nullptr;
206  }
207 
208  // initialize newly loaded module
209  if (m->initialize()) {
210  char buff[1024];
211  snprintf(buff, sizeof(buff),
212  "could not initialize '%s'", it->second.c_str());
213  rvs::logger::Err(buff, MODULE_NAME_CAPS);
214  dlclose(psolib);
215  delete m;
216  return nullptr;
217  }
218 
219  // add to map
220  modulemap.insert(t_mmpair(name, m));
221  } else {
222  m = it->second;
223  }
224 
225  return m;
226 }

267 rvs::action* rvs::module::action_create(const char* name) {
268  // find module
269  rvs::module* m = module::find_create_module(name);
270  if (!m) {
271  char buff[1024];
272  snprintf(buff, sizeof(buff),
273  "module '%s' not available.", name);
274  rvs::logger::Err(buff, MODULE_NAME_CAPS);
275  return nullptr;
276  }
277 
278  // create lib action objct
279  void* plibaction = m->action_create();
280  if (!plibaction) {
281  char buff[1024];
282  snprintf(buff, sizeof(buff),
283  "module '%s' could not create lib action.", name);
284  rvs::logger::Err(buff, MODULE_NAME_CAPS);
285  return nullptr;
286  }
287 
288  // create action proxy object
289  rvs::action* pa = new rvs::action(name, plibaction);
290  if (!pa) {
291  char buff[1024];
292  snprintf(buff, sizeof(buff),
293  "module '%s' could not create action proxy.", name);
294  rvs::logger::Err(buff, MODULE_NAME_CAPS);
295  return nullptr;
296  }
297 
298  // create interfaces for the proxy
299  // clone from module and assign libaction ptr
300 
301  for (auto it = m->ifmap.begin(); it != m->ifmap.end(); it++) {
302  std::shared_ptr<rvs::ifbase> sptrif(it->second->clone());
303  sptrif->plibaction = plibaction;
304  pa->ifmap.insert(rvs::action::t_impair(it->first, sptrif));
305  }
306 
307  return pa;
308 }

564 void rvs::module::do_list_modules(void) {
565  // for all modules
566  for (auto it = filemap.begin(); it != filemap.end(); it++) {
567  // create action
568  rvs::action* pa = rvs::module::action_create(it->first.c_str());
569  if (!pa) {
570  char buff[1024];
571  snprintf(buff, sizeof(buff),
572  "could not open module '%s'", it->first.c_str());
573  rvs::logger::Err(buff, MODULE_NAME_CAPS);
574  continue;
575  }
576 
577  // output module name
578  std::cout << "\t" << it->first.c_str() << ":" << std::endl;
579 
580  // obtain IF0
581  rvs::if0* pif0 = dynamic_cast<rvs::if0*>(pa->get_interface(0));
582  if (!pif0) {
583  // action no longer needed so destroy it
584  rvs::module::action_destroy(pa);
585  rvs::logger::Err("could not obtain interface if0", MODULE_NAME_CAPS);
586  continue;
587  }
588 
589  // output info
590  std::cout << "\t\tDescription: " << pif0->get_description() << std::endl;
591  std::cout << "\t\tconfig: " << pif0->get_config() << std::endl;
592  std::cout << "\t\toutput: " << pif0->get_output() << std::endl;
593 
594  // action no longer needed so destroy it
595  rvs::module::action_destroy(pa);
596  }
597 }
