78 int rvs::module::initialize(const char* pConfig) {
79  // Check if pConfig file exists
80  std::ifstream file(pConfig);
81 
82  if (!file.good()) {
83  std::cerr << "RVS-CLI: " << pConfig << " file does not exist.\n";
84  return -1;
85  } else {
86  file.close();
87  }
88 
89  // load list of supported modules from config file
90  YAML::Node config = YAML::LoadFile(pConfig);
91 
92  // verify that that the file format is supported
93  YAML::const_iterator it = config.begin();
94  if (it == config.end()) {
95  std::cerr << "RVS-CLI: unsupported file format. Version string not found."
96  << std::endl;
97  return -1;
98  }
99 
100  std::string key = it->first.as<std::string>();
101  std::string value = it->second.as<std::string>();
102 
103  if (key != "version") {
104  std::cerr << "RVS-CLI: unsupported file format. Version string not found."
105  << std::endl;
106  return -1;
107  }
108 
109  if (value != "1") {
110  std::cerr << "RVS-CLI: version is not 1" << std::endl;
111  return -1;
112  }
113 
114  // load nam-file pairs:
115  for (it++; it != config.end(); ++it) {
116  key = it->first.as<std::string>();
117  value = it->second.as<std::string>();
118  filemap.insert(std::pair<string, string>(key, value));
119  }
120 
121  return 0;
122 }

133 rvs::module* rvs::module::find_create_module(const char* name) {
134  module* m = nullptr;
135 
136  // find module based on short name
137  auto it = modulemap.find(std::string(name));
138 
139  // not found...
140  if (it == modulemap.end()) {
141  // ... try opening .so
142 
143  // first find proper .so filename
144  auto it = filemap.find(std::string(name));
145 
146  // not found...
147  if (it == filemap.end()) {
148  // this should never happen if .config is OK
149  std::cerr << "RVS-CLI: module '" << name << "' not found in configuration."
150  << std::endl;
151  return NULL;
152  }
153 
154  // open .so
155  string libpath;
156  if (rvs::options::has_option("-m", &libpath)) {
157  libpath += "/";
158  } else {
159  rvs::options::has_option("pwd", &libpath);
160  libpath += "/";
161  }
162  string sofullname(libpath + it->second);
163  void* psolib = dlopen(sofullname.c_str(), RTLD_NOW);
164 
165  // error?
166  if (!psolib) {
167  std::cerr << "RVS-CLI: could not load .so '" << sofullname << "' reason: "
168  << dlerror() << std::endl;
169  return NULL; // fail
170  }
171 
172  // create module object
173  m = new rvs::module(name, psolib);
174  if (!m) {
175  dlclose(psolib);
176  return NULL;
177  }
178 
179  // initialize API function pointers
180  if (m->init_interfaces()) {
181  std::cerr << "RVS-CLI: could not init interfaces for '"
182  << it->second.c_str() << "'" << std::endl;
183  dlclose(psolib);
184  delete m;
185  return nullptr;
186  }
187 
188  // initialize newly loaded module
189  if (m->initialize()) {
190  std::cerr << "RVS-CLI: could not initialize '" << it->second.c_str() << "'"
191  << std::endl;
192  dlclose(psolib);
193  delete m;
194  return nullptr;
195  }
196 
197  // add to map
198  modulemap.insert(t_mmpair(name, m));
199  } else {
200  m = it->second;
201  }
202 
203  return m;
204 }

240 rvs::action* rvs::module::action_create(const char* name) {
241  // find module
242  rvs::module* m = module::find_create_module(name);
243  if (!m) {
244  std::cerr << "RVS-CLI: module '" << name << "' not available." << std::endl;
245  return nullptr;
246  }
247 
248  // create lib action objct
249  void* plibaction = m->action_create();
250  if (!plibaction) {
251  std::cerr << "RVS-CLI: module '" << name << "' could not create lib action."
252  << std::endl;
253  return nullptr;
254  }
255 
256  // create action proxy object
257  rvs::action* pa = new rvs::action(name, plibaction);
258  if (!pa) {
259  std::cerr << "RVS-CLI: module '" << name << "' could not create action proxy."
260  << std::endl;
261  return nullptr;
262  }
263 
264  // create interfaces for the proxy
265  // clone from module and assign libaction ptr
266 
267  for (auto it = m->ifmap.begin(); it != m->ifmap.end(); it++) {
268  std::shared_ptr<rvs::ifbase> sptrif(it->second->clone());
269  sptrif->plibaction = plibaction;
270  pa->ifmap.insert(rvs::action::t_impair(it->first, sptrif));
271  }
272 
273  return pa;
274 }

516 void rvs::module::do_list_modules(void) {
517  // for all modules
518  for (auto it = filemap.begin(); it != filemap.end(); it++) {
519  // create action
520  rvs::action* pa = rvs::module::action_create(it->first.c_str());
521  if (!pa) {
522  std::cerr << "RVS-CLI: could not open module: " << it->first.c_str()
523  << std::endl;
524  continue;
525  }
526 
527  // output module name
528  std::cout << "\t" << it->first.c_str() << ":" << std::endl;
529 
530  // obtain IF0
531  rvs::if0* pif0 = dynamic_cast<rvs::if0*>(pa->get_interface(0));
532  if (!pif0) {
533  // action no longer needed so destroy it
534  rvs::module::action_destroy(pa);
535 
536  std::cerr << "RVS-CLI: could not obtain data." << std::endl;
537  continue;
538  }
539 
540  // output info
541  std::cout << "\t\tDescription: " << pif0->get_description() << std::endl;
542  std::cout << "\t\tconfig: " << pif0->get_config() << std::endl;
543  std::cout << "\t\toutput: " << pif0->get_output() << std::endl;
544 
545  // action no longer needed so destroy it
546  rvs::module::action_destroy(pa);
547  }
548 }
