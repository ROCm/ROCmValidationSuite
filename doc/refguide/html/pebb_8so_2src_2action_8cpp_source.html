223 int pebb_action::create_threads() {
224  std::string msg;
225  std::vector<uint16_t> gpu_id;
226  std::vector<uint16_t> gpu_device_id;
227  uint16_t transfer_ix = 0;
228  bool bmatch_found = false;
229 
230  RVSTRACE_
231  gpu_get_all_gpu_id(&gpu_id);
232  gpu_get_all_device_id(&gpu_device_id);
233 
234 
235  RVSTRACE_
236  for (size_t i = 0; i < gpu_id.size(); i++) {
237  RVSTRACE_
238  if (property_device_id > 0) {
239  RVSTRACE_
240  if (property_device_id != gpu_device_id[i]) {
241  RVSTRACE_
242  continue;
243  }
244  }
245 
246  // filter out by listed sources
247  RVSTRACE_
248  if (!property_device_all) {
249  RVSTRACE_
250  const auto it = std::find(property_device.cbegin(),
251  property_device.cend(),
252  gpu_id[i]);
253  if (it == property_device.cend()) {
254  RVSTRACE_
255  continue;
256  }
257  }
258 
259  uint16_t dstnode;
260  int srcnode;
261 
262  RVSTRACE_
263  for (uint cpu_index = 0;
264  cpu_index < rvs::hsa::Get()->cpu_list.size();
265  cpu_index++) {
266  RVSTRACE_
267 
268  if (rvs::gpulist::gpu2node(gpu_id[i], &dstnode)) {
269  RVSTRACE_
270  msg = "no node found for destination GPU ID "
271  + std::to_string(gpu_id[i]);
272  rvs::lp::Err(msg, MODULE_NAME_CAPS, action_name);
273  return -1;
274  }
275  RVSTRACE_
276  srcnode = rvs::hsa::Get()->cpu_list[cpu_index].node;
277 
278  // get link info regardless of peer status (just in case...)
279  uint32_t distance = 0;
280  bool b_reverse = false;
281 
282  std::vector<rvs::linkinfo_t> arr_linkinfo;
283  rvs::hsa::Get()->GetLinkInfo(srcnode, dstnode,
284  &distance, &arr_linkinfo);
285  if (distance == rvs::hsa::NO_CONN) {
286  RVSTRACE_
287  rvs::hsa::Get()->GetLinkInfo(dstnode, srcnode,
288  &distance, &arr_linkinfo);
289  if (distance != rvs::hsa::NO_CONN) {
290  RVSTRACE_
291  // there is a path if transfer is initiated by
292  // destination agent:
293  b_reverse = true;
294  }else{// if no connection either way, no point in adding to list
295  continue;
296  }
297  }
298 
299  // if link type is specified, check that it matches
300  if (!rvs::hsa::check_link_type(arr_linkinfo, link_type))
301  continue;
302 
303  bmatch_found = true;
304  transfer_ix += 1;
305 
306  print_link_info(srcnode, dstnode, gpu_id[i],
307  distance, arr_linkinfo, b_reverse);
308 
309  // if GPUs are peers, create transaction for them
310  if (rvs::hsa::Get()->GetPeerStatus(srcnode, dstnode)) {
311  RVSTRACE_
312  pebbworker* p = nullptr;
313  if (property_parallel && b2b_block_size > 0) {
314  RVSTRACE_
315  pebbworker_b2b* pb2b = new pebbworker_b2b;
316  if (pb2b == nullptr) {
317  RVSTRACE_
318  msg = "internal error";
319  rvs::lp::Err(msg, MODULE_NAME_CAPS, action_name);
320  return -1;
321  }
322  pb2b->initialize(srcnode, dstnode,
323  prop_h2d, prop_d2h, b2b_block_size);
324  p = pb2b;
325  } else {
326  RVSTRACE_
327  p = new pebbworker;
328  if (p == nullptr) {
329  RVSTRACE_
330  msg = "internal error";
331  rvs::lp::Err(msg, MODULE_NAME_CAPS, action_name);
332  return -1;
333  }
334  p->initialize(srcnode, dstnode, prop_h2d, prop_d2h);
335  }
336  RVSTRACE_
337  p->set_name(action_name);
338  p->set_stop_name(action_name);
339  p->set_transfer_ix(transfer_ix);
340  p->set_block_sizes(block_size);
341  p->set_loglevel(property_log_level);
342  test_array.push_back(p);
343  }
344  }
345  }
346 
347  RVSTRACE_
348  if (test_array.size() < 1) {
349  std::string diag;
350  if (bmatch_found) {
351  diag = "No peers found";
352  } else {
353  diag = "No devices match criteria from the test configuration";
354  }
355  msg = "[" + action_name + "] pcie-bandwidth " + diag;
356  rvs::lp::Log(msg, rvs::logerror);
357  if (bjson) {
358  unsigned int sec;
359  unsigned int usec;
360  rvs::lp::get_ticks(&sec, &usec);
361  void* pjson = rvs::lp::LogRecordCreate("pcie-bandwidth",
362  action_name.c_str(), rvs::logerror, sec, usec);
363  if (pjson != NULL) {
364  rvs::lp::AddString(pjson,
365  "message",
366  diag);
367  rvs::lp::LogRecordFlush(pjson);
368  }
369  }
370  return -1;
371  }
372 
373  for (auto it = test_array.begin(); it != test_array.end(); ++it) {
374  RVSTRACE_
375  (*it)->set_transfer_num(test_array.size());
376  }
377 
378  RVSTRACE_
379  return 0;
380 }

466 int pebb_action::print_running_average(pebbworker* pWorker) {
467  uint16_t src_node, dst_node;
468  uint16_t dst_id;
469  bool bidir;
470  size_t current_size;
471  double duration;
472  std::string msg;
473  char buff[64];
474  double bandwidth;
475  uint16_t transfer_ix;
476  uint16_t transfer_num;
477 
478  RVSTRACE_
479  // get running average
480  pWorker->get_running_data(&src_node, &dst_node, &bidir,
481  &current_size, &duration);
482 
483  if (duration > 0) {
484  RVSTRACE_
485  bandwidth = current_size/duration/1000/1000/1000;
486  if (bidir) {
487  RVSTRACE_
488  bandwidth *=2;
489  }
490  snprintf( buff, sizeof(buff), "%.3f GBps", bandwidth);
491  } else {
492  RVSTRACE_
493  // no running average in this iteration, try getting total so far
494  // (do not reset final totals as this is just intermediate query)
495  pWorker->get_final_data(&src_node, &dst_node, &bidir,
496  &current_size, &duration, false);
497  RVSTRACE_
498  bandwidth = current_size/duration/1000/1000/1000;
499  if (bidir) {
500  RVSTRACE_
501  bandwidth *=2;
502  }
503  snprintf( buff, sizeof(buff), "%.3f GBps (*)", bandwidth);
504  }
505 
506 // dst_id = rvs::gpulist::GetGpuIdFromNodeId(dst_node);
507 
508  RVSTRACE_
509  if (rvs::gpulist::node2gpu(dst_node, &dst_id)) {
510  RVSTRACE_
511  std::string msg = "could not find GPU id for node " +
512  std::to_string(dst_node);
513  rvs::lp::Err(msg, MODULE_NAME_CAPS, action_name);
514  return -1;
515  }
516  RVSTRACE_
517  transfer_ix = pWorker->get_transfer_ix();
518  transfer_num = pWorker->get_transfer_num();
519 
520  msg = "[" + action_name + "] pcie-bandwidth ["
521  + std::to_string(transfer_ix) + "/" + std::to_string(transfer_num)
522  + "] "
523  + std::to_string(src_node) + " " + std::to_string(dst_id)
524  + " h2d: " + (prop_h2d ? "true" : "false")
525  + " d2h: " + (prop_d2h ? "true" : "false") + " "
526  + buff;
527 
528  rvs::lp::Log(msg, rvs::loginfo);
529 
530  log_json_bandwidth(std::to_string(src_node), std::to_string(dst_id),rvs::logresults, buff);
531  RVSTRACE_
532  return 0;
533 }
