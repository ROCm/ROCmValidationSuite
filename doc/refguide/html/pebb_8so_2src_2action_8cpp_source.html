235 int pebbaction::create_threads() {
236  std::string msg;
237  std::vector<uint16_t> gpu_id;
238  std::vector<uint16_t> gpu_device_id;
239  uint16_t transfer_ix = 0;
240  bool bmatch_found = false;
241 
242  RVSTRACE_
243  gpu_get_all_gpu_id(&gpu_id);
244  gpu_get_all_device_id(&gpu_device_id);
245 
246  RVSTRACE_
247  for (size_t i = 0; i < gpu_id.size(); i++) {
248  RVSTRACE_
249  if (prop_device_id_filtering) {
250  RVSTRACE_
251  if (prop_deviceid != gpu_device_id[i]) {
252  RVSTRACE_
253  continue;
254  }
255  }
256 
257  // filter out by listed sources
258  RVSTRACE_
259  if (!prop_device_all_selected) {
260  RVSTRACE_
261  const auto it = std::find(device_prop_gpu_id_list.cbegin(),
262  device_prop_gpu_id_list.cend(),
263  std::to_string(gpu_id[i]));
264  if (it == device_prop_gpu_id_list.cend()) {
265  RVSTRACE_
266  continue;
267  }
268  }
269 
270  bmatch_found = true;
271 
272  int dstnode;
273  int srcnode;
274 
275  RVSTRACE_
276  for (uint cpu_index = 0;
277  cpu_index < rvs::hsa::Get()->cpu_list.size();
278  cpu_index++) {
279  RVSTRACE_
280 
281  bool b_reverse = false;
282 
283  dstnode = rvs::gpulist::GetNodeIdFromGpuId(gpu_id[i]);
284  if (dstnode < 0) {
285  RVSTRACE_
286  std::cerr << "RVS-PEBB: no node found for destination GPU ID "
287  << std::to_string(gpu_id[i]);
288  return -1;
289  }
290  RVSTRACE_
291  transfer_ix += 1;
292  srcnode = rvs::hsa::Get()->cpu_list[cpu_index].node;
293 
294  // get link info regardless of peer status (just in case...)
295  uint32_t distance = 0;
296  std::vector<rvs::linkinfo_t> arr_linkinfo;
297  rvs::hsa::Get()->GetLinkInfo(srcnode, dstnode,
298  &distance, &arr_linkinfo);
299  if (distance == rvs::hsa::NO_CONN) {
300  RVSTRACE_
301  rvs::hsa::Get()->GetLinkInfo(dstnode, srcnode,
302  &distance, &arr_linkinfo);
303  if (distance != rvs::hsa::NO_CONN) {
304  RVSTRACE_
305  // there is a path if transfer is initiated by
306  // destination agent:
307  b_reverse = true;
308  }
309  }
310  print_link_info(srcnode, dstnode, gpu_id[i],
311  distance, arr_linkinfo, b_reverse);
312 
313  // if GPUs are peers, create transaction for them
314  if (rvs::hsa::Get()->GetPeerStatus(srcnode, dstnode)) {
315  RVSTRACE_
316  pebbworker* p = new pebbworker;
317  if (p == nullptr) {
318  RVSTRACE_
319  std::cerr << "RVS-PEBB: internal error";
320  return -1;
321  }
322  p->initialize(srcnode, dstnode, prop_h2d, prop_d2h);
323  p->set_name(action_name);
324  p->set_stop_name(action_name);
325  p->set_transfer_ix(transfer_ix);
326  p->set_block_sizes(block_size);
327  test_array.push_back(p);
328  }
329  }
330  }
331 
332  RVSTRACE_
333  if (test_array.size() < 1) {
334  std::string diag;
335  if (bmatch_found) {
336  diag = "No peers found";
337  } else {
338  diag = "No devices match criteria from the test configuation";
339  }
340  msg = "[" + action_name + "] pcie-bandwidth " + diag;
341  rvs::lp::Log(msg, rvs::logerror);
342  if (bjson) {
343  unsigned int sec;
344  unsigned int usec;
345  rvs::lp::get_ticks(&sec, &usec);
346  void* pjson = rvs::lp::LogRecordCreate("pcie-bandwidth",
347  action_name.c_str(), rvs::logerror, sec, usec);
348  if (pjson != NULL) {
349  rvs::lp::AddString(pjson,
350  "message",
351  diag);
352  rvs::lp::LogRecordFlush(pjson);
353  }
354  }
355  return 0;
356  }
357 
358  for (auto it = test_array.begin(); it != test_array.end(); ++it) {
359  RVSTRACE_
360  (*it)->set_transfer_num(test_array.size());
361  }
362 
363  RVSTRACE_
364  return 0;
365 }

389 int pebbaction::run() {
390  int sts;
391  string msg;
392 
393  RVSTRACE_
394  if (property.find("cli.-j") != property.end()) {
395  bjson = true;
396  }
397 
398  if (!get_all_common_config_keys())
399  return -1;
400  if (!get_all_pebb_config_keys())
401  return -1;
402 
403  // log_interval must be less than duration
404  if (prop_log_interval > 0 && gst_run_duration_ms > 0) {
405  if (static_cast<uint64_t>(prop_log_interval) > gst_run_duration_ms) {
406  cerr << "RVS-PEBB: action: " << action_name <<
407  " log_interval must be less than duration" << std::endl;
408  return -1;
409  }
410  }
411 
412  sts = create_threads();
413 
414  if (sts != 0) {
415  return sts;
416  }
417 
418  if (gst_runs_parallel) {
419  sts = run_parallel();
420  } else {
421  sts = run_single();
422  }
423 
424  destroy_threads();
425 
426  return sts;
427 }

436 int pebbaction::run_single() {
437  RVSTRACE_
438  // define timers
439  rvs::timer<pebbaction> timer_running(&pebbaction::do_running_average, this);
440  rvs::timer<pebbaction> timer_final(&pebbaction::do_final_average, this);
441 
442  // let the test run
443  brun = true;
444 
445  unsigned int iter = gst_run_count > 0 ? gst_run_count : 1;
446  unsigned int step = gst_run_count == 0 ? 0 : 1;
447 
448  // start timers
449  if (gst_run_duration_ms) {
450  RVSTRACE_
451  timer_final.start(gst_run_duration_ms, true); // ticks only once
452  }
453 
454  if (prop_log_interval) {
455  RVSTRACE_
456  timer_running.start(prop_log_interval); // ticks continuously
457  }
458 
459  // iterate through test array and invoke tests one by one
460  do {
461  RVSTRACE_
462  for (auto it = test_array.begin(); brun && it != test_array.end(); ++it) {
463  RVSTRACE_
464  (*it)->do_transfer();
465 
466  // if log interval is zero, print current results immediately
467  if (prop_log_interval == 0) {
468  print_running_average(*it);
469  }
470  sleep(1);
471 
472  if (rvs::lp::Stopping()) {
473  RVSTRACE_
474  brun = false;
475  break;
476  }
477  }
478 
479  RVSTRACE_
480  iter -= step;
481 
482  // insert wait between runs if needed
483  if (iter > 0 && gst_run_wait_ms > 0) {
484  RVSTRACE_
485  sleep(gst_run_wait_ms);
486  }
487  } while (brun && iter);
488 
489  RVSTRACE_
490  timer_running.stop();
491  timer_final.stop();
492 
493  print_final_average();
494 
495  RVSTRACE_
496  return rvs::lp::Stopping() ? -1 : 0;
497 }

506 int pebbaction::run_parallel() {
507  RVSTRACE_
508  // define timers
509  rvs::timer<pebbaction> timer_running(&pebbaction::do_running_average, this);
510  rvs::timer<pebbaction> timer_final(&pebbaction::do_final_average, this);
511 
512  // let the test run
513  brun = true;
514 
515  // start all worker threads
516  for (auto it = test_array.begin(); it != test_array.end(); ++it) {
517  (*it)->start();
518  }
519 
520  // start timers
521  if (gst_run_duration_ms) {
522  timer_final.start(gst_run_duration_ms, true); // ticks only once
523  }
524 
525  if (prop_log_interval) {
526  timer_running.start(prop_log_interval); // ticks continuously
527  }
528 
529  // wait for test to complete
530  while (brun) {
531  if (rvs::lp::Stopping()) {
532  RVSTRACE_
533  brun = false;
534  }
535  sleep(1);
536  }
537 
538  timer_running.stop();
539  timer_final.stop();
540 
541  // signal all worker threads to stop
542  for (auto it = test_array.begin(); it != test_array.end(); ++it) {
543  (*it)->stop();
544  }
545  sleep(10);
546 
547  // join all worker threads
548  for (auto it = test_array.begin(); it != test_array.end(); ++it) {
549  (*it)->join();
550  }
551 
552  print_final_average();
553 
554  return rvs::lp::Stopping() ? -1 : 0;
555 }

580 int pebbaction::print_running_average(pebbworker* pWorker) {
581  int src_node, dst_node;
582  int dst_id;
583  bool bidir;
584  size_t current_size;
585  double duration;
586  std::string msg;
587  char buff[64];
588  double bandwidth;
589  uint16_t transfer_ix;
590  uint16_t transfer_num;
591 
592  // get running average
593  pWorker->get_running_data(&src_node, &dst_node, &bidir,
594  &current_size, &duration);
595 
596  if (duration > 0) {
597  bandwidth = current_size/duration/(1024*1024*1024);
598  if (bidir) {
599  bandwidth *=2;
600  }
601  snprintf( buff, sizeof(buff), "%.3f GBps", bandwidth);
602  } else {
603  // no running average in this iteration, try getting total so far
604  // (do not reset final totals as this is just intermediate query)
605  pWorker->get_final_data(&src_node, &dst_node, &bidir,
606  &current_size, &duration, false);
607  if (duration > 0) {
608  bandwidth = current_size/duration/(1024*1024*1024);
609  if (bidir) {
610  bandwidth *=2;
611  }
612  snprintf( buff, sizeof(buff), "%.3f GBps (*)", bandwidth);
613  } else {
614  // not transfers at all - print "pending"
615  snprintf( buff, sizeof(buff), "(pending)");
616  }
617  }
618 
619  dst_id = rvs::gpulist::GetGpuIdFromNodeId(dst_node);
620  transfer_ix = pWorker->get_transfer_ix();
621  transfer_num = pWorker->get_transfer_num();
622 
623  msg = "[" + action_name + "] pcie-bandwidth ["
624  + std::to_string(transfer_ix) + "/" + std::to_string(transfer_num)
625  + "] "
626  + std::to_string(src_node) + " " + std::to_string(dst_id)
627  + " h2d: " + (prop_h2d ? "true" : "false")
628  + " d2h: " + (prop_d2h ? "true" : "false") + " "
629  + buff;
630 
631  rvs::lp::Log(msg, rvs::loginfo);
632 
633  if (bjson) {
634  unsigned int sec;
635  unsigned int usec;
636  rvs::lp::get_ticks(&sec, &usec);
637  void* pjson = rvs::lp::LogRecordCreate(MODULE_NAME,
638  action_name.c_str(), rvs::loginfo, sec, usec);
639  if (pjson != NULL) {
640  rvs::lp::AddString(pjson,
641  "transfer_ix", std::to_string(transfer_ix));
642  rvs::lp::AddString(pjson,
643  "transfer_num", std::to_string(transfer_num));
644  rvs::lp::AddString(pjson, "src", std::to_string(src_node));
645  rvs::lp::AddString(pjson, "dst", std::to_string(dst_id));
646  rvs::lp::AddString(pjson, "pcie-bandwidth (GBps)", buff);
647  rvs::lp::LogRecordFlush(pjson);
648  }
649  }
650 
651  return 0;
652 }
