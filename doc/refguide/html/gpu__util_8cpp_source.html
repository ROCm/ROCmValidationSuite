58 int gpu_num_subdirs(const char* dirpath, const char* prefix) {
59  int count = 0;
60  DIR *dirp;
61  struct dirent *dir;
62  int prefix_len = strlen(prefix);
63 
64  dirp = opendir(dirpath);
65  if (dirp) {
66  while ((dir = readdir(dirp)) != 0) {
67  if ((strcmp(dir->d_name, ".") == 0) ||
68  (strcmp(dir->d_name, "..") == 0))
69  continue;
70  if (prefix_len &&
71  strncmp(dir->d_name, prefix, prefix_len))
72  continue;
73  count++;
74  }
75  closedir(dirp);
76  }
77  return count;
78 }

85 void gpu_get_all_location_id(std::vector<uint16_t>* pgpus_location_id) {
86  ifstream f_id, f_prop;
87  char path[KFD_PATH_MAX_LENGTH];
88 
89  std::string prop_name;
90  int gpu_id;
91  uint32_t prop_val;
92 
93 
94  // Discover the number of nodes: Inside nodes folder there are only folders
95  // that represent the node number
96  int num_nodes = gpu_num_subdirs(KFD_SYS_PATH_NODES, "");
97 
98  // get all GPUs device id
99  for (int node_id = 0; node_id < num_nodes; node_id++) {
100  snprintf(path, KFD_PATH_MAX_LENGTH, "%s/%d/gpu_id", KFD_SYS_PATH_NODES,
101  node_id);
102  f_id.open(path);
103  snprintf(path, KFD_PATH_MAX_LENGTH, "%s/%d/properties",
104  KFD_SYS_PATH_NODES, node_id);
105  f_prop.open(path);
106 
107  f_id >> gpu_id;
108  if (gpu_id != 0) {
109  while (f_prop >> prop_name) {
110  if (prop_name == "location_id") {
111  f_prop >> prop_val;
112  (*pgpus_location_id).push_back(prop_val);
113  break;
114  }
115  }
116  }
117  f_id.close();
118  f_prop.close();
119  }
120 }
